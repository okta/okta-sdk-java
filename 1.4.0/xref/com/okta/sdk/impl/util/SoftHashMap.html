23  import java.lang.ref.ReferenceQueue;
24  import java.lang.ref.SoftReference;

34  import java.util.concurrent.ConcurrentLinkedQueue;
35  import java.util.concurrent.locks.ReentrantLock;

38  /**
39   * A <code><em>Soft</em>HashMap</code> is a memory-constrained map that stores its <em>values</em> in
40   * {@link SoftReference SoftReference}s.  (Contrast this with the JDK's
41   * {@link WeakHashMap WeakHashMap}, which uses weak references for its <em>keys</em>, which is of little value if you
42   * want the cache to auto-resize itself based on memory constraints).
43   * <p>
44   * Having the values wrapped by soft references allows the cache to automatically reduce its size based on memory
45   * limitations and garbage collection.  This ensures that the cache will not cause memory leaks by holding strong
46   * references to all of its values.
47   * <p>
48   * This class is a generics-enabled Map based on initial ideas from Heinz Kabutz's and Sydney Redelinghuys's
49   * <a href="http://www.javaspecialists.eu/archive/Issue015.html">publicly posted version (with their approval)</a>, with
50   * continued modifications.  It was copied from the <a href="http://shiro.apache.org">Apache Shiro</a> framework.
51   * <p>
52   * This implementation is thread-safe and usable in concurrent environments.
53   *
54   * @since 0.5.0
55   * @see SoftReference
56   * @see <a href="http://shiro.apache.org">Apache Shiro</a>
57   */
58  public class SoftHashMap<K, V> implements Map<K, V> {
59  
60      /**
61       * The default value of the RETENTION_SIZE attribute, equal to 100.
62       */
63      private static final int DEFAULT_RETENTION_SIZE = 100;
64  
65      /**
66       * The internal HashMap that will hold the SoftReference.
67       */
68      private final Map<K, SoftValue<V, K>> map;
69  
70      /**
71       * The number of strong references to hold internally, that is, the number of instances to prevent
72       * from being garbage collected automatically (unlike other soft references).
73       */
74      private final int RETENTION_SIZE;
75  
76      /**
77       * The FIFO list of strong references (not to be garbage collected), order of last access.
78       */
79      private final Queue<V> strongReferences; //guarded by 'strongReferencesLock'
80      private final ReentrantLock strongReferencesLock;
81  
82      /**
83       * Reference queue for cleared SoftReference objects.
84       */
85      private final ReferenceQueue<? super V> queue;
86  
87      /**
88       * Creates a new SoftHashMap with a default retention size size of
89       * {@link #DEFAULT_RETENTION_SIZE DEFAULT_RETENTION_SIZE} (100 entries).
90       *
91       * @see #SoftHashMap(int)
92       */
93      public SoftHashMap() {
94          this(DEFAULT_RETENTION_SIZE);
95      }
96  
97      /**
98       * Creates a new SoftHashMap with the specified retention size.
99       * <p>
100      * The retention size (n) is the total number of most recent entries in the map that will be strongly referenced
101      * (ie 'retained') to prevent them from being eagerly garbage collected.  That is, the point of a SoftHashMap is to
102      * allow the garbage collector to remove as many entries from this map as it desires, but there will always be (n)
103      * elements retained after a GC due to the strong references.
104      * <p>
105      * Note that in a highly concurrent environments the exact total number of strong references may differ slightly
106      * than the actual {@code retentionSize} value.  This number is intended to be a best-effort retention low
107      * water mark.
108      *
109      * @param retentionSize the total number of most recent entries in the map that will be strongly referenced
110      *                      (retained), preventing them from being eagerly garbage collected by the JVM.
111      */
112     @SuppressWarnings({"unchecked"})
113     public SoftHashMap(int retentionSize) {
114         super();
115         RETENTION_SIZE = Math.max(0, retentionSize);
116         queue = new ReferenceQueue<V>();
117         strongReferencesLock = new ReentrantLock();
118         map = new ConcurrentHashMap<K, SoftValue<V, K>>();
119         strongReferences = new ConcurrentLinkedQueue<V>();
120     }
121 
122     /**
123      * Creates a {@code SoftHashMap} backed by the specified {@code source}, with a default retention
124      * size of {@link #DEFAULT_RETENTION_SIZE DEFAULT_RETENTION_SIZE} (100 entries).
125      *
126      * @param source the backing map to populate this {@code SoftHashMap}
127      * @see #SoftHashMap(Map, int)
128      */
129     public SoftHashMap(Map<K, V> source) {
130         this(DEFAULT_RETENTION_SIZE);
131         putAll(source);
132     }
133 
134     /**
135      * Creates a {@code SoftHashMap} backed by the specified {@code source}, with the specified retention size.
136      * <p>
137      * The retention size (n) is the total number of most recent entries in the map that will be strongly referenced
138      * (ie 'retained') to prevent them from being eagerly garbage collected.  That is, the point of a SoftHashMap is to
139      * allow the garbage collector to remove as many entries from this map as it desires, but there will always be (n)
140      * elements retained after a GC due to the strong references.
141      * <p>
142      * Note that in a highly concurrent environments the exact total number of strong references may differ slightly
143      * than the actual {@code retentionSize} value.  This number is intended to be a best-effort retention low
144      * water mark.
145      *
146      * @param source        the backing map to populate this {@code SoftHashMap}
147      * @param retentionSize the total number of most recent entries in the map that will be strongly referenced
148      *                      (retained), preventing them from being eagerly garbage collected by the JVM.
149      */
150     public SoftHashMap(Map<K, V> source, int retentionSize) {
151         this(retentionSize);
152         putAll(source);
153     }
154 
155     public V get(Object key) {
156         processQueue();
157 
158         V result = null;
159         SoftValue<V, K> value = map.get(key);
160 
161         if (value != null) {
162             //unwrap the 'real' value from the SoftReference
163             result = value.get();
164             if (result == null) {
165                 //The wrapped value was garbage collected, so remove this entry from the backing map:
166                 //noinspection SuspiciousMethodCalls
167                 map.remove(key);
168             } else {
169                 //Add this value to the beginning of the strong reference queue (FIFO).
170                 addToStrongReferences(result);
171             }
172         }
173         return result;
174     }
175 
176     private void addToStrongReferences(V result) {
177         strongReferencesLock.lock();
178         try {
179             strongReferences.add(result);
180             trimStrongReferencesIfNecessary();
181         } finally {
182             strongReferencesLock.unlock();
183         }
184 
185     }
186 
187     //Guarded by the strongReferencesLock in the addToStrongReferences method
188 
189     private void trimStrongReferencesIfNecessary() {
190         //trim the strong ref queue if necessary:
191         while (strongReferences.size() > RETENTION_SIZE) {
192             strongReferences.poll();
193         }
194     }
195 
196     /**
197      * Traverses the ReferenceQueue and removes garbage-collected SoftValue objects from the backing map
198      * by looking them up using the SoftValue.key data member.
199      */
200     private void processQueue() {
201         SoftValue sv;
202         while ((sv = (SoftValue) queue.poll()) != null) {
203             //noinspection SuspiciousMethodCalls
204             map.remove(sv.key); // we can access private data!
205         }
206     }
207 
208     public boolean isEmpty() {
209         processQueue();
210         return map.isEmpty();
211     }
212 
213     public boolean containsKey(Object key) {
214         processQueue();
215         return map.containsKey(key);
216     }
217 
218     public boolean containsValue(Object value) {
219         processQueue();
220         Collection values = values();
221         return values != null && values.contains(value);
222     }
223 
224     public void putAll(Map<? extends K, ? extends V> m) {
225         if (m == null || m.isEmpty()) {
226             processQueue();
227             return;
228         }
229         for (Map.Entry<? extends K, ? extends V> entry : m.entrySet()) {
230             put(entry.getKey(), entry.getValue());
231         }
232     }
233 
234     public Set<K> keySet() {
235         processQueue();
236         return map.keySet();
237     }
238 
239     public Collection<V> values() {
240         processQueue();
241         Collection<K> keys = map.keySet();
242         if (keys.isEmpty()) {
243             //noinspection unchecked
244             return Collections.EMPTY_SET;
245         }
246         Collection<V> values = new ArrayList<V>(keys.size());
247         for (K key : keys) {
248             V v = get(key);
249             if (v != null) {
250                 values.add(v);
251             }
252         }
253         return values;
254     }
255 
256     /**
257      * Creates a new entry, but wraps the value in a SoftValue instance to enable auto garbage collection.
258      */
259     public V put(K key, V value) {
260         processQueue(); // throw out garbage collected values first
261         SoftValue<V, K> sv = new SoftValue<V, K>(value, key, queue);
262         SoftValue<V, K> previous = map.put(key, sv);
263         addToStrongReferences(value);
264         return previous != null ? previous.get() : null;
265     }
266 
267     public V remove(Object key) {
268         processQueue(); // throw out garbage collected values first
269         SoftValue<V, K> raw = map.remove(key);
270         return raw != null ? raw.get() : null;
271     }
272 
273     public void clear() {
274         strongReferencesLock.lock();
275         try {
276             strongReferences.clear();
277         } finally {
278             strongReferencesLock.unlock();
279         }
280         processQueue(); // throw out garbage collected values
281         map.clear();
282     }
283 
284     public int size() {
285         processQueue(); // throw out garbage collected values first
286         return map.size();
287     }
288 
289     public Set<Map.Entry<K, V>> entrySet() {
290         processQueue(); // throw out garbage collected values first
291         Collection<K> keys = map.keySet();
292         if (keys.isEmpty()) {
293             //noinspection unchecked
294             return Collections.EMPTY_SET;
295         }
296 
297         Map<K, V> kvPairs = new HashMap<K, V>(keys.size());
298         for (K key : keys) {
299             V v = get(key);
300             if (v != null) {
301                 kvPairs.put(key, v);
302             }
303         }
304         return kvPairs.entrySet();
305     }
306 
307     /**
308      * We define our own subclass of SoftReference which contains
309      * not only the value but also the key to make it easier to find
310      * the entry in the HashMap after it's been garbage collected.
311      */
312     private static class SoftValue<V, K> extends SoftReference<V> {
313 
314         private final K key;
315 
316         /**
317          * Constructs a new instance, wrapping the value, key, and queue, as
318          * required by the superclass.
319          *
320          * @param value the map value
321          * @param key   the map key
322          * @param queue the soft reference queue to poll to determine if the entry had been reaped by the GC.
323          */
324         private SoftValue(V value, K key, ReferenceQueue<? super V> queue) {
325             super(value, queue);
326             this.key = key;
327         }
328 
329     }
