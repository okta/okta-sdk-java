36      /**
37       * Maximum exponential back-off time before retrying a request
38       */
39      private static final int DEFAULT_MAX_BACKOFF_IN_MILLISECONDS = 20 * 1000;
40  
41      private static final int DEFAULT_MAX_RETRIES = 4;
42  
43      private int maxRetries = DEFAULT_MAX_RETRIES;

47      private BackoffStrategy backoffStrategy;

49      private final RequestExecutor delegate;

51      public RetryRequestExecutor(ClientConfiguration clientConfiguration, RequestExecutor delegate) {
52          this.delegate = delegate;
53  
54          if (clientConfiguration.getRetryMaxElapsed() >= 0) {
55              maxElapsedMillis = clientConfiguration.getRetryMaxElapsed() * 1000;
56          }
57  
58          if (clientConfiguration.getRetryMaxAttempts() > 0) {
59              maxRetries = clientConfiguration.getRetryMaxAttempts();
60          }
61      }

63      @Override
64      public Response executeRequest(Request request) throws RestException {
65  
66          Assert.notNull(request, "Request argument cannot be null.");
67  
68          int retryCount = 0;
69          Response response = null;
70          String requestId = null;
71          Timer timer = new Timer();
72  
73          // Make a copy of the original request params and headers so that we can
74          // permute them in the loop and start over with the original every time.
75          QueryString originalQuery = new QueryString();
76          originalQuery.putAll(request.getQueryString());
77  
78          HttpHeaders originalHeaders = new HttpHeaders();
79          originalHeaders.putAll(request.getHeaders());
80  
81          while (true) {
82  
83              try {
84  
85                  if (retryCount > 0) {
86                      request.setQueryString(originalQuery);
87                      request.setHeaders(originalHeaders);
88  
89                      // remember the request-id header if we need to retry
90                      if (requestId == null) {
91                          requestId = getRequestId(response);
92                      }
93  
94                      InputStream content = request.getBody();
95                      if (content != null && content.markSupported()) {
96                          content.reset();
97                      }
98  
99                      try {
100                         // if we cannot pause, then return the original response
101                         pauseBeforeRetry(retryCount, response, timer.split());
102                     } catch (RestException e) {
103                         if (log.isDebugEnabled()) {
104                             log.warn("Unable to pause for retry: {}", e.getMessage(), e);
105                         } else {
106                             log.warn("Unable to pause for retry: {}", e.getMessage());
107                         }
108 
109                         return response;
110                     }
111                 }
112 
113                 retryCount++;
114 
115                 // include X-Okta headers when retrying
116                 setOktaHeaders(request, requestId, retryCount);
117 
118                 response = doExecuteRequest(request);
119 
120                 //allow the loop to continue to execute a retry request
121                 if (!shouldRetry(response, retryCount, timer.split())) {
122                     return response;
123                 }
124 
125             } catch (SocketException | SocketTimeoutException e) { // known generic retryable exceptions
126                 if (!shouldRetry(retryCount, timer.split())) {
127                     throw new RestException("Unable to execute HTTP request: " + e.getMessage(), e);
128                 }
129                 log.debug("Retrying on {}: {}", e.getClass().getName(), e.getMessage());
130             } catch (RestException e) {
131                 // exceptions from delegate marked as retrHttpClientRequestExecutoryable
132                 if (!e.isRetryable() || !shouldRetry(retryCount, timer.split())) {
133                     throw e;
134                 }
135             } catch (Exception e) {
136                 throw new RestException("Unable to execute HTTP request: " + e.getMessage(), e);
137             }
138         }
139     }
140 
141     // exposed to allow HttpClientRequestExecutor to be backward compatible
142     // do NOT use directly
143     protected Response doExecuteRequest(Request request) {
144         return delegate.executeRequest(request);
145     }

147     /**
148      * Exponential sleep on failed request to avoid flooding a service with
149      * retries.
150      *
151      * @param retries           Current retry count.
152      */
153     private void pauseBeforeRetry(int retries, Response response, long timeElapsed) throws RestException {
154         long delay = -1;
155         long timeElapsedLeft = maxElapsedMillis - timeElapsed;
156 
157         // check before continuing
158         if (!shouldRetry(retries, timeElapsed)) {
159             throw failedToRetry();
160         }
161 
162         if (backoffStrategy != null) {
163             delay = Math.min(this.backoffStrategy.getDelayMillis(retries), timeElapsedLeft);
164         } else if (response != null && response.getHttpStatus() == 429) {
165             delay = get429DelayMillis(response);
166             if (!shouldRetry(retries, timeElapsed + delay)) {
167                 throw failedToRetry();
168             }
169             log.debug("429 detected, will retry in {}ms, attempt number: {}", delay, retries);
170         }
171 
172         // default / fallback strategy (backwards compatible implementation)
173         if (delay < 0) {
174             delay = Math.min(getDefaultDelayMillis(retries), timeElapsedLeft);
175         }
176 
177         // this shouldn't happen, but guard against a negative delay at this point
178         if (delay < 0) {
179             throw failedToRetry();
180         }
181 
182         log.debug("Retryable condition detected, will retry in {}ms, attempt number: {}", delay, retries);
183 
184         try {
185             Thread.sleep(delay);
186         } catch (InterruptedException e) {
187             Thread.currentThread().interrupt();
188             throw new RestException(e.getMessage(), e);
189         }
190     }
191 
192     private long get429DelayMillis(Response response) {
193 
194         // the time at which the rate limit will reset, specified in UTC epoch time.
195         String resetLimit = getOnlySingleHeaderValue(response,"X-Rate-Limit-Reset");
196         if (resetLimit == null || !resetLimit.chars().allMatch(Character::isDigit)) {
197             return -1;
198         }
199 
200         // If the Date header is not set, do not continue
201         Date requestDate = dateFromHeader(response);
202         if (requestDate == null) {
203             return -1;
204         }
205 
206         long waitUntil = Long.parseLong(resetLimit) * 1000L;
207         long requestTime = requestDate.getTime();
208         long delay = waitUntil - requestTime + 1000;
209         log.debug("429 wait: {} - {} + {} = {}", waitUntil, requestTime, 1000, delay);
210 
211         return delay;
212     }
213 
214     private Date dateFromHeader(Response response) {
215         Date result = null;
216         long dateLong = response.getHeaders().getDate();
217         if (dateLong > 0) {
218             result = new Date(dateLong);
219         }
220         return result;
221     }
222 
223     private long getDefaultDelayMillis(int retries) {
224         long scaleFactor = 300;
225         long result = (long) (Math.pow(2, retries) * scaleFactor);
226         return Math.min(result, DEFAULT_MAX_BACKOFF_IN_MILLISECONDS);
227     }
228 
229     private boolean shouldRetry(int retryCount, long timeElapsed) {
230                // either maxRetries or maxElapsedMillis is enabled
231         return (maxRetries > 0 || maxElapsedMillis > 0)
232 
233                // maxRetries count is disabled OR if set check it
234                && (maxRetries <= 0 || retryCount <= this.maxRetries)
235 
236                // maxElapsedMillis is disabled OR if set check it
237                && (maxElapsedMillis <= 0 || timeElapsed < maxElapsedMillis);
238     }
239 
240     private boolean shouldRetry(Response response, int retryCount, long timeElapsed) {
241         int httpStatus = response.getHttpStatus();
242 
243         // supported status codes
244         return shouldRetry(retryCount, timeElapsed)
245             && (httpStatus == 429
246              || httpStatus == 503
247              || httpStatus == 504);
248     }
249 
250     private RestException failedToRetry() {
251         return new RestException("Cannot retry request, next request will exceed retry configuration.");
252     }
253 
254     private String getOnlySingleHeaderValue(Response response, String name) {
255 
256         if (response.getHeaders() != null) {
257             List<String> values = response.getHeaders().get(name);
258             if (!Collections.isEmpty(values) && values.size() == 1) {
259                 return values.get(0);
260             }
261         }
262         return null;
263     }
264 
265     private String getRequestId(Response response) {
266         if (response != null) {
267             return response.getHeaders().getFirst("X-Okta-Request-Id");
268         }
269         return null;
270     }

272     /**
273      * Adds {@code X-Okta-Retry-For} and {@code X-Okta-Retry-Count} headers to request if not null/empty or zero.
274      *
275      * @param request the request to add headers too
276      * @param requestId request ID of the original request that failed
277      * @param retryCount the number of times the request has been retried
278      */
279     private void setOktaHeaders(Request request, String requestId, int retryCount) {
280         if (Strings.hasText(requestId)) {
281             request.getHeaders().add("X-Okta-Retry-For", requestId);
282         }
283         if (retryCount > 1) {
284             request.getHeaders().add("X-Okta-Retry-Count", Integer.toString(retryCount));
285         }
286     }

318     private static class Timer {
319 
320         private long startTime = System.currentTimeMillis();
321 
322         long split() {
323             return System.currentTimeMillis() - startTime;
324         }
