60  /**
61   * Implementation of {@link AccessTokenRetrieverService} interface.
62   * This has logic to fetch OAuth2 access token from the Authorization server endpoint.
63   * @since 1.6.0
64   */
65  public class AccessTokenRetrieverServiceImpl implements AccessTokenRetrieverService {
66      private static final Logger log = LoggerFactory.getLogger(AccessTokenRetrieverServiceImpl.class);
67  
68      static final String TOKEN_URI  = "/oauth2/v1/token";
69  
70      private static final KeyPair DUMMY_KEY_PAIR = Jwts.SIG.RS256.keyPair().build();
71  
72      /**
73       * Custom SecureDigestAlgorithm that delegates signature to the jwtSigner in tokenClientConfiguration
74       */
75      private class CustomJwtSigningAlgorithm implements SecureDigestAlgorithm<PrivateKey, Key> {
76          @Override
77          public byte[] digest(SecureRequest<InputStream, PrivateKey> request) throws SecurityException {
78              try {
79                  byte[] bytes = readAllBytes(request.getPayload());
80                  return tokenClientConfiguration.getJwtSigner().apply(bytes);
81              } catch (IOException e) {
82                  throw new RuntimeException(e);
83              }
84          }
85  
86          //to replace with InputStream.readAllBytes after migrating to Java 9+
87          private byte[] readAllBytes(InputStream payload) throws IOException {
88              ByteArrayOutputStream buffer = new ByteArrayOutputStream();
89              int nRead;
90              byte[] data = new byte[16384];
91              while ((nRead = payload.read(data, 0, data.length)) != -1) {
92                  buffer.write(data, 0, nRead);
93              }
94              return buffer.toByteArray();
95          }
96  
97          @Override
98          public boolean verify(VerifySecureDigestRequest<Key> request) throws SecurityException {
99              //no need to verify JWTs
100             throw new UnsupportedOperationException();
101         }
102 
103         @Override
104         public String getId() {
105             return tokenClientConfiguration.getJwtSigningAlgorithm();
106         }
107     }
108 
109     private final ClientConfiguration tokenClientConfiguration;
110     private final ApiClient apiClient;
111     private final CustomJwtSigningAlgorithm customJwtSigningAlgorithm = new CustomJwtSigningAlgorithm();
112 
113     public AccessTokenRetrieverServiceImpl(ClientConfiguration apiClientConfiguration, ApiClient apiClient) {
114         Assert.notNull(apiClientConfiguration, "apiClientConfiguration must not be null.");
115         Assert.notNull(apiClient, "apiClient must not be null.");
116         this.apiClient = apiClient;
117         this.tokenClientConfiguration = constructTokenClientConfig(apiClientConfiguration);
118     }

120     /**
121      * {@inheritDoc}
122      */
123     @Override
124     public OAuth2AccessToken getOAuth2AccessToken() throws IOException, InvalidKeyException, OAuth2TokenRetrieverException {
125         log.debug("Attempting to get OAuth2 access token for client id {} from {}",
126             tokenClientConfiguration.getClientId(), tokenClientConfiguration.getBaseUrl() + TOKEN_URI);
127 
128         String signedJwt = createSignedJWT();
129         String scope = String.join(" ", tokenClientConfiguration.getScopes());
130 
131         try {
132             Map<String, Object> formParameters = new HashMap<>();
133             formParameters.put("grant_type", "client_credentials");
134             formParameters.put("client_assertion_type", "urn:ietf:params:oauth:client-assertion-type:jwt-bearer");
135             formParameters.put("client_assertion", signedJwt);
136             formParameters.put("scope", scope);
137 
138             OAuth2AccessToken oAuth2AccessToken = apiClient.invokeAPI(
139                 TOKEN_URI,
140                 HttpMethod.POST.name(),
141                 new LinkedList<>(),
142                 new LinkedList<>(),
143                 null,
144                 null,
145                 new LinkedHashMap<>(),
146                 new LinkedHashMap<>(),
147                 formParameters,
148                 MediaType.APPLICATION_JSON_VALUE,
149                 MediaType.APPLICATION_FORM_URLENCODED_VALUE,
150                 new String[] { "oauth2" },
151                 new TypeReference<OAuth2AccessToken>() {});
152 
153             log.debug("Got OAuth2 access token for client id {} from {}",
154                 tokenClientConfiguration.getClientId(), tokenClientConfiguration.getBaseUrl() + TOKEN_URI);
155 
156             apiClient.setAccessToken(oAuth2AccessToken.getAccessToken());
157 
158             return oAuth2AccessToken;
159         } catch (DPoPHandshakeException e) {
160             if (e.continueHandshake) {
161                 return getOAuth2AccessToken();
162             }
163             throw new OAuth2HttpException(e.getMessage(), e, false);
164         } catch (ApiException e) {
165             throw new OAuth2HttpException(e.getMessage(), e, e.getCode() == 401);
166         } catch (Exception e) {
167             throw new OAuth2TokenRetrieverException("Exception while trying to get " +
168                 "OAuth2 access token for client id " + tokenClientConfiguration.getClientId(), e);
169         }
170     }

172     /**
173      * Create signed JWT string with the supplied token client configuration details.
174      * Expiration value should be not more than one hour in the future.
175      * We use 50 minutes in order to have a 10 minutes leeway in case of clock skew.
176      *
177      * @return signed JWT string
178      * @throws InvalidKeyException if the supplied key is invalid
179      * @throws IOException if the key could not be read
180      */
181     String createSignedJWT() throws InvalidKeyException, IOException {
182         String clientId = tokenClientConfiguration.getClientId();
183         Instant now = Instant.now();
184 
185         JwtBuilder builder = Jwts.builder()
186             .audience().add(tokenClientConfiguration.getBaseUrl() + TOKEN_URI).and()
187             .issuedAt(Date.from(now))
188             .expiration(Date.from(now.plus(50, ChronoUnit.MINUTES)))             // see Javadoc
189             .issuer(clientId)
190             .subject(clientId)
191             .claim("jti", UUID.randomUUID().toString());
192 
193         if (tokenClientConfiguration.hasCustomJwtSigner()) {
194             //JwtBuilder requires a key to be passed, even if it's actually not used by the algorithm
195             builder.signWith(DUMMY_KEY_PAIR.getPrivate(), customJwtSigningAlgorithm);
196         } else {
197             builder = builder.signWith(parsePrivateKey(getPemReader()));
198         }
199 
200         if (Strings.hasText(tokenClientConfiguration.getKid())) {
201             builder.header().add("kid", tokenClientConfiguration.getKid());
202         }
203 
204         return builder.compact();
205     }

207     /**
208      * Parse private key from the supplied configuration.
209      *
210      * @param pemReader a {@link Reader} that has access to a full PEM resource
211      * @return {@link PrivateKey}
212      * @throws IOException if the private key could not be read
213      * @throws InvalidKeyException if the supplied key is invalid
214      */
215     PrivateKey parsePrivateKey(Reader pemReader) throws IOException, InvalidKeyException {
216 
217         PrivateKey privateKey = getPrivateKeyFromPEM(pemReader);
218         String algorithm = privateKey.getAlgorithm();
219 
220         if (!algorithm.equals("RSA") &&
221             !algorithm.equals("EC")) {
222             throw new InvalidKeyException("Supplied privateKey is not an RSA or EC key - " + algorithm);
223         }
224 
225         return privateKey;
226     }
227 
228     private Reader getPemReader() throws IOException {
229         String privateKey = tokenClientConfiguration.getPrivateKey();
230         if (ConfigUtil.hasPrivateKeyContentWrapper(privateKey)) {
231             return new StringReader(privateKey);
232         } else {
233             return Files.newBufferedReader(Paths.get(privateKey), Charset.defaultCharset());
234         }
235     }
236 
237     /**
238      * Get Private key from input PEM file.
239      *
240      * @param reader the reader instance
241      * @return {@link PrivateKey} private key instance
242      * @throws IOException if the parser could not read the reader object
243      */
244     PrivateKey getPrivateKeyFromPEM(Reader reader) throws IOException {
245         PrivateKey privateKey;
246 
247         try (PEMParser pemParser = new PEMParser(reader)) {
248             JcaPEMKeyConverter jcaPEMKeyConverter = new JcaPEMKeyConverter();
249             Object pemContent = pemParser.readObject();
250 
251             if (pemContent == null) {
252                 throw new IllegalArgumentException("Invalid Private Key PEM file");
253             }
254 
255             if (pemContent instanceof PEMKeyPair) {
256                 PEMKeyPair pemKeyPair = (PEMKeyPair) pemContent;
257                 KeyPair keyPair = jcaPEMKeyConverter.getKeyPair(pemKeyPair);
258                 privateKey = keyPair.getPrivate();
259             } else if (pemContent instanceof PrivateKeyInfo) {
260                 PrivateKeyInfo privateKeyInfo = (PrivateKeyInfo) pemContent;
261                 privateKey = jcaPEMKeyConverter.getPrivateKey(privateKeyInfo);
262             } else {
263                 throw new IllegalArgumentException("Unsupported Private Key format '" +
264                     pemContent.getClass().getSimpleName() + '"');
265             }
266         }
267 
268         return privateKey;
269     }

271     /**
272      * Create token client config from the supplied API client config.
273      * Token client needs to retry http 401 errors only once which is not the case with the API client.
274      * We therefore effect this token client specific config by setting 'retryMaxElapsed'
275      * & 'retryMaxAttempts' fields.
276      *
277      * @param apiClientConfiguration supplied in the client configuration.
278      * @return ClientConfiguration to be used by token retrieval client.
279      */
280     ClientConfiguration constructTokenClientConfig(ClientConfiguration apiClientConfiguration) {
281         ClientConfiguration tokenClientConfiguration = new ClientConfiguration();
282 
283         tokenClientConfiguration.setClientCredentialsResolver(
284             new DefaultClientCredentialsResolver(Optional::empty));
285 
286         tokenClientConfiguration.setRequestAuthenticator(new DisabledAuthenticator());
287 
288         // TODO: set this to false explicitly when caching is implemented in OASv3 SDK
289         //tokenClientConfiguration.setCacheManagerEnabled(false);
290 
291         if (apiClientConfiguration.getBaseUrlResolver() != null)
292             tokenClientConfiguration.setBaseUrlResolver(apiClientConfiguration.getBaseUrlResolver());
293 
294         if (apiClientConfiguration.getProxy() != null)
295             tokenClientConfiguration.setProxy(apiClientConfiguration.getProxy());
296 
297         tokenClientConfiguration.setBaseUrl(apiClientConfiguration.getBaseUrl());
298         tokenClientConfiguration.setAuthenticationScheme(AuthenticationScheme.OAUTH2_PRIVATE_KEY);
299         tokenClientConfiguration.setAuthorizationMode(AuthorizationMode.get(tokenClientConfiguration.getAuthenticationScheme()));
300         tokenClientConfiguration.setClientId(apiClientConfiguration.getClientId());
301         tokenClientConfiguration.setScopes(apiClientConfiguration.getScopes());
302         tokenClientConfiguration.setPrivateKey(apiClientConfiguration.getPrivateKey());
303         tokenClientConfiguration.setJwtSigner(apiClientConfiguration.getJwtSigner(), apiClientConfiguration.getJwtSigningAlgorithm());
304         tokenClientConfiguration.setKid(apiClientConfiguration.getKid());
305 
306         // setting this to '0' will disable this check and only 'retryMaxAttempts' will be effective
307         tokenClientConfiguration.setRetryMaxElapsed(0);
308         // retry only once for token requests (http 401 errors)
309         tokenClientConfiguration.setRetryMaxAttempts(1);
310 
311         return tokenClientConfiguration;
312     }
