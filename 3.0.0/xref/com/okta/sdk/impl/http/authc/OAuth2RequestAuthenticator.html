32  /**
33   * This implementation used by OAuth2 flow adds Bearer header with access token as the
34   * value in all outgoing requests. This has logic to fetch a new token and store it in
35   * {@link OAuth2ClientCredentials} object if the existing one has expired.
36   *
37   * @since 1.6.0
38   */
39  public class OAuth2RequestAuthenticator implements RequestAuthenticator {
40      private static final Logger log = LoggerFactory.getLogger(OAuth2RequestAuthenticator.class);
41  
42      private final ClientCredentials<OAuth2AccessToken> clientCredentials;
43  
44      public OAuth2RequestAuthenticator(ClientCredentials<OAuth2AccessToken> clientCredentials) {
45          Assert.notNull(clientCredentials, "clientCredentials may not be null");
46          this.clientCredentials = clientCredentials;
47      }
48  
49      @Override
50      public void authenticate(Request request) throws RequestAuthenticationException {
51          OAuth2AccessToken oAuth2AccessToken = clientCredentials.getCredentials();
52  
53          if (oAuth2AccessToken.hasExpired()) {
54              log.debug("OAuth2 access token expiry detected. Will fetch a new token from Authorization server");
55  
56              synchronized (this) {
57                  if (oAuth2AccessToken.hasExpired()) {
58                      try {
59                          OAuth2ClientCredentialsl/oauth2/OAuth2ClientCredentials.html#OAuth2ClientCredentials">OAuth2ClientCredentials oAuth2ClientCredentials = (OAuth2ClientCredentials) clientCredentials;
60                          // fetch new token
61                          oAuth2AccessToken = oAuth2ClientCredentials.getAccessTokenRetrieverService().getOAuth2AccessToken();
62                          // store the new token
63                          oAuth2ClientCredentials.setCredentials(oAuth2AccessToken);
64                      } catch (IOException | InvalidKeyException e) {
65                          throw new OAuth2TokenRetrieverException("Failed to renew expired OAuth2 access token", e);
66                      }
67                  }
68              }
69          }
70  
71          // add Bearer header with token value
72          request.getHeaders().set(AUTHORIZATION_HEADER, "Bearer " + oAuth2AccessToken.getAccessToken());
73      }

75  }
