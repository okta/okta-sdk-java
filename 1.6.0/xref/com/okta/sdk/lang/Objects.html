60      /**
61       * Return whether the given throwable is a checked exception:
62       * that is, neither a RuntimeException nor an Error.
63       * @param ex the throwable to check
64       * @return whether the throwable is a checked exception
65       * @see java.lang.Exception
66       * @see java.lang.RuntimeException
67       * @see java.lang.Error
68       */
69      public static boolean isCheckedException(Throwable ex) {
70          return !(ex instanceof RuntimeException || ex instanceof Error);
71      }

73      /**
74       * Check whether the given exception is compatible with the specified
75       * exception types, as declared in a throws clause.
76       * @param ex the exception to check
77       * @param declaredExceptions the exception types declared in the throws clause
78       * @return whether the given exception is compatible
79       */
80      public static boolean isCompatibleWithThrowsClause(Throwable ex, Class<?>... declaredExceptions) {
81          if (!isCheckedException(ex)) {
82              return true;
83          }
84          if (declaredExceptions != null) {
85              for (Class<?> declaredException : declaredExceptions) {
86                  if (declaredException.isInstance(ex)) {
87                      return true;
88                  }
89              }
90          }
91          return false;
92      }

113     /**
114      * Determine whether the given object is empty.
115      * <p>This method supports the following object types.
116      * <ul>
117      * <li>{@code Array}: considered empty if its length is zero</li>
118      * <li>{@link CharSequence}: considered empty if its length is zero</li>
119      * <li>{@link Collection}: delegates to {@link Collection#isEmpty()}</li>
120      * <li>{@link Map}: delegates to {@link Map#isEmpty()}</li>
121      * </ul>
122      * <p>If the given object is non-null and not one of the aforementioned
123      * supported types, this method returns {@code false}.
124      * @param obj the object to check
125      * @return {@code true} if the object is {@code null} or <em>empty</em>
126      * @since 4.2
127      * @see Objects#isEmpty(Object[])
128      * @see Strings#hasLength(CharSequence)
129      * @see Strings#isEmpty(Object)
130      * @see Collections#isEmpty(java.util.Collection)
131      * @see Collections#isEmpty(java.util.Map)
132      */
133     @SuppressWarnings("rawtypes")
134     public static boolean isEmpty(Object obj) {
135         if (obj == null) {
136             return true;
137         }
138 
139         if (obj instanceof CharSequence) {
140             return ((CharSequence) obj).length() == 0;
141         }
142         if (obj.getClass().isArray()) {
143             return Array.getLength(obj) == 0;
144         }
145         if (obj instanceof Collection) {
146             return ((Collection) obj).isEmpty();
147         }
148         if (obj instanceof Map) {
149             return ((Map) obj).isEmpty();
150         }
151 
152         // else
153         return false;
154     }

186     /**
187      * Check whether the given array of enum constants contains a constant with the given name.
188      * @param enumValues the enum values to check, typically the product of a call to MyEnum.values()
189      * @param constant the constant name to find (must not be null or empty string)
190      * @param caseSensitive whether case is significant in determining a match
191      * @return whether the constant has been found in the given array
192      */
193     public static boolean containsConstant(Enum<?>[] enumValues, String constant, boolean caseSensitive) {
194         for (Enum<?> candidate : enumValues) {
195             if (caseSensitive ?
196                     candidate.toString().equals(constant) :
197                     candidate.toString().equalsIgnoreCase(constant)) {
198                 return true;
199             }
200         }
201         return false;
202     }

204     /**
205      * Case insensitive alternative to {@link Enum#valueOf(Class, String)}.
206      * @param <E> the concrete Enum type
207      * @param enumValues the array of all Enum constants in question, usually per Enum.values()
208      * @param constant the constant to get the enum value of
209      * @throws IllegalArgumentException if the given constant is not found in the given array
210      * of enum values. Use {@link #containsConstant(Enum[], String)} as a guard to avoid this exception.
211      */
212     public static <E extends Enum<?>> E caseInsensitiveValueOf(E[] enumValues, String constant) {
213         for (E candidate : enumValues) {
214             if (candidate.toString().equalsIgnoreCase(constant)) {
215                 return candidate;
216             }
217         }
218         throw new IllegalArgumentException(
219                 String.format("constant [%s] does not exist in enum type %s",
220                         constant, enumValues.getClass().getComponentType().getName()));
221     }

223     /**
224      * Append the given object to the given array, returning a new array
225      * consisting of the input array contents plus the given object.
226      * @param array the array to append to (can be {@code null})
227      * @param obj the object to append
228      * @return the new array (of the same component type; never {@code null})
229      */
230     public static <A, O extends A> A[] addObjectToArray(A[] array, O obj) {
231         Class<?> compType = Object.class;
232         if (array != null) {
233             compType = array.getClass().getComponentType();
234         }
235         else if (obj != null) {
236             compType = obj.getClass();
237         }
238         int newArrLength = (array != null ? array.length + 1 : 1);
239         @SuppressWarnings("unchecked")
240         A[] newArr = (A[]) Array.newInstance(compType, newArrLength);
241         if (array != null) {
242             System.arraycopy(array, 0, newArr, 0, array.length);
243         }
244         newArr[newArr.length - 1] = obj;
245         return newArr;
246     }

248     /**
249      * Convert the given array (which may be a primitive array) to an
250      * object array (if necessary of primitive wrapper objects).
251      * <p>A {@code null} source value will be converted to an
252      * empty Object array.
253      * @param source the (potentially primitive) array
254      * @return the corresponding object array (never {@code null})
255      * @throws IllegalArgumentException if the parameter is not an array
256      */
257     public static Object[] toObjectArray(Object source) {
258         if (source instanceof Object[]) {
259             return (Object[]) source;
260         }
261         if (source == null) {
262             return new Object[0];
263         }
264         if (!source.getClass().isArray()) {
265             throw new IllegalArgumentException("Source is not an array: " + source);
266         }
267         int length = Array.getLength(source);
268         if (length == 0) {
269             return new Object[0];
270         }
271         Class<?> wrapperType = Array.get(source, 0).getClass();
272         Object[] newArray = (Object[]) Array.newInstance(wrapperType, length);
273         for (int i = 0; i < length; i++) {
274             newArray[i] = Array.get(source, i);
275         }
276         return newArray;
277     }

284     /**
285      * Determine if the given objects are equal, returning {@code true} if
286      * both are {@code null} or {@code false} if only one is {@code null}.
287      * <p>Compares arrays with {@code Arrays.equals}, performing an equality
288      * check based on the array elements rather than the array reference.
289      * @param o1 first Object to compare
290      * @param o2 second Object to compare
291      * @return whether the given objects are equal
292      * @see Object#equals(Object)
293      * @see java.util.Arrays#equals
294      */
295     public static boolean nullSafeEquals(Object o1, Object o2) {
296         if (o1 == o2) {
297             return true;
298         }
299         if (o1 == null || o2 == null) {
300             return false;
301         }
302         if (o1.equals(o2)) {
303             return true;
304         }
305         if (o1.getClass().isArray() && o2.getClass().isArray()) {
306             return arrayEquals(o1, o2);
307         }
308         return false;
309     }

311     /**
312      * Compare the given arrays with {@code Arrays.equals}, performing an equality
313      * check based on the array elements rather than the array reference.
314      * @param o1 first array to compare
315      * @param o2 second array to compare
316      * @return whether the given objects are equal
317      * @see #nullSafeEquals(Object, Object)
318      * @see java.util.Arrays#equals
319      */
320     private static boolean arrayEquals(Object o1, Object o2) {
321         if (o1 instanceof Object[] && o2 instanceof Object[]) {
322             return Arrays.equals((Object[]) o1, (Object[]) o2);
323         }
324         if (o1 instanceof boolean[] && o2 instanceof boolean[]) {
325             return Arrays.equals((boolean[]) o1, (boolean[]) o2);
326         }
327         if (o1 instanceof byte[] && o2 instanceof byte[]) {
328             return Arrays.equals((byte[]) o1, (byte[]) o2);
329         }
330         if (o1 instanceof char[] && o2 instanceof char[]) {
331             return Arrays.equals((char[]) o1, (char[]) o2);
332         }
333         if (o1 instanceof double[] && o2 instanceof double[]) {
334             return Arrays.equals((double[]) o1, (double[]) o2);
335         }
336         if (o1 instanceof float[] && o2 instanceof float[]) {
337             return Arrays.equals((float[]) o1, (float[]) o2);
338         }
339         if (o1 instanceof int[] && o2 instanceof int[]) {
340             return Arrays.equals((int[]) o1, (int[]) o2);
341         }
342         if (o1 instanceof long[] && o2 instanceof long[]) {
343             return Arrays.equals((long[]) o1, (long[]) o2);
344         }
345         if (o1 instanceof short[] && o2 instanceof short[]) {
346             return Arrays.equals((short[]) o1, (short[]) o2);
347         }
348         return false;
349     }

351     /**
352      * Return as hash code for the given object; typically the value of
353      * {@code Object#hashCode()}}. If the object is an array,
354      * this method will delegate to any of the {@code nullSafeHashCode}
355      * methods for arrays in this class. If the object is {@code null},
356      * this method returns 0.
357      * @see Object#hashCode()
358      * @see #nullSafeHashCode(Object[])
359      * @see #nullSafeHashCode(boolean[])
360      * @see #nullSafeHashCode(byte[])
361      * @see #nullSafeHashCode(char[])
362      * @see #nullSafeHashCode(double[])
363      * @see #nullSafeHashCode(float[])
364      * @see #nullSafeHashCode(int[])
365      * @see #nullSafeHashCode(long[])
366      * @see #nullSafeHashCode(short[])
367      */
368     public static int nullSafeHashCode(Object obj) {
369         if (obj == null) {
370             return 0;
371         }
372         if (obj.getClass().isArray()) {
373             if (obj instanceof Object[]) {
374                 return nullSafeHashCode((Object[]) obj);
375             }
376             if (obj instanceof boolean[]) {
377                 return nullSafeHashCode((boolean[]) obj);
378             }
379             if (obj instanceof byte[]) {
380                 return nullSafeHashCode((byte[]) obj);
381             }
382             if (obj instanceof char[]) {
383                 return nullSafeHashCode((char[]) obj);
384             }
385             if (obj instanceof double[]) {
386                 return nullSafeHashCode((double[]) obj);
387             }
388             if (obj instanceof float[]) {
389                 return nullSafeHashCode((float[]) obj);
390             }
391             if (obj instanceof int[]) {
392                 return nullSafeHashCode((int[]) obj);
393             }
394             if (obj instanceof long[]) {
395                 return nullSafeHashCode((long[]) obj);
396             }
397             if (obj instanceof short[]) {
398                 return nullSafeHashCode((short[]) obj);
399             }
400         }
401         return obj.hashCode();
402     }

404     /**
405      * Return a hash code based on the contents of the specified array.
406      * If {@code array} is {@code null}, this method returns 0.
407      */
408     public static int nullSafeHashCode(Object[] array) {
409         if (array == null) {
410             return 0;
411         }
412         int hash = INITIAL_HASH;
413         for (Object element : array) {
414             hash = MULTIPLIER * hash + nullSafeHashCode(element);
415         }
416         return hash;
417     }

669     /**
670      * Return a String representation of the contents of the specified array.
671      * <p>The String representation consists of a list of the array's elements,
672      * enclosed in curly braces ({@code "{}"}). Adjacent elements are separated
673      * by the characters {@code ", "} (a comma followed by a space). Returns
674      * {@code "null"} if {@code array} is {@code null}.
675      * @param array the array to build a String representation for
676      * @return a String representation of {@code array}
677      */
678     public static String nullSafeToString(Object[] array) {
679         if (array == null) {
680             return NULL_STRING;
681         }
682         int length = array.length;
683         if (length == 0) {
684             return EMPTY_ARRAY;
685         }
686         StringBuilder sb = new StringBuilder();
687         for (int i = 0; i < length; i++) {
688             if (i == 0) {
689                 sb.append(ARRAY_START);
690             }
691             else {
692                 sb.append(ARRAY_ELEMENT_SEPARATOR);
693             }
694             sb.append(String.valueOf(array[i]));
695         }
696         sb.append(ARRAY_END);
697         return sb.toString();
698     }

700     /**
701      * Return a String representation of the contents of the specified array.
702      * <p>The String representation consists of a list of the array's elements,
703      * enclosed in curly braces ({@code "{}"}). Adjacent elements are separated
704      * by the characters {@code ", "} (a comma followed by a space). Returns
705      * {@code "null"} if {@code array} is {@code null}.
706      * @param array the array to build a String representation for
707      * @return a String representation of {@code array}
708      */
709     public static String nullSafeToString(boolean[] array) {
710         if (array == null) {
711             return NULL_STRING;
712         }
713         int length = array.length;
714         if (length == 0) {
715             return EMPTY_ARRAY;
716         }
