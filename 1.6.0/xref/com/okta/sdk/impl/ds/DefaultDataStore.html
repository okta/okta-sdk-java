185     @SuppressWarnings("unchecked")
186     private ResourceDataResult getResourceData(String href, Class<? extends Resource> clazz, Map<String,?> queryParameters, HttpHeaders httpHeaders) {
187 
188         Assert.hasText(href, "href argument cannot be null or empty.");
189         Assert.notNull(clazz, "Resource class argument cannot be null.");
190 
191         FilterChain chain = new DefaultFilterChain(this.filters, req -> {
192 
193             CanonicalUri uri = req.getUri();
194 
195             Request getRequest = new DefaultRequest(HttpMethod.GET, uri.getAbsolutePath(), uri.getQuery(), req.getHttpHeaders());
196             Response getResponse = execute(getRequest);
197             Map<String,?> body = getBody(getResponse);
198 
199             if (Collections.isEmpty(body)) {
200                 throw new IllegalStateException("Unable to obtain resource data from the API server or from cache.");
201             }
202 
203             return new DefaultResourceDataResult(req.getAction(), uri, req.getResourceClass(), (Map<String,Object>)body);
204         });
205 
206         CanonicalUri uri = canonicalize(href, queryParameters);
207         ResourceDataRequest req = new DefaultResourceDataRequest(ResourceAction.READ, uri, clazz, new HashMap<>(), httpHeaders);
208         return chain.filter(req);
209     }

211     private ResourceAction getPostAction(ResourceDataRequest request, Response response) {
212         int httpStatus = response.getHttpStatus();
213         if (httpStatus == 201) {
214             return ResourceAction.CREATE;
215         }
216         // TODO: verify this is an issue with Okta
217         //Fix for https://github.com/stormpath/stormpath-sdk-java/issues/403
218         if (httpStatus == 200) {
219             return ResourceAction.READ;
220         }
221 
222         return request.getAction();
223     }

271     @SuppressWarnings("unchecked")
272     private <T extends Resource, R extends Resource> R save(String href,
273                                                             final T resource,
274                                                             final T parentResource,
275                                                             HttpHeaders requestHeaders,
276                                                             final Class<? extends R> returnType,
277                                                             final QueryString qs,
278                                                             final boolean create) {
279         Assert.hasText(href, "href argument cannot be null or empty.");
280         Assert.notNull(resource, "resource argument cannot be null.");
281         Assert.notNull(returnType, "returnType class cannot be null.");
282         Assert.isInstanceOf(AbstractResource.class, resource);
283         Assert.isTrue(!CollectionResource.class.isAssignableFrom(resource.getClass()), "Collections cannot be persisted.");
284 
285         final CanonicalUri uri = canonicalize(href, qs);
286         final AbstractResourcekta/sdk/impl/resource/AbstractResource.html#AbstractResource">AbstractResource abstractResource = (AbstractResource) resource;
287         // Most Okta endpoints do not support partial update, we can revisit in the future.
288         final Map<String, Object> props = resourceConverter.convert(abstractResource, false);
289 
290         FilterChain chain = new DefaultFilterChain(this.filters, req -> {
291 
292             CanonicalUri uri1 = req.getUri();
293             String href1 = uri1.getAbsolutePath();
294             QueryString qs1 = uri1.getQuery();
295 
296             HttpHeaders httpHeaders = req.getHttpHeaders();
297 
298             // create == POST
299             HttpMethod method = HttpMethod.POST;
300             if (!create) {
301                 method = HttpMethod.PUT;
302             }
303 
304             InputStream body;
305             long length = 0;
306             if (resource instanceof VoidResource) {
307                 body = new ByteArrayInputStream(new byte[0]);
308             } else {
309                 ByteArrayOutputStream bodyOut = new ByteArrayOutputStream();
310                 mapMarshaller.marshal(bodyOut, req.getData());
311                 body = new ByteArrayInputStream(bodyOut.toByteArray());
312                 length = bodyOut.size();
313             }
314 
315             Request request = new DefaultRequest(method, href1, qs1, httpHeaders, body, length);
316 
317             Response response = execute(request);
318             Map<String, Object> responseBody = getBody(response);
319 
320             if (Collections.isEmpty(responseBody)) {
321                 // Fix for https://github.com/stormpath/stormpath-sdk-java/issues/218
322                 // Okta response with 200 for deactivate requests (i.e. /api/v1/apps/<id>/lifecycle/deactivate)
323                 if (response.getHttpStatus() == 202
324                         || response.getHttpStatus() == 200
325                         || response.getHttpStatus() == 204) {
326                     //202 means that the request has been accepted for processing, but the processing has not been completed. Therefore we do not have a response setBody.
327                     responseBody = java.util.Collections.emptyMap();
328                 } else {
329                     throw new IllegalStateException("Unable to obtain resource data from the API server.");
330                 }
331             }
332 
333             ResourceAction responseAction = getPostAction(req, response);
334 
335             return new DefaultResourceDataResult(responseAction, uri1, returnType, responseBody);
336         });
337 
338         ResourceAction action = create ? ResourceAction.CREATE : ResourceAction.UPDATE;
339 
340         ResourceDataRequest request = new DefaultResourceDataRequest(
341                 action,
342                 uri,
343                 canonicalizeParent(parentResource),
344                 returnType,
345                 getResourceClass(parentResource),
346                 props,
347                 requestHeaders);
348 
349         ResourceDataResult result = chain.filter(request);
350 
351         Map<String,Object> data = result.getData();
352 
353         //ensure the caller's argument is updated with what is returned from the server if the types are the same:
354         if (returnType.isAssignableFrom(abstractResource.getClass())) {
355             abstractResource.setInternalProperties(data);
356         }
357 
358         return resourceFactory.instantiate(returnType, data);
359     }

395     private void doDelete(String resourceHref, Class resourceClass, QueryString qs, HttpHeaders httpHeaders) {
396 
397         Assert.hasText(resourceHref, "This resource does not have an href value, therefore it cannot be deleted.");
398 
399         // if this URL is a partial, then we MUST add the baseUrl
400         final CanonicalUri uri = canonicalize(resourceHref, qs);
401 
402         FilterChain chain = new DefaultFilterChain(this.filters, request -> {
403             Request deleteRequest = new DefaultRequest(HttpMethod.DELETE, uri.getAbsolutePath(), uri.getQuery(), httpHeaders);
404             execute(deleteRequest);
405             //delete requests have HTTP 204 (no content), so just create an empty setBody for the result:
406             return new DefaultResourceDataResult(request.getAction(), request.getUri(), request.getResourceClass(), new HashMap<>());
407         });
408 
409         final CanonicalUri resourceUri = canonicalize(resourceHref, null);
410         ResourceDataRequest request = new DefaultResourceDataRequest(ResourceAction.DELETE, resourceUri, resourceClass, new HashMap<>());
411         chain.filter(request);
412     }

439     private Response execute(Request request) throws ResourceException {
440 
441         applyDefaultRequestHeaders(request);
442 
443         Response response = this.requestExecutor.executeRequest(request);
444         log.trace("Executed HTTP request.");
445 
446         if (requestLog.isTraceEnabled()) {
447             requestLog.trace("Executing request: method: '{}', url: {}", request.getMethod(), request.getResourceUrl());
448         }
449 
450         if (response.isError()) {
451             Map<String, Object> body = getBody(response);
452 
453             String requestId = response.getHeaders().getOktaRequestId();
454 
455             if (Strings.hasText(requestId)) {
456                 body.put(DefaultError.ERROR_ID.getName(), requestId);
457             }
458 
459             throw new ResourceException(new DefaultError(body)
460                                             .setHeaders(response.getHeaders().getXHeaders())
461                                             .setStatus(response.getHttpStatus()));
462         }
463 
464         return response;
465     }

474             out = mapMarshaller.unmarshal(response.getBody(), response.getHeaders().getLinkMap());

480     protected void applyDefaultRequestHeaders(Request request) {
481         request.getHeaders().setAccept(java.util.Collections.singletonList(MediaType.APPLICATION_JSON));
482 
483         // Get runtime headers from http client
484         Map<String, List<String>> headerMap = HttpHeadersHolder.get();
485         String oktaAgentHeaderName = OKTA_AGENT.toLowerCase(Locale.ENGLISH);
486         if (headerMap != null && headerMap.get(oktaAgentHeaderName) != null) {
487             List<String> oktaAgents = headerMap.get(oktaAgentHeaderName);
488             if (oktaAgents != null && !oktaAgents.isEmpty()) {
489                 String oktaAgent = Strings.arrayToDelimitedString(oktaAgents.toArray(), " ");
490                 applyUserAgent(oktaAgent + " " + USER_AGENT_STRING, request);
491             }
492         } else {
493             applyUserAgent(USER_AGENT_STRING, request);
494         }
495 
496         if (request.getHeaders().getContentType() == null && request.getBody() != null) {
497             // We only add the default content type (application/json) if a content type is not already in the request
498             request.getHeaders().setContentType(MediaType.APPLICATION_JSON);
499         }
500 
501         List<String> clientRequestId;
502         if (headerMap != null && (clientRequestId = headerMap.get(OKTA_CLIENT_REQUEST_ID)) != null) {
503             request.getHeaders().put(OKTA_CLIENT_REQUEST_ID, clientRequestId);
504         }
505     }

507     private void applyUserAgent(String userAgentString, Request request) {
508 
509         if (request.getHeaders().containsKey(USER_AGENT)) {
510             request.getHeaders().set(OKTA_USER_AGENT, userAgentString);
511         } else {
512             request.getHeaders().set(USER_AGENT, userAgentString);
513         }
514     }

516     private CanonicalUri canonicalize(String href, Map<String,?> queryParams) {
517         href = ensureFullyQualified(href);
518         return DefaultCanonicalUri.create(href, queryParams);
519     }

521     private CanonicalUri canonicalizeParent(Resource parentResource) {
522         if (parentResource != null && parentResource.getResourceHref() != null) {
523             String href = ensureFullyQualified(parentResource.getResourceHref());
524             return DefaultCanonicalUri.create(href, null);
525         }
526         return null;
527     }

529     private String ensureFullyQualified(String href) {
530         String value = href;
531         if (!isFullyQualified(href)) {
532             value = qualify(href);
533         }
534         return value;
535     }
536 
537     private boolean isFullyQualified(String href) {
538 
539         if (href == null || href.length() < 5) {
540             return false;
541         }
542         return href.regionMatches(true, 0, "http", 0, 4);
543     }

545     private String qualify(String href) {
546         StringBuilder sb = new StringBuilder(this.baseUrlResolver.getBaseUrl());
547         if (!href.startsWith("/")) {
548             sb.append("/");
549         }
550         sb.append(href);
551         return sb.toString();
552     }
