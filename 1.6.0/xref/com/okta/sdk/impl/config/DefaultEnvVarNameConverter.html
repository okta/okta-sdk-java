17  package com.okta.sdk.impl.config;
18  
19  import com.okta.commons.lang.Assert;
20  import com.okta.commons.lang.Strings;
21  import com.okta.sdk.client.ClientBuilder;
22  
23  import java.util.Arrays;
24  import java.util.Map;
25  import java.util.stream.Collectors;
26  
27  public class DefaultEnvVarNameConverter implements EnvVarNameConverter {
28  
29      private final Map<String, String> envToDotPropMap;
30  
31      public DefaultEnvVarNameConverter() {
32  
33          // this dependency on ClientBuilder isn't great, in the future we can change the API to ONLY support a one way conversion
34          this.envToDotPropMap = buildReverseLookupToMap(
35              ClientBuilder.DEFAULT_CLIENT_CACHE_TTL_PROPERTY_NAME,
36              ClientBuilder.DEFAULT_CLIENT_CACHE_TTI_PROPERTY_NAME,
37              ClientBuilder.DEFAULT_CLIENT_ORG_URL_PROPERTY_NAME,
38              ClientBuilder.DEFAULT_CLIENT_CONNECTION_TIMEOUT_PROPERTY_NAME,
39              ClientBuilder.DEFAULT_CLIENT_AUTHENTICATION_SCHEME_PROPERTY_NAME,
40              ClientBuilder.DEFAULT_CLIENT_REQUEST_TIMEOUT_PROPERTY_NAME,
41              ClientBuilder.DEFAULT_CLIENT_RETRY_MAX_ATTEMPTS_PROPERTY_NAME,
42              ClientBuilder.DEFAULT_CLIENT_TESTING_DISABLE_HTTPS_CHECK_PROPERTY_NAME,
43              ClientBuilder.DEFAULT_CLIENT_AUTHORIZATION_MODE_PROPERTY_NAME,
44              ClientBuilder.DEFAULT_CLIENT_ID_PROPERTY_NAME,
45              ClientBuilder.DEFAULT_CLIENT_SCOPES_PROPERTY_NAME,
46              ClientBuilder.DEFAULT_CLIENT_PRIVATE_KEY_PROPERTY_NAME);
47      }
48  
49      private Map<String, String> buildReverseLookupToMap(String... dottedPropertyNames) {
50          return Arrays.stream(dottedPropertyNames)
51              .collect(Collectors.toMap(this::toEnvVarName, dottedPropertyName -> dottedPropertyName));
52      }
53  
54      @Override
55      public String toEnvVarName(String dottedPropertyName) {
56          Assert.hasText(dottedPropertyName, "dottedPropertyName argument cannot be null or empty.");
57          dottedPropertyName = Strings.trimWhitespace(dottedPropertyName);
58  
59          StringBuilder sb = new StringBuilder();
60  
61          for(char c : dottedPropertyName.toCharArray()) {
62              if (c == '.') {
63                  sb.append('_');
64                  continue;
65              }
66              sb.append(Character.toUpperCase(c));
67          }
68  
69          return sb.toString();
70      }
71  
72      @Override
73      public String toDottedPropertyName(String envVarName) {
74          Assert.hasText(envVarName, "envVarName argument cannot be null or empty.");
75          envVarName = Strings.trimWhitespace(envVarName);
76  
77          //special cases (camel case):
78          if (envToDotPropMap.containsKey(envVarName)) {
79              return envToDotPropMap.get(envVarName);
80          }
81  
82          //default cases:
83          StringBuilder sb = new StringBuilder();
84  
85          for(char c : envVarName.toCharArray()) {
86              if (c == '_') {
87                  sb.append('.');
88                  continue;
89              }
90              sb.append(Character.toLowerCase(c));
91          }
92  
93          return sb.toString();
94      }
95  }
