124     /**
125      * Create signed JWT string with the supplied token client configuration details.
126      *
127      * @return signed JWT string
128      * @throws InvalidKeyException
129      * @throws IOException
130      */
131     String createSignedJWT() throws InvalidKeyException, IOException {
132         String clientId = tokenClientConfiguration.getClientId();
133         PrivateKey privateKey = parsePrivateKey(tokenClientConfiguration.getPrivateKey());
134         Instant now = Instant.now();
135 
136         String jwt = Jwts.builder()
137             .setAudience(tokenClientConfiguration.getBaseUrl() + TOKEN_URI)
138             .setIssuedAt(Date.from(now))
139             .setExpiration(Date.from(now.plus(1L, ChronoUnit.HOURS)))
140             .setIssuer(clientId)
141             .setSubject(clientId)
142             .claim("jti", UUID.randomUUID().toString())
143             .signWith(privateKey)
144             .compact();
145 
146         return jwt;
147     }

149     /**
150      * Parse private key from the supplied path.
151      *
152      * @param privateKeyFilePath
153      * @return {@link PrivateKey}
154      * @throws IOException
155      * @throws InvalidKeyException
156      */
157     PrivateKey parsePrivateKey(String privateKeyFilePath) throws IOException, InvalidKeyException {
158         Assert.notNull(privateKeyFilePath, "privateKeyFilePath may not be null");
159 
160         Path privateKeyPemFilePath = Paths.get(privateKeyFilePath);
161         Reader reader = Files.newBufferedReader(privateKeyPemFilePath, Charset.defaultCharset());
162 
163         PrivateKey privateKey = getPrivateKeyFromPEM(reader);
164         String algorithm = privateKey.getAlgorithm();
165 
166         if (!algorithm.equals("RSA") &&
167             !algorithm.equals("EC")) {
168             throw new InvalidKeyException("Supplied privateKey is not an RSA or EC key - " + algorithm);
169         }
170 
171         return privateKey;
172     }

174     /**
175      * Get Private key from input PEM file.
176      *
177      * @param reader
178      * @return {@link PrivateKey}
179      * @throws IOException
180      */
181     PrivateKey getPrivateKeyFromPEM(Reader reader) throws IOException {
182         PrivateKey privateKey;
183 
184         try (PEMParser pemParser = new PEMParser(reader)) {
185             JcaPEMKeyConverter jcaPEMKeyConverter = new JcaPEMKeyConverter();
186             Object pemContent = pemParser.readObject();
187 
188             if (pemContent == null) {
189                 throw new IllegalArgumentException("Invalid Private Key PEM file");
190             }
191 
192             if (pemContent instanceof PEMKeyPair) {
193                 PEMKeyPair pemKeyPair = (PEMKeyPair) pemContent;
194                 KeyPair keyPair = jcaPEMKeyConverter.getKeyPair(pemKeyPair);
195                 privateKey = keyPair.getPrivate();
196             } else if (pemContent instanceof PrivateKeyInfo) {
197                 PrivateKeyInfo privateKeyInfo = (PrivateKeyInfo) pemContent;
198                 privateKey = jcaPEMKeyConverter.getPrivateKey(privateKeyInfo);
199             } else {
200                 throw new IllegalArgumentException("Unsupported Private Key format '" +
201                     pemContent.getClass().getSimpleName() + '"');
202             }
203         }
204 
205         return privateKey;
206     }

208     /**
209      * Create token client config from the supplied API client config.
210      *
211      * Token client needs to retry http 401 errors only once which is not the case with the API client.
212      * We therefore effect this token client specific config by setting 'retryMaxElapsed'
213      * & 'retryMaxAttempts' fields.
214      *
215      * @param apiClientConfiguration supplied in the client configuration.
216      * @return ClientConfiguration to be used by token retrieval client.
217      */
218     ClientConfigurationpl/config/ClientConfiguration.html#ClientConfiguration">ClientConfiguration constructTokenClientConfig(ClientConfiguration apiClientConfiguration) {
219         ClientConfigurationConfiguration">ClientConfiguration tokenClientConfiguration = new ClientConfiguration();
220 
221         tokenClientConfiguration.setClientCredentialsResolver(
222             new DefaultClientCredentialsResolver(() -> Optional.empty()));
223 
224         tokenClientConfiguration.setRequestAuthenticator(new DisabledAuthenticator());
225 
226         tokenClientConfiguration.setCacheManagerEnabled(false);
227 
228         if (apiClientConfiguration.getBaseUrlResolver() != null)
229             tokenClientConfiguration.setBaseUrlResolver(apiClientConfiguration.getBaseUrlResolver());
230 
231         if (apiClientConfiguration.getProxy() != null)
232             tokenClientConfiguration.setProxy(apiClientConfiguration.getProxy());
233 
234         tokenClientConfiguration.setAuthenticationScheme(AuthenticationScheme.NONE);
235         tokenClientConfiguration.setAuthorizationMode(AuthorizationMode.get(tokenClientConfiguration.getAuthenticationScheme()));
236         tokenClientConfiguration.setClientId(apiClientConfiguration.getClientId());
237         tokenClientConfiguration.setScopes(apiClientConfiguration.getScopes());
238         tokenClientConfiguration.setPrivateKey(apiClientConfiguration.getPrivateKey());
239 
240         // setting this to '0' will disable this check and only 'retryMaxAttempts' will be effective
241         tokenClientConfiguration.setRetryMaxElapsed(0);
242         // retry only once for token requests (http 401 errors)
243         tokenClientConfiguration.setRetryMaxAttempts(1);
244 
245         return tokenClientConfiguration;
246     }
