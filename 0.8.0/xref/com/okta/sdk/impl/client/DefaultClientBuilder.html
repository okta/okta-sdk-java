56  /**
57   * <p>The default {@link ClientBuilder} implementation. This looks for configuration files
58   * in the following locations and order of precedence (last one wins).</p>
59   * <ul>
60   * <li>classpath:com/okta/sdk/config/okta.properties</li>
61   * <li>classpath:okta.properties</li>
62   * <li>classpath:okta.json</li>
63   * <li>classpath:okta.yaml</li>
64   * <li>~/.okta/okta.properties</li>
65   * <li>~/.okta/okta.json</li>
66   * <li>~/.okta/okta.yaml</li>
67   * <li>~/okta.properties</li>
68   * <li>~/okta.json</li>
69   * <li>~/okta.yaml</li>
70   * </ul>
71   *
72   * @since 0.5.0
73   */
74  public class DefaultClientBuilder implements ClientBuilder {
75  
76      private static final Logger log = LoggerFactory.getLogger(DefaultClientBuilder.class);
77  
78      private Proxy proxy;
79      private CacheManager cacheManager;
80      private ClientCredentials clientCredentials;

82      public static final  String ENVVARS_TOKEN  = "envvars";
83      public static final  String SYSPROPS_TOKEN = "sysprops";
84      private static final String OKTA_YAML      = "okta.yaml";
85      private static final String USER_HOME      = System.getProperty("user.home") + File.separatorChar;
86  
87      private static final String[] DEFAULT_OKTA_PROPERTIES_FILE_LOCATIONS = {
88                                                   ClasspathResource.SCHEME_PREFIX + "com/okta/sdk/config/" + OKTA_YAML,
89                                                   ClasspathResource.SCHEME_PREFIX + OKTA_YAML,
90                                                   USER_HOME + ".okta" + File.separatorChar + OKTA_YAML,
91                                                   ENVVARS_TOKEN,
92                                                   SYSPROPS_TOKEN
93      };
94  
95      private ClientConfiguration clientConfig = new ClientConfiguration();
96  
97      public DefaultClientBuilder() {
98          this(new DefaultResourceFactory());
99      }

101     DefaultClientBuilder(ResourceFactory resourceFactory) {
102         Collection<PropertiesSource> sources = new ArrayList<>();
103 
104         for (String location : DEFAULT_OKTA_PROPERTIES_FILE_LOCATIONS) {
105 
106             if (ENVVARS_TOKEN.equalsIgnoreCase(location)) {
107                 sources.add(EnvironmentVariablesPropertiesSource.oktaFilteredPropertiesSource());
108             }
109             else if (SYSPROPS_TOKEN.equalsIgnoreCase(location)) {
110                 sources.add(SystemPropertiesSource.oktaFilteredPropertiesSource());
111             }
112             else {
113                 Resource resource = resourceFactory.createResource(location);
114 
115                 PropertiesSource wrappedSource;
116                 if (Strings.endsWithIgnoreCase(location, ".yaml") ||
117                         Strings.endsWithIgnoreCase(location, ".yml")) {
118                     wrappedSource = new YAMLPropertiesSource(resource);
119                 } else {
120                     wrappedSource = new ResourcePropertiesSource(resource);
121                 }
122 
123                 PropertiesSource propertiesSource = new OptionalPropertiesSource(wrappedSource);
124                 sources.add(propertiesSource);
125             }
126         }
127 
128         Map<String, String> props = new LinkedHashMap<>();
129 
130         for (PropertiesSource source : sources) {
131             Map<String, String> srcProps = source.getProperties();
132             props.putAll(srcProps);
133         }
134 
135         // check to see if property value is null before setting value
136         // if != null, allow it to override previously set values
137         if (Strings.hasText(props.get(DEFAULT_CLIENT_API_TOKEN_PROPERTY_NAME))) {
138             clientConfig.setApiToken(props.get(DEFAULT_CLIENT_API_TOKEN_PROPERTY_NAME));
139         }
140 
141         if (Strings.hasText(props.get(DEFAULT_CLIENT_CACHE_ENABLED_PROPERTY_NAME))) {
142             clientConfig.setCacheManagerEnabled(Boolean.valueOf(props.get(DEFAULT_CLIENT_CACHE_ENABLED_PROPERTY_NAME)));
143         }
144 
145         if (Strings.hasText(props.get(DEFAULT_CLIENT_CACHE_TTL_PROPERTY_NAME))) {
146             clientConfig.setCacheManagerTtl(Long.parseLong(props.get(DEFAULT_CLIENT_CACHE_TTL_PROPERTY_NAME)));
147         }
148 
149         if (Strings.hasText(props.get(DEFAULT_CLIENT_CACHE_TTI_PROPERTY_NAME))) {
150             clientConfig.setCacheManagerTti(Long.parseLong(props.get(DEFAULT_CLIENT_CACHE_TTI_PROPERTY_NAME)));
151         }
152 
153         for (String prop : props.keySet()) {
154             boolean isPrefix = prop.length() == DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME.length();
155             if (!isPrefix && prop.startsWith(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME)) {
156                 // get class from prop name
157                 String cacheClass = prop.substring(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME.length() + 1, prop.length() - 4);
158                 String cacheTti = props.get(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME + "." + cacheClass + ".tti");
159                 String cacheTtl = props.get(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME + "." + cacheClass + ".ttl");
160                 CacheConfigurationBuilder cacheBuilder = Caches.forResource(Classes.forName(cacheClass));
161                 if (Strings.hasText(cacheTti)) {
162                     cacheBuilder.withTimeToIdle(Long.parseLong(cacheTti), TimeUnit.SECONDS);
163                 }
164                 if (Strings.hasText(cacheTtl)) {
165                     cacheBuilder.withTimeToLive(Long.parseLong(cacheTtl), TimeUnit.SECONDS);
166                 }
167                 if (!clientConfig.getCacheManagerCaches().containsKey(cacheClass)) {
168                     clientConfig.getCacheManagerCaches().put(cacheClass, cacheBuilder);
169                 }
170             }
171         }
172 
173         if (Strings.hasText(props.get(DEFAULT_CLIENT_ORG_URL_PROPERTY_NAME))) {
174             String baseUrl = props.get(DEFAULT_CLIENT_ORG_URL_PROPERTY_NAME);
175             // remove backslashes that can end up in file when it's written programmatically, e.g. in a test
176             baseUrl = baseUrl.replace("\\:", ":");
177             clientConfig.setBaseUrl(baseUrl);
178         }
179 
180         if (Strings.hasText(props.get(DEFAULT_CLIENT_CONNECTION_TIMEOUT_PROPERTY_NAME))) {
181             clientConfig.setConnectionTimeout(Integer.parseInt(props.get(DEFAULT_CLIENT_CONNECTION_TIMEOUT_PROPERTY_NAME)));
182         }
183 
184         if (Strings.hasText(props.get(DEFAULT_CLIENT_AUTHENTICATION_SCHEME_PROPERTY_NAME))) {
185             clientConfig.setAuthenticationScheme(Enum.valueOf(AuthenticationScheme.class, props.get(DEFAULT_CLIENT_AUTHENTICATION_SCHEME_PROPERTY_NAME)));
186         }
187 
188         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_PORT_PROPERTY_NAME))) {
189             clientConfig.setProxyPort(Integer.parseInt(props.get(DEFAULT_CLIENT_PROXY_PORT_PROPERTY_NAME)));
190         }
191 
192         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_HOST_PROPERTY_NAME))) {
193             clientConfig.setProxyHost(props.get(DEFAULT_CLIENT_PROXY_HOST_PROPERTY_NAME));
194         }
195 
196         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_USERNAME_PROPERTY_NAME))) {
197             clientConfig.setProxyUsername(props.get(DEFAULT_CLIENT_PROXY_USERNAME_PROPERTY_NAME));
198         }
199 
200         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_PASSWORD_PROPERTY_NAME))) {
201             clientConfig.setProxyPassword(props.get(DEFAULT_CLIENT_PROXY_PASSWORD_PROPERTY_NAME));
202         }
203     }

258     @Override
259     public Client build() {
260         if (!this.clientConfig.isCacheManagerEnabled()) {
261             log.debug("CacheManager disabled. Defaulting to DisabledCacheManager");
262             this.cacheManager = Caches.newDisabledCacheManager();
263         } else if (this.cacheManager == null) {
264             log.debug("No CacheManager configured. Defaulting to in-memory CacheManager with default TTL and TTI of five minutes.");
265 
266             CacheManagerBuilder cacheManagerBuilder = Caches.newCacheManager()
267                     .withDefaultTimeToIdle(this.clientConfig.getCacheManagerTti(), TimeUnit.SECONDS)
268                     .withDefaultTimeToLive(this.clientConfig.getCacheManagerTtl(), TimeUnit.SECONDS);
269             if (this.clientConfig.getCacheManagerCaches().size() > 0) {
270                 for (CacheConfigurationBuilder builder : this.clientConfig.getCacheManagerCaches().values()) {
271                     cacheManagerBuilder.withCache(builder);
272                 }
273             }
274 
275             this.cacheManager = cacheManagerBuilder.build();
276         }
277 
278         // use proxy overrides if they're set
279         if (this.clientConfig.getProxyPort() > 0 || this.clientConfig.getProxyHost() != null &&
280                 (this.clientConfig.getProxyUsername() == null || this.clientConfig.getProxyPassword() == null)) {
281             this.proxy = new Proxy(this.clientConfig.getProxyHost(), this.clientConfig.getProxyPort());
282         } else if (this.clientConfig.getProxyUsername() != null && this.clientConfig.getProxyPassword() != null) {
283             this.proxy = new Proxy(this.clientConfig.getProxyHost(), this.clientConfig.getProxyPort(),
284                     this.clientConfig.getProxyUsername(), this.clientConfig.getProxyPassword());
285         }
286 
287         ClientCredentialsResolver clientCredentialsResolver = this.clientConfig.getClientCredentialsResolver();
288 
289         if (clientCredentialsResolver == null && this.clientCredentials != null) {
290             clientCredentialsResolver = new DefaultClientCredentialsResolver(this.clientCredentials);
291         }
292         else if (clientCredentialsResolver == null) {
293             clientCredentialsResolver = new DefaultClientCredentialsResolver(clientConfig);
294         }
295 
296         BaseUrlResolver baseUrlResolver = this.clientConfig.getBaseUrlResolver();
297 
298         if (baseUrlResolver == null) {
299             Assert.notNull(this.clientConfig.getBaseUrl(), "Okta org url must not be null.");
300             baseUrlResolver = new DefaultBaseUrlResolver(this.clientConfig.getBaseUrl());
301         }
302 
303         return new DefaultClient(clientCredentialsResolver, baseUrlResolver, this.proxy, this.cacheManager,
304                 this.clientConfig.getAuthenticationScheme(), this.clientConfig.getRequestAuthenticatorFactory(), this.clientConfig.getConnectionTimeout());
305     }

309         if (baseUrl == null) {
310             throw new IllegalArgumentException("baseUrl argument cannot be null.");
311         }

313         return this;
314     }
