234     /**
235      * This method provides the ability to instruct the DataStore how to decide which class of a resource hierarchy will
236      * be instantiated.
237      *
238      * @param href            the endpoint where the request will be targeted to.
239      * @param parent          the root class of the Resource hierarchy (helps to validate that the idClassMap contains
240      *                        subclasses of it).
241      * @param childIdProperty the property whose value will be used to identify the specific class in the hierarchy that
242      *                        we need to instantiate.
243      * @param idClassMap      a mapping to be able to know which class corresponds to each <code>childIdProperty</code>
244      *                        value.
245      * @param <T>             the root of the hierarchy of the Resource we want to instantiate.
246      * @param <R>             the sub-class of the root Resource.
247      * @return the retrieved resource
248      */
249     @Override
250     public <T extends Resource, R extends T> R getResource(String href, Class<T> parent, String childIdProperty,
251                                                            Map<String, Class<? extends R>> idClassMap) {
252         Assert.hasText(childIdProperty, "childIdProperty cannot be null or empty.");
253         Assert.notEmpty(idClassMap, "idClassMap cannot be null or empty.");
254 
255         ResourceDataResult result = getResourceData(href, parent, null);
256         Map<String, ?> data = result.getData();
257 
258         if (Collections.isEmpty(data) || !data.containsKey(childIdProperty)) {
259             throw new IllegalStateException(childIdProperty + " could not be found in: " + data + ".");
260         }
261 
262         Object propertyValue = data.get(childIdProperty);
263         if (propertyValue == null) {
264             throw new IllegalStateException("No Class mapping could be found for " + childIdProperty + ".");
265         }
266 
267         Class<? extends R> childClass = idClassMap.get(propertyValue.toString());
268 
269         return instantiate(childClass, data, result.getUri().getQuery());
270     }

283     @SuppressWarnings("unchecked")
284     private ResourceDataResult getResourceData(String href, Class<? extends Resource> clazz, Map<String,?> queryParameters) {
285 
286         Assert.hasText(href, "href argument cannot be null or empty.");
287         Assert.notNull(clazz, "Resource class argument cannot be null.");
288 
289         FilterChain chain = new DefaultFilterChain(this.filters, new FilterChain() {
290             @Override
291             public ResourceDataResult filter(final ResourceDataRequest req) {
292 
293                 CanonicalUri uri = req.getUri();
294 
295                 Request getRequest = new DefaultRequest(HttpMethod.GET, uri.getAbsolutePath(), uri.getQuery());
296                 Response getResponse = execute(getRequest);
297                 Map<String,?> body = getBody(getResponse);
298 
299                 if (Collections.isEmpty(body)) {
300                     throw new IllegalStateException("Unable to obtain resource data from the API server or from cache.");
301                 }
302 
303                 return new DefaultResourceDataResult(req.getAction(), uri, req.getResourceClass(), (Map<String,Object>)body);
304             }
305         });
306 
307         CanonicalUri uri = canonicalize(href, queryParameters);
308         ResourceDataRequest req = new DefaultResourceDataRequest(ResourceAction.READ, uri, clazz, new HashMap<String,Object>());
309         return chain.filter(req);
310     }

312     private ResourceAction getPostAction(ResourceDataRequest request, Response response) {
313         int httpStatus = response.getHttpStatus();
314         if (httpStatus == 201) {
315             return ResourceAction.CREATE;
316         }
317         // TODO: verify this is an issue with Okta
318         //Fix for https://github.com/stormpath/stormpath-sdk-java/issues/403
319         if (httpStatus == 200) {
320             return ResourceAction.READ;
321         }
322 
323         return request.getAction();
324     }

394     @SuppressWarnings("unchecked")
395     private <T extends Resource, R extends Resource> R save(String href, final T resource, HttpHeaders requestHeaders, final Class<? extends R> returnType, final QueryString qs, final boolean create) {
396         Assert.hasText(href, "href argument cannot be null or empty.");
397         Assert.notNull(resource, "resource argument cannot be null.");
398         Assert.notNull(returnType, "returnType class cannot be null.");
399         Assert.isInstanceOf(AbstractResource.class, resource);
400         Assert.isTrue(!CollectionResource.class.isAssignableFrom(resource.getClass()), "Collections cannot be persisted.");
401 
402         final CanonicalUri uri = canonicalize(href, qs);
403         final AbstractResource abstractResource = (AbstractResource) resource;
404         final Map<String, Object> props = resourceConverter.convert(abstractResource, false); // TODO: Okta doesn't have great support for updating just dirty props so always include clean and dirty props
405 
406         FilterChain chain = new DefaultFilterChain(this.filters, new FilterChain() {
407             @Override
408             public ResourceDataResult filter(final ResourceDataRequest req) {
409 
410                 String bodyString;
411                 if (req.getHttpHeaders().getContentType() != null && req.getHttpHeaders().getContentType().equals(MediaType.APPLICATION_FORM_URLENCODED)){
412                     bodyString = buildCanonicalBodyQueryParams(req.getData());
413                 } else {
414                     if (resource instanceof VoidResource) {
415                         bodyString = "";
416                     } else {
417                         bodyString = mapMarshaller.marshal(req.getData());
418                     }
419                 }
420                 StringInputStream body = new StringInputStream(bodyString);
421                 long length = body.available();
422 
423                 CanonicalUri uri = req.getUri();
424                 String href = uri.getAbsolutePath();
425                 QueryString qs = uri.getQuery();
426 
427                 HttpHeaders httpHeaders = req.getHttpHeaders();
428 
429                 // if this is an Okta user, we must use a PUT and not a POST
430                 HttpMethod method = HttpMethod.POST;
431                 if (!create) {
432                     method = HttpMethod.PUT;
433                 }
434                 Request request = new DefaultRequest(method, href, qs, httpHeaders, body, length);
435 
436                 Response response = execute(request);
437                 Map<String, Object> responseBody = getBody(response);
438 
439                 if (Collections.isEmpty(responseBody)) {
440                     // Fix for https://github.com/stormpath/stormpath-sdk-java/issues/218
441                     // Okta response with 200 for deactivate requests (i.e. /api/v1/apps/<id>/lifecycle/deactivate)
442                     if (response.getHttpStatus() == 202
443                             || response.getHttpStatus() == 200
444                             || response.getHttpStatus() == 204) {
445                         //202 means that the request has been accepted for processing, but the processing has not been completed. Therefore we do not have a response body.
446                         responseBody = java.util.Collections.emptyMap();
447                     } else {
448                         throw new IllegalStateException("Unable to obtain resource data from the API server.");
449                     }
450                 }
451 
452                 ResourceAction responseAction = getPostAction(req, response);
453 
454                 return new DefaultResourceDataResult(responseAction, uri, returnType, responseBody);
455             }
456         });
457 
458         ResourceAction action = create ? ResourceAction.CREATE : ResourceAction.UPDATE;
459         ResourceDataRequest request = new DefaultResourceDataRequest(action, uri, abstractResource.getClass(), props, requestHeaders);
460 
461         ResourceDataResult result = chain.filter(request);
462 
463         Map<String,Object> data = result.getData();
464 
465         //ensure the caller's argument is updated with what is returned from the server if the types are the same:
466         if (returnType.equals(abstractResource.getClass())) {
467             abstractResource.setInternalProperties(data);
468         }
469 
470         return resourceFactory.instantiate(returnType, data);
471     }

493     private String buildCanonicalBodyQueryParams(Map<String, Object> bodyData){
494         StringBuilder builder = new StringBuilder();
495         Map<String, Object> treeMap = new TreeMap<String, Object>(bodyData);
496         try {
497             for (Map.Entry<String,Object> entry : treeMap.entrySet()) {
498                 Object value = entry.getValue();
499                 if (value != null) {
500                     if (builder.length() > 0) {
501                         builder.append(APPEND_PARAM_CHAR);
502                     }
503                     builder.append(String.format("%s=%s", URLEncoder.encode(entry.getKey(), "UTF-8"), URLEncoder.encode(value.toString(), "UTF-8")));
504                 }
505             }
506         } catch (UnsupportedEncodingException e){
507             log.trace("Body content could not be properly encoded");
508             return null;
509         }
510         return builder.toString();
511     }

513     private void doDelete(String resourceHref, Class resourceClass, final String possiblyNullPropertyName) {
514 
515         Assert.hasText(resourceHref, "This resource does not have an href value, therefore it cannot be deleted.");
516 
517         // if this URL is a partial, then we MUST add the baseUrl
518         if (resourceHref.startsWith("/") ) {
519             resourceHref = qualify(resourceHref);
520         }
521 
522         final String requestHref;
523         if (Strings.hasText(possiblyNullPropertyName)) { //delete just that property, not the entire resource:
524             requestHref = resourceHref + "/" + possiblyNullPropertyName;
525         } else {
526             requestHref = resourceHref;
527         }
528 
529         FilterChain chain = new DefaultFilterChain(this.filters, new FilterChain() {
530 
531             @Override
532             public ResourceDataResult filter(ResourceDataRequest request) {
533                 Request deleteRequest = new DefaultRequest(HttpMethod.DELETE, requestHref);
534                 execute(deleteRequest);
535                 //delete requests have HTTP 204 (no content), so just create an empty body for the result:
536                 return new DefaultResourceDataResult(request.getAction(), request.getUri(), request.getResourceClass(), new HashMap<String, Object>());
537             }
538         });
539 
540         final CanonicalUri resourceUri = canonicalize(resourceHref, null);
541         ResourceDataRequest request = new DefaultResourceDataRequest(ResourceAction.DELETE, resourceUri, resourceClass, new HashMap<String, Object>());
542         chain.filter(request);
543     }

546 
547         Assert.notNull(resource, "resource argument cannot be null.");
548         Assert.isInstanceOf(AbstractResource.class, resource, "Resource argument must be an AbstractResource.");
549 
550         doDelete(resource.getResourceHref(), resource.getClass(), possiblyNullPropertyName);
551     }

570     private Response execute(Request request) throws ResourceException {
571 
572         applyDefaultRequestHeaders(request);
573 
574         Response response = this.requestExecutor.executeRequest(request);
575         log.trace("Executed HTTP request.");
576 
577         if (requestLog.isTraceEnabled()) {
578             requestLog.trace("Executing request: method: '{}', url: {}", request.getMethod(), request.getResourceUrl());
579         }
580 
581         if (response.isError()) {
582             Map<String, Object> body = getBody(response);
583 
584             String requestId = response.getHeaders().getOktaRequestId();
585 
586             if (Strings.hasText(requestId)) {
587                 body.put(DefaultError.ERROR_ID.getName(), requestId);
588             }
589 
590             DefaultError error = new DefaultError(body)
591                     .setHeaders(response.getHeaders().getXHeaders());
592             throw new ResourceException(error);
593         }
594 
595         return response;
596     }

598     private Map<String, Object> getBody(Response response) {
599 
600         Assert.notNull(response, "response argument cannot be null.");
601 
602         Map<String, Object> out = null;
603 
604         if (response.hasBody()) {
605             out = mapMarshaller.unmarshall(response.getBody(), response.getHeaders().getLinkMap());
606         }
607 
608         return out;
609     }

611     protected void applyDefaultRequestHeaders(Request request) {
612         request.getHeaders().setAccept(java.util.Collections.singletonList(MediaType.APPLICATION_JSON));
613 
614         // Get runtime headers from http client
615         Map<String, List<String>> headerMap = HttpHeadersHolder.get();
616         String oktaAgentHeaderName = OKTA_AGENT.toLowerCase(Locale.ENGLISH);
617         if (headerMap != null && headerMap.get(oktaAgentHeaderName) != null) {
618             List<String> oktaAgents = headerMap.get(oktaAgentHeaderName);
619             if (oktaAgents != null && oktaAgents.size() > 0) {
620                 String oktaAgent = Strings.arrayToDelimitedString(oktaAgents.toArray(), " ");
621                 request.getHeaders().set("User-Agent", oktaAgent + " " + USER_AGENT_STRING);
622             }
623         } else {
624             request.getHeaders().set("User-Agent", USER_AGENT_STRING);
625         }
626         if (request.getHeaders().getContentType() == null && request.getBody() != null) {
627             // We only add the default content type (application/json) if a content type is not already in the request
628             request.getHeaders().setContentType(MediaType.APPLICATION_JSON);
629         }
630 
631         List<String> clientRequestId;
632         if (headerMap != null && (clientRequestId = headerMap.get(OKTA_CLIENT_REQUEST_ID)) != null) {
633             request.getHeaders().put(OKTA_CLIENT_REQUEST_ID, clientRequestId);
634         }
635     }

637     protected CanonicalUri canonicalize(String href, Map<String,?> queryParams) {
638         href = ensureFullyQualified(href);
639         return DefaultCanonicalUri.create(href, queryParams);

642     protected String ensureFullyQualified(String href) {
643         String value = href;
644         if (!isFullyQualified(href)) {
645             value = qualify(href);
646         }
647         return value;
648     }

650     protected boolean isFullyQualified(String href) {
651 
652         if (href == null || href.length() < 5) {
653             return false;
654         }
655 
656         char c = href.charAt(0);
657         if (c == 'h' || c == 'H') {
658             c = href.charAt(1);
659             if (c == 't' || c == 'T') {
660                 c = href.charAt(2);
661                 if (c == 't' || c == 'T') {
662                     c = href.charAt(3);
663                     if (c == 'p' || c == 'P') {
664                         return true;
665                     }
666                 }
667             }
668         }
669 
670         return false;
671     }

673     protected String qualify(String href) {
674         StringBuilder sb = new StringBuilder(this.baseUrlResolver.getBaseUrl());
675         if (!href.startsWith("/")) {
676             sb.append("/");
677         }
678         sb.append(href);
679         return sb.toString();
680     }
