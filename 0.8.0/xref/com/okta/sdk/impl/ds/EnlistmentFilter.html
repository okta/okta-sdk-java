16  package com.okta.sdk.impl.ds;
17  
18  import com.okta.sdk.impl.resource.AbstractInstanceResource;
19  import com.okta.sdk.impl.util.SoftHashMap;
20  import com.okta.sdk.lang.Assert;
21  
22  import java.util.LinkedHashMap;
23  import java.util.Map;
24  
25  /**
26   *
27   * @since 0.5.0
28   */
29  public class EnlistmentFilter implements Filter {
30  
31      /**
32       * Fix for https://github.com/stormpath/stormpath-sdk-java/issues/47. Data map is now shared among all Resource
33       * instances referencing the same {@code href}.
34       */
35      private final Map<String, Enlistment> hrefMapStore;
36  
37      public EnlistmentFilter() {
38          this.hrefMapStore = new SoftHashMap<String, Enlistment>();
39      }
40  
41      @Override
42      public ResourceDataResult filter(ResourceDataRequest request, FilterChain chain) {
43  
44          ResourceDataResult result = chain.filter(request);
45  
46          Map<String,Object> data = result.getData();
47  
48          if (request.getAction() == ResourceAction.DELETE) {
49              hrefMapStore.remove(result.getUri().getAbsolutePath());
50          } else if (AbstractInstanceResource.isInstanceResource(data)) {
51              data = toEnlistment(data);
52              result = new DefaultResourceDataResult(result.getAction(), result.getUri(), result.getResourceClass(), data);
53          }
54  
55          return result;
56      }
57  
58      @SuppressWarnings({ "SuspiciousMethodCalls", "unchecked" })
59      private Enlistment toEnlistment(final Map<String, ?> data) {
60  
61          Assert.notEmpty(data, "data cannot be null or empty.");
62          String href = (String)data.get("href");
63          Assert.hasText(href, "href cannot be null or empty.");
64  
65          Map modified = new LinkedHashMap<String, Object>(data.size());
66  
67          //since 1.0.RC4.3 - need to recursively add enlistments if the data is expanded:
68          for(Object o : data.entrySet()) {
69              Map.Entry entry = (Map.Entry)o;
70              Object key = entry.getKey();
71              Object value = entry.getValue();
72              if (value instanceof Map && AbstractInstanceResource.isInstanceResource((Map<String, ?>) value)) {
73                  value = toEnlistment((Map<String, ?>)value);
74              }
75              modified.put(key, value);
76          }
77  
78          Enlistment enlistment;
79          if (this.hrefMapStore.containsKey(href)) {
80              enlistment = this.hrefMapStore.get(href);
81              enlistment.setProperties((Map<String, Object>) modified);
82          } else {
83              enlistment = new Enlistment((Map<String, Object>) modified);
84              this.hrefMapStore.put(href, enlistment);
85          }
86  
87          return enlistment;
88      }
89  }
