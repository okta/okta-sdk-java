37  public abstract class AbstractCollectionResource<T extends Resource> extends AbstractResource implements CollectionResource<T> {
38  
39      public static final StringProperty NEXT_PAGE = new StringProperty("nextPage");
40      public static final String ITEMS_PROPERTY_NAME = "items";
41  
42      private final Map<String, Object> queryParams;
43      private String nextPageHref = null;
44  
45      private AtomicBoolean firstPageQueryRequired = new AtomicBoolean();
46  
47      protected AbstractCollectionResource(InternalDataStore dataStore) {
48          super(dataStore);
49          this.queryParams = Collections.emptyMap();
50      }
51  
52      protected AbstractCollectionResource(InternalDataStore dataStore, Map<String, Object> properties) {
53          super(dataStore, properties);
54          this.queryParams = Collections.emptyMap();
55          this.nextPageHref = getString(NEXT_PAGE);
56      }
57  
58      protected AbstractCollectionResource(InternalDataStore dataStore, Map<String, Object> properties, Map<String, Object> queryParams) {
59          super(dataStore, properties);
60          this.nextPageHref = getString(NEXT_PAGE);
61          if (queryParams != null) {
62              this.queryParams = queryParams;
63          } else {
64              this.queryParams = Collections.emptyMap();
65          }
66      }
67  
68      /**
69       * Returns {@code true} if the specified data map represents a materialized collection resource data set, {@code
70       * false} otherwise.
71       *
72       * @param props the data properties to test
73       * @return {@code true} if the specified data map represents a materialized collection resource data set, {@code
74       * false} otherwise.
75       */
76      public static boolean isCollectionResource(Map<String,?> props) {
77          return isMaterialized(props)
78                  && props.get(ITEMS_PROPERTY_NAME) instanceof Iterable;
79      }
80  
81      private String getNextPageHref() {
82          return nextPageHref;
83      }
84  
85      private boolean hasNextPage() {
86          return getNextPageHref() != null;
87      }
88  
89      @Override
90      public T single() {
91          Iterator<T> iterator = iterator();
92          if (!iterator.hasNext()) {
93              throw new IllegalStateException("This list is empty while it was expected to contain one (and only one) element.");
94          }
95          T itemToReturn = iterator.next();
96          if (iterator.hasNext()) {
97              throw new IllegalStateException("Only a single resource was expected, but this list contains more than one item.");
98          }
99          return itemToReturn;
100     }
101 
102     protected abstract Class<T> getItemType();
103 
104     @SuppressWarnings("unchecked")
105     public Page<T> getCurrentPage() {
106 
107         Collection<T> items = Collections.emptyList();
108 
109         Object value = getProperty(ITEMS_PROPERTY_NAME);
110 
111         if (value != null) {
112             Collection c = null;
113             if (value instanceof Map[]) {
114                 Map[] vals = (Map[]) value;
115                 if (vals.length > 0) {
116                     c = Arrays.asList((Map[]) vals);
117                 }
118             } else if (value instanceof Collection) {
119                 Collection vals = (Collection) value;
120                 if (vals.size() > 0) {
121                     c = vals;
122                 }
123             }
124             if (c != null && !c.isEmpty()) {
125                 //do a look ahead to see if resource conversion has already taken place:
126                 if (!getItemType().isInstance(c.iterator().next())) {
127                     //need to convert the list of links to a list of unmaterialized Resources
128                     items = toResourceList(c, getItemType());
129                     //replace the existing list of links with the newly constructed list of Resources.  Don't dirty
130                     //the instance - we're just swapping out a property that already exists for the materialized version.
131                     setProperty(ITEMS_PROPERTY_NAME, items, false);
132                 } else {
133                     //the collection has already been converted to Resources - use it directly:
134                     items = c;
135                 }
136             }
137         }
138 
139         return new DefaultPage<>(items);
140     }
141 
142 
143     @Override
144     public Iterator<T> iterator() {
145         //firstPageQueryRequired ensures that newly obtained collection resources don't need to query unnecessarily
146         return new PaginatedIterator<>(this, firstPageQueryRequired.getAndSet(true));
147     }
148 
149     private Collection<T> toResourceList(Collection vals, Class<T> itemType) {
150 
151         List<T> list = new ArrayList<>(vals.size());
152 
153         for (Object o : vals) {
154             Map<String, Object> properties = (Map<String, Object>) o;
155             T resource = toResource(itemType, properties);
156             list.add(resource);
157         }
158 
159         return list;
160     }
161 
162     protected T toResource(Class<T> resourceClass, Map<String, Object> properties) {
163         return getDataStore().instantiate(resourceClass, properties);
164     }
165 
166     private class PaginatedIterator<T extends Resource> implements Iterator<T> {
167 
168         private AbstractCollectionResource<T> resource;
169 
170         private Page<T> currentPage;
171         private Iterator<T> currentPageIterator;
172         private int currentItemIndex;
173 
174         private PaginatedIterator(AbstractCollectionResource<T> resource, boolean firstPageQueryRequired) {
175 
176             if (firstPageQueryRequired) {
177                 //We get a new resource in order to have different iterator instances: issue 62 (https://github.com/stormpath/stormpath-sdk-java/issues/62)
178                 this.resource = getDataStore().getResource(resource.getResourceHref(), resource.getClass(), resource.queryParams);
179                 this.currentPage = this.resource.getCurrentPage();
180             } else {
181                 this.resource = resource;
182                 this.currentPage = resource.getCurrentPage();
183             }
184 
185             this.currentPageIterator = this.currentPage.getItems().iterator();
186             this.currentItemIndex = 0;
187         }
188 
189         @SuppressWarnings("unchecked")
190         @Override
191         public boolean hasNext() {
192 
193             boolean hasNext = currentPageIterator.hasNext();
194 
195             //If we have already exhausted the whole collection size there is no need to contact the backend again
196             if (!hasNext && hasNextPage()) {
197 
198                 //if we're done with the current page, and we've exhausted the page limit (i.e. we've read a
199                 //full page), we will have to execute another request to check to see if another page exists.
200                 //We can't 'trust' the current page iterator to know if more results exist on the server since it
201                 //only represents a single page.
202 
203                 AbstractCollectionResource nextResource =
204                         getDataStore().getResource(getNextPageHref(), resource.getClass());
205                 Page<T> nextPage = nextResource.getCurrentPage();
206                 Iterator<T> nextIterator = nextPage.getItems().iterator();
207 
208                 if (nextIterator.hasNext()) {
209                     hasNext = true;
210                     //update to reflect the new page:
211                     this.resource = nextResource;
212                     this.currentPage = nextPage;
213                     this.currentPageIterator = nextIterator;
214                     this.currentItemIndex = 0;
215                     nextPageHref = nextResource.getString(NEXT_PAGE);
216                 }
217             }
218 
219             return hasNext;
220         }
221 
222         @Override
223         public T next() {
224             T item = currentPageIterator.next();
225             currentItemIndex++;
226             return item;
227         }
228 
229         @Override
230         public void remove() {
231             throw new UnsupportedOperationException("Remove is not supported.");
232         }
233     }
234 
235     private static class DefaultPage<T> implements Page<T> {
236 
237         private final Collection<T> items;
238 
239         DefaultPage(Collection<T> items) {
240             this.items = Collections.unmodifiableCollection(items);
241         }
242 
243         @Override
244         public Collection<T> getItems() {
245             return this.items;
246         }
247     }

249     public static final <T extends Resource, R extends CollectionResource<T>> R emptyCollectionResource(InternalDataStore internalDataStore, Class<R> implClass) {
250         List<T> emptyList = Collections.emptyList();
251 
252         Map<String, Object> properties = new LinkedHashMap<>();
253         properties.put(ITEMS_PROPERTY_NAME, emptyList);
254 
255         Constructor<R> ctor = Classes.getConstructor(implClass, InternalDataStore.class, Map.class);
256         return Classes.instantiate(ctor, internalDataStore, properties);
257     }
258 
259     public static final <T extends Resource, R extends CollectionResource<T>> R singletonCollectionResource(InternalDataStore internalDataStore, Class<R> implClass, T item) {
260 
261         List<T> singletonList = Collections.singletonList(item);
262 
263         Map<String, Object> properties = new LinkedHashMap<>();
264         properties.put(ITEMS_PROPERTY_NAME, singletonList);
265 
266         Constructor<R> ctor = Classes.getConstructor(implClass, InternalDataStore.class, Map.class);
267         return Classes.instantiate(ctor, internalDataStore, properties);
268     }
