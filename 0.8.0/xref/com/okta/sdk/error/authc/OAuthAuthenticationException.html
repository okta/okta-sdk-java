16  package com.okta.sdk.error.authc;
17  
18  import com.okta.sdk.error.Error;
19  import com.okta.sdk.resource.ResourceException;
20  
21  /**
22   * A sub-class of {@link com.okta.sdk.resource.ResourceException} representing an attempt fail an oauth authentication.
23   *
24   * @since 0.5.0
25   */
26  public class OAuthAuthenticationException extends ResourceException {
27  
28      public static final String INVALID_REQUEST = "invalid_request";
29  
30      public static final String INVALID_CLIENT = "invalid_client";
31  
32      public static final String UNSUPPORTED_GRANT_TYPE = "unsupported_grant_type";
33  
34      public static final String INVALID_GRANT = "invalid_grant";
35  
36      public static final String INVALID_SCOPE = "invalid_scope";
37  
38      private final String oauthError;
39  
40      public OAuthAuthenticationException(Error error, String oauthError) {
41          super(error);
42          this.oauthError = oauthError;
43      }
44  
45      public String getOAuthError() {
46          return oauthError;
47      }
48  }
