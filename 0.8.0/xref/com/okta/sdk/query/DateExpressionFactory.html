16  package com.okta.sdk.query;
17  
18  import com.okta.sdk.lang.Duration;
19  
20  import java.util.Date;
21  
22  /**
23   * An {@code DateExpressionFactory} creates a single condition (matches) for resource properties of Date type.
24   *
25   * @since 0.5.0
26   */
27  public interface DateExpressionFactory {
28  
29      /**
30       * Convenience method that returns a new equals expression as a String expression, indicating the specified value should match the property value.
31       * Multiple values are allowed, for example:
32       * <pre>
33       * account.createdAt().matches("[,2014-04-05T12:00:00]"): matches all Accounts created between the beginning of time and noon of 2014/04/05
34       * account.createdAt().matches("[2015-01-01, 2015-02-01)"): matches all Accounts created/modified between 2015/01/01 and 2015/02/01, excluding those created on 2015/02/01
35       * account.createdAt().matches("[2015-02-01T12:00:00,]"): matches all Accounts created between noon of 2015-02-01 and now
36       * account.modifiedAt().matches("2015-01"): is equivalent to "[2015-01-01T00:00:00.000Z,2015-02-01T00:00:00.000Z)" and matches all Accounts modified during January 2015
37       * account.modifiedAt().matches("2015-01-03T11:23:14.233"): matches all Accounts modified only at that exact instant represented by the ISO-8601 Timestamp
38       * </pre>
39       *
40       * @param value the value that should equal the property value (ignoring case).
41       * @return a new case-insensitive equals expression reflecting the property name and the specified value.
42       */
43      Criterion matches(String value);
44  
45      /**
46       * Returns a new expression indicating the property value must be greater than the specified date.
47       * For example:
48       * <pre>
49       * application.createdAt().gt(new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss").parse("2014-01-01T12:00:00")): matches only those Applications created after 2014/01/01 at noon
50       * </pre>
51       *
52       * @param date the {@link Date} to compare the property value
53       * @return a new case-insensitive expression reflecting the property name and the specified value.
54       */
55      Criterion gt(Date date);
56  
57      /**
58       * Returns a new expression indicating the property value must be greater than or equal to the specified date.
59       * For example:
60       * <pre>
61       * application.createdAt().gt(new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss").parse("2014-01-01T12:00:00")): matches those applications created after or exactly at noon of 2014/01/01
62       * </pre>
63       *
64       * @param date the {@link Date} to compare the property value
65       * @return a new case-insensitive expression reflecting the property name and the specified value.
66       */
67      Criterion gte(Date date);
68  
69      /**
70       * Returns a new expression indicating the property value must be less than the specified date.
71       * For example:
72       * <pre>
73       * application.createdAt().lt(new Date()): matches only those applications created before the instant specified by {@code new Date()}
74       * </pre>
75       *
76       * @param date the {@link Date} to compare the property value
77       * @return a new case-insensitive expression reflecting the property name and the specified value.
78       */
79      Criterion lt(Date date);
80  
81      /**
82       * Returns a new expression indicating the specified date must be greater than or equal to the property value.
83       * For example:
84       * <pre>
85       * application.createdAt().gte(new Date()): matches those applications created before or at the exact instant specified by {@code new Date()}
86       * </pre>
87       *
88       * @param date the {@link Date} to compare the property value
89       * @return a new case-insensitive expression reflecting the property name and the specified value.
90       */
91      Criterion lte(Date date);
92  
93      /**
94       * Returns a new expression indicating the specified date must be equal to the property value.
95       * For example:
96       * <pre>
97       * application.createdAt().gte(new Date()): matches only those applications created at the exact instant specified by {@code new Date()}
98       * </pre>
99       *
100      * @param date the {@link Date} to compare the property value
101      * @return a new case-insensitive expression reflecting the property name and the specified value.
102      */
103     Criterion equals(Date date);
104 
105     /**
106      * Returns a new expression indicating the property value must belong to the range specified by the {@code begin} and {@code end} dates
107      * Using "in" is equivalent to gte(begin).lt(end) where begin time is inclusive and end time is exclusive.
108      * For example:
109      * <pre>
110      * application.modifiedAt().in(new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss").parse("2014-01-01T12:00:00"), new Date()) matches those applications modified after or exactly at the noon of 2014/01/01 and before the exact instant represented by {@code new Date()}
111      * </pre>
112      *
113      * @param begin the {@link Date} to use as the range start
114      * @param end a {@link Date}  to use as the range end
115      * @return a new case-insensitive expression reflecting the property name and the specified value.
116      */
117     Criterion in(Date begin, Date end);
118 
119     /**
120      * This is a convenience method used to calculate the end timestamp for the range based on the {@code begin} date and the specified {@link Duration}
121      *
122      * @param begin the {@link Date} to use as the range start
123      * @param duration the {@link Duration} used to calculate the end timestamp based on the value specified by {@code begin}
124      * @return a Criterion representing a given date range.
125      */
126     Criterion in(Date begin, Duration duration);
127 }
