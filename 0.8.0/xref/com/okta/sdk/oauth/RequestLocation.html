16  package com.okta.sdk.oauth;
17  
18  /**
19   * The possible locations in an HTTP request where an OAuth 2 bearer token may be found and used for authenticating
20   * the request.  By default, the SDK will only inspect the {@link #HEADER} and {@link #BODY} during a request; using
21   * request parameters for authentication is generally discouraged for security reasons.  That being said, if you have a
22   * specific need for using request parameters, such as supporting a legacy HTTP client, you can configure the SDK to
23   * check the {@link #QUERY_PARAM} location as well.  Example configuration below.
24   *
25   * <h3>Usage</h3>
26   *
27   * <p>When you accept a known OAuth HTTP request, you can specify the request locations that will be checked when
28   * retrieving the OAuth Access Token used to authenticate the request.  For example:</p>
29   *
30   * <pre>
31   * import static com.okta.sdk.oauth.RequestLocation.*;
32   * ...
33   *
34   * application.authenticateOauth(httpRequest).<b>{@code OAuthApiRequestAuthenticator.inLocation(RequestLocation...) inLocation}</b>(HEADER, BODY, QUERY_PARAM).execute();
35   * </pre>
36   *
37   * <p>
38   * Again, by default, the SDK will automatically inspect the {@link #HEADER} and {@link #BODY}.  You must explicitly
39   * add {@link #QUERY_PARAM} if you wish to use query parameters for authentication, as query parameters are generally
40   * not considered as secure as the other two locations.
41   * </p>
42   *
43   * @since 0.5.0
44   */
45  public enum RequestLocation {
46  
47      /**
48       * Try to find the OAuth 2 bearer token in the the HTTP request's {@code Authorization} header.  This is an option
49       * defined in the <a href="http://tools.ietf.org/html/rfc6750#section-2.1">OAuth 2 Bearer Token
50       * specification, Section 2.1</a>.
51       */
52      HEADER,
53  
54      /**
55       * Try to find the OAuth 2 bearer token in the HTTP request's {@code application/x-www-form-urlencoded} body as a
56       * {@code access_token} parameter.  This is an option defined in the
57       * <a href="http://tools.ietf.org/html/rfc6750#section-2.2">OAuth 2 Bearer Token specification, Section 2.2</a>.
58       */
59      BODY,
60  
61      /**
62       * Try to find the OAuth 2 bearer token in an HTTP request {@code access_token} query parameter.  This is an option
63       * defined in the <a href="http://tools.ietf.org/html/rfc6750#section-2.3">OAuth 2 Bearer Token
64       * specification, Section 2.3</a>.
65       * <p>Unlike the {@link #HEADER} and {@link #BODY} locations, <b>this location is NOT
66       * checked by default</b>, as query parameters are generally perceived as less secure than the other two
67       * locations.  If you wish to also inspect request query parameters for authenticating the OAuth request, this
68       * option must be configured explicitly, as shown in this enum's top-level JavaDoc.</p>
69       */
70      QUERY_PARAM
71  }
