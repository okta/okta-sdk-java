16  package com.okta.sdk.impl.resource;
17  
18  import com.okta.commons.lang.Collections;
19  import com.okta.commons.lang.Strings;
20  import com.okta.sdk.helper.PolicyApiHelper;
21  import com.okta.sdk.resource.client.ApiException;
22  import com.okta.sdk.resource.api.PolicyApi;
23  import com.okta.sdk.resource.model.*;
24  import com.okta.sdk.resource.policy.OktaSignOnPolicyBuilder;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  import java.util.Objects;
29  
30  public class DefaultOktaSignOnPolicyBuilder extends DefaultPolicyBuilder<OktaSignOnPolicyBuilder> implements OktaSignOnPolicyBuilder {
31  
32      private List<String> groupIds = new ArrayList<>();
33      private List<String> userIds = new ArrayList<>();
34  
35      @Override
36      public OktaSignOnPolicyBuilder setGroups(List<String> groupIds) {
37          this.groupIds = groupIds;
38          return this;
39      }
40  
41      @Override
42      public OktaSignOnPolicyBuilder addGroup(String groupId) {
43          this.groupIds.add(groupId);
44          return this;
45      }
46  
47      @Override
48      public OktaSignOnPolicyBuilder setUsers(List<String> userIds) {
49          this.userIds = userIds;
50          return this;
51      }
52  
53      @Override
54      public OktaSignOnPolicyBuilder addUser(String userId) {
55          this.userIds.add(userId);
56          return this;
57      }
58  
59      @Override
60      public OktaSignOnPolicy buildAndCreate(PolicyApi client) throws ApiException {
61          return new PolicyApiHelper<>(new PolicyApiHelper<>(client.getApiClient()))
62              .createPolicyOfType(OktaSignOnPolicy.class, build(), isActive);
63      }
64  
65      private OktaSignOnPolicy build() {
66          OktaSignOnPolicy policy = new OktaSignOnPolicy();
67  
68          if (Strings.hasText(name)) policy.setName(name);
69          if (Strings.hasText(description)) policy.setDescription(description);
70          if (priority != null)
71              policy.setPriority(priority);
72  
73          if (PolicyType.OKTA_SIGN_ON.equals(policyType))
74              policy.setType(policyType);
75          else
76              throw new IllegalArgumentException("PolicyType should be 'OKTA_SIGN_ON', please use PolicyBuilder for other policy types.");
77  
78          if (Objects.nonNull(status))
79              policy.setStatus(status);
80  
81          policy.setConditions( new OktaSignOnPolicyConditions());
82          OktaSignOnPolicyConditions oktaSignOnPolicyConditions = policy.getConditions();
83  
84          if (!Collections.isEmpty(groupIds)) {
85              GroupCondition groupCondition = new GroupCondition();
86              groupCondition.setInclude(groupIds);
87              PolicyPeopleCondition policyPeopleCondition = new PolicyPeopleCondition();
88              policyPeopleCondition.setGroups(groupCondition);
89              oktaSignOnPolicyConditions.setPeople(policyPeopleCondition);
90          }
91  
92          if (!Collections.isEmpty(userIds)) {
93              UserCondition userCondition = new UserCondition();
94              userCondition.setInclude(userIds);
95              PolicyPeopleCondition policyPeopleCondition = new PolicyPeopleCondition();
96              policyPeopleCondition.setUsers(userCondition);
97              oktaSignOnPolicyConditions.setPeople(policyPeopleCondition);
98          }
99  
100         return policy;
101     }
102 }
