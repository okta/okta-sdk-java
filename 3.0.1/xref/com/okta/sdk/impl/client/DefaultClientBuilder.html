68  /**
69   * <p>The default {@link ClientBuilder} implementation. This looks for configuration files
70   * in the following locations and order of precedence (last one wins).</p>
71   * <ul>
72   * <li>classpath:com/okta/sdk/config/okta.properties</li>
73   * <li>classpath:com/okta/sdk/config/okta.yaml</li>
74   * <li>classpath:okta.properties</li>
75   * <li>classpath:okta.yaml</li>
76   * <li>~/.okta/okta.yaml</li>
77   * <li>Environment Variables (with dot notation converted to uppercase + underscores)</li>
78   * <li>System Properties</li>
79   * <li>Programmatically</li>
80   * </ul>
81   *
82   * Please be aware that, in general, loading secrets (such as api-keys or PEM-content) from environment variables
83   * or system properties can lead to those secrets being leaked.
84   *
85   * @since 0.5.0
86   */

109     DefaultClientBuilder(ResourceFactory resourceFactory) {
110         Collection<PropertiesSource> sources = new ArrayList<>();
111 
112         for (String location : configSources()) {
113 
114             if (ENVVARS_TOKEN.equalsIgnoreCase(location)) {
115                 sources.add(EnvironmentVariablesPropertiesSource.oktaFilteredPropertiesSource());
116             }
117             else if (SYSPROPS_TOKEN.equalsIgnoreCase(location)) {
118                 sources.add(SystemPropertiesSource.oktaFilteredPropertiesSource());
119             }
120             else {
121                 Resource resource = resourceFactory.createResource(location);
122 
123                 PropertiesSource wrappedSource;
124                 if (Strings.endsWithIgnoreCase(location, ".yaml")) {
125                     wrappedSource = new YAMLPropertiesSource(resource);
126                 } else {
127                     wrappedSource = new ResourcePropertiesSource(resource);
128                 }
129 
130                 PropertiesSource propertiesSource = new OptionalPropertiesSource(wrappedSource);
131                 sources.add(propertiesSource);
132             }
133         }
134 
135         Map<String, String> props = new LinkedHashMap<>();
136 
137         for (PropertiesSource source : sources) {
138             Map<String, String> srcProps = source.getProperties();
139             props.putAll(srcProps);
140         }
141 
142         // check to see if property value is null before setting value
143         // if != null, allow it to override previously set values
144         if (Strings.hasText(props.get(DEFAULT_CLIENT_API_TOKEN_PROPERTY_NAME))) {
145             clientConfig.setApiToken(props.get(DEFAULT_CLIENT_API_TOKEN_PROPERTY_NAME));
146         }
147 
148         if (Strings.hasText(props.get(DEFAULT_CLIENT_CACHE_ENABLED_PROPERTY_NAME))) {
149             clientConfig.setCacheManagerEnabled(Boolean.valueOf(props.get(DEFAULT_CLIENT_CACHE_ENABLED_PROPERTY_NAME)));
150         }
151 
152         if (Strings.hasText(props.get(DEFAULT_CLIENT_CACHE_TTL_PROPERTY_NAME))) {
153             clientConfig.setCacheManagerTtl(Long.parseLong(props.get(DEFAULT_CLIENT_CACHE_TTL_PROPERTY_NAME)));
154         }
155 
156         if (Strings.hasText(props.get(DEFAULT_CLIENT_CACHE_TTI_PROPERTY_NAME))) {
157             clientConfig.setCacheManagerTti(Long.parseLong(props.get(DEFAULT_CLIENT_CACHE_TTI_PROPERTY_NAME)));
158         }
159 
160         for (String prop : props.keySet()) {
161             boolean isPrefix = prop.length() == DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME.length();
162             if (!isPrefix && prop.startsWith(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME)) {
163                 // get class from prop name
164                 String cacheClass = prop.substring(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME.length() + 1, prop.length() - 4);
165                 String cacheTti = props.get(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME + "." + cacheClass + ".tti");
166                 String cacheTtl = props.get(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME + "." + cacheClass + ".ttl");
167                 CacheConfigurationBuilder cacheBuilder = Caches.forResource(Classes.forName(cacheClass));
168                 if (Strings.hasText(cacheTti)) {
169                     cacheBuilder.withTimeToIdle(Long.parseLong(cacheTti), TimeUnit.SECONDS);
170                 }
171                 if (Strings.hasText(cacheTtl)) {
172                     cacheBuilder.withTimeToLive(Long.parseLong(cacheTtl), TimeUnit.SECONDS);
173                 }
174                 if (!clientConfig.getCacheManagerCaches().containsKey(cacheClass)) {
175                     clientConfig.getCacheManagerCaches().put(cacheClass, cacheBuilder);
176                 }
177             }
178         }
179 
180         if (Strings.hasText(props.get(DEFAULT_CLIENT_TESTING_DISABLE_HTTPS_CHECK_PROPERTY_NAME))) {
181             allowNonHttpsForTesting = Boolean.valueOf(props.get(DEFAULT_CLIENT_TESTING_DISABLE_HTTPS_CHECK_PROPERTY_NAME));
182         }
183 
184         if (Strings.hasText(props.get(DEFAULT_CLIENT_ORG_URL_PROPERTY_NAME))) {
185             String baseUrl = props.get(DEFAULT_CLIENT_ORG_URL_PROPERTY_NAME);
186             // remove backslashes that can end up in file when it's written programmatically, e.g. in a test
187             baseUrl = baseUrl.replace("\\:", ":");
188             ConfigurationValidator.assertOrgUrl(baseUrl, allowNonHttpsForTesting);
189             clientConfig.setBaseUrl(baseUrl);
190         }
191 
192         if (Strings.hasText(props.get(DEFAULT_CLIENT_CONNECTION_TIMEOUT_PROPERTY_NAME))) {
193             clientConfig.setConnectionTimeout(Integer.parseInt(props.get(DEFAULT_CLIENT_CONNECTION_TIMEOUT_PROPERTY_NAME)));
194         }
195 
196         if (Strings.hasText(props.get(DEFAULT_CLIENT_AUTHENTICATION_SCHEME_PROPERTY_NAME))) {
197             clientConfig.setAuthenticationScheme(Enum.valueOf(AuthenticationScheme.class, props.get(DEFAULT_CLIENT_AUTHENTICATION_SCHEME_PROPERTY_NAME)));
198         }
199 
200         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_PORT_PROPERTY_NAME))) {
201             clientConfig.setProxyPort(Integer.parseInt(props.get(DEFAULT_CLIENT_PROXY_PORT_PROPERTY_NAME)));
202         }
203 
204         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_HOST_PROPERTY_NAME))) {
205             clientConfig.setProxyHost(props.get(DEFAULT_CLIENT_PROXY_HOST_PROPERTY_NAME));
206         }
207 
208         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_USERNAME_PROPERTY_NAME))) {
209             clientConfig.setProxyUsername(props.get(DEFAULT_CLIENT_PROXY_USERNAME_PROPERTY_NAME));
210         }
211 
212         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_PASSWORD_PROPERTY_NAME))) {
213             clientConfig.setProxyPassword(props.get(DEFAULT_CLIENT_PROXY_PASSWORD_PROPERTY_NAME));
214         }
215 
216         if (Strings.hasText(props.get(DEFAULT_CLIENT_AUTHORIZATION_MODE_PROPERTY_NAME))) {
217             clientConfig.setAuthorizationMode(AuthorizationMode.getAuthorizationMode(props.get(DEFAULT_CLIENT_AUTHORIZATION_MODE_PROPERTY_NAME)));
218         }
219 
220         if (Strings.hasText(props.get(DEFAULT_CLIENT_ID_PROPERTY_NAME))) {
221             clientConfig.setClientId(props.get(DEFAULT_CLIENT_ID_PROPERTY_NAME));
222         }
223 
224         if (Strings.hasText(props.get(DEFAULT_CLIENT_SCOPES_PROPERTY_NAME))) {
225             Set<String> scopes = new HashSet<>(Arrays.asList(props.get(DEFAULT_CLIENT_SCOPES_PROPERTY_NAME).split("[\\s,]+")));
226             clientConfig.setScopes(scopes);
227         }
228 
229         if (Strings.hasText(props.get(DEFAULT_CLIENT_PRIVATE_KEY_PROPERTY_NAME))) {
230             clientConfig.setPrivateKey(props.get(DEFAULT_CLIENT_PRIVATE_KEY_PROPERTY_NAME));
231         }
232 
233         if (Strings.hasText(props.get(DEFAULT_CLIENT_REQUEST_TIMEOUT_PROPERTY_NAME))) {
234             clientConfig.setRetryMaxElapsed(Integer.parseInt(props.get(DEFAULT_CLIENT_REQUEST_TIMEOUT_PROPERTY_NAME)));
235         }
236 
237         if (Strings.hasText(props.get(DEFAULT_CLIENT_RETRY_MAX_ATTEMPTS_PROPERTY_NAME))) {
238             clientConfig.setRetryMaxAttempts(Integer.parseInt(props.get(DEFAULT_CLIENT_RETRY_MAX_ATTEMPTS_PROPERTY_NAME)));
239         }
240     }

304     @Override
305     public Client build() {
306         if (!this.clientConfig.isCacheManagerEnabled()) {
307             log.debug("CacheManager disabled. Defaulting to DisabledCacheManager");
308             this.cacheManager = Caches.newDisabledCacheManager();
309         } else if (this.cacheManager == null) {
310             log.debug("No CacheManager configured. Defaulting to in-memory CacheManager with default TTL and TTI of five minutes.");
311 
312             CacheManagerBuilder cacheManagerBuilder = Caches.newCacheManager()
313                     .withDefaultTimeToIdle(this.clientConfig.getCacheManagerTti(), TimeUnit.SECONDS)
314                     .withDefaultTimeToLive(this.clientConfig.getCacheManagerTtl(), TimeUnit.SECONDS);
315             if (this.clientConfig.getCacheManagerCaches().size() > 0) {
316                 for (CacheConfigurationBuilder builder : this.clientConfig.getCacheManagerCaches().values()) {
317                     cacheManagerBuilder.withCache(builder);
318                 }
319             }
320 
321             this.cacheManager = cacheManagerBuilder.build();
322         }
323 
324         if (this.clientConfig.getBaseUrlResolver() == null) {
325             ConfigurationValidator.assertOrgUrl(this.clientConfig.getBaseUrl(), allowNonHttpsForTesting);
326             this.clientConfig.setBaseUrlResolver(new DefaultBaseUrlResolver(this.clientConfig.getBaseUrl()));
327         }
328 
329         if (!isOAuth2Flow()) {
330             if (this.clientConfig.getClientCredentialsResolver() == null && this.clientCredentials != null) {
331                 this.clientConfig.setClientCredentialsResolver(new DefaultClientCredentialsResolver(this.clientCredentials));
332             } else if (this.clientConfig.getClientCredentialsResolver() == null) {
333                 this.clientConfig.setClientCredentialsResolver(new DefaultClientCredentialsResolver(this.clientConfig));
334             }
335         } else {
336             this.clientConfig.setAuthenticationScheme(AuthenticationScheme.OAUTH2_PRIVATE_KEY);
337 
338             validateOAuth2ClientConfig(this.clientConfig);
339 
340             accessTokenRetrieverService = new AccessTokenRetrieverServiceImpl(clientConfig);
341 
342             OAuth2ClientCredentials oAuth2ClientCredentials =
343                 new OAuth2ClientCredentials(accessTokenRetrieverService);
344 
345             this.clientConfig.setClientCredentialsResolver(new DefaultClientCredentialsResolver(oAuth2ClientCredentials));
346         }
347 
348         return new DefaultClient(clientConfig, cacheManager);
349     }

351     /**
352      * @since 1.6.0
353      */
354     private void validateOAuth2ClientConfig(ClientConfiguration clientConfiguration) {
355         Assert.notNull(clientConfiguration.getClientId(), "clientId cannot be null");
356         Assert.isTrue(clientConfiguration.getScopes() != null && !clientConfiguration.getScopes().isEmpty(),
357             "At least one scope is required");
358         String privateKey = clientConfiguration.getPrivateKey();
359         Assert.hasText(privateKey, "privateKey cannot be null (either PEM file path (or) full PEM content must be supplied)");
360 
361         if (!ConfigUtil.hasPrivateKeyContentWrapper(privateKey)) {
362             // privateKey is a file path, check if the file exists
363             Path privateKeyPemFilePath;
364             try {
365                 privateKeyPemFilePath = Paths.get(privateKey);
366             } catch (InvalidPathException ipe) {
367                 throw new IllegalArgumentException("Invalid privateKey file path", ipe);
368             }
369             boolean privateKeyPemFileExists = Files.exists(privateKeyPemFilePath, LinkOption.NOFOLLOW_LINKS);
370             Assert.isTrue(privateKeyPemFileExists, "privateKey file does not exist");
371         }
372     }

424     @Override
425     public ClientBuilder setPrivateKey(PrivateKey privateKey) {
426         if (isOAuth2Flow()) {
427             Assert.notNull(privateKey, "Missing privateKeyFile");
428             String algorithm = privateKey.getAlgorithm();
429             if (algorithm.equals("RSA")) {
430                 String encodedString = ConfigUtil.RSA_PRIVATE_KEY_HEADER + "\n"
431                     + Base64.getEncoder().encodeToString(privateKey.getEncoded()) + "\n"
432                     + ConfigUtil.RSA_PRIVATE_KEY_FOOTER;
433                 this.clientConfig.setPrivateKey(encodedString);
434             } else if(algorithm.equals("EC")) {
435                 String encodedString = ConfigUtil.EC_PRIVATE_KEY_HEADER + "\n"
436                     + Base64.getEncoder().encodeToString(privateKey.getEncoded()) + "\n"
437                     + ConfigUtil.EC_PRIVATE_KEY_FOOTER;
438                 this.clientConfig.setPrivateKey(encodedString);
439             } else {
440                 throw new IllegalArgumentException("Supplied privateKey is not an RSA or EC key - " + algorithm);
441             }
442         }
443         return this;
444     }

446     private String getFileContent(File file) {
447         try (InputStream inputStream = new FileInputStream(file)) {
448             return readFromInputStream(inputStream);
449         } catch (IOException e) {
450             throw new IllegalArgumentException("Could not read from supplied private key file");
451         }
452     }

454     private String getFileContent(Path path) {
455         Assert.notNull(path, "The path to the privateKey cannot be null.");
456         return getFileContent(path.toFile());
457     }

459     private String getFileContent(InputStream privateKeyStream) {
460         try {
461             return readFromInputStream(privateKeyStream);
462         } catch (IOException e) {
463             throw new IllegalArgumentException("Could not read from supplied privateKey input stream");
464         }
465     }
466 
467     private String readFromInputStream(InputStream inputStream) throws IOException {
468         Assert.notNull(inputStream, "InputStream cannot be null.");
469         StringBuilder resultStringBuilder = new StringBuilder();
470         try (BufferedReader br = new BufferedReader(new InputStreamReader(
471             inputStream, Charset.forName(StandardCharsets.UTF_8.name())))) {
472             String line;
473             while ((line = br.readLine()) != null) {
474                 resultStringBuilder.append(line).append("\n");
475             }
476         }
477         return resultStringBuilder.toString();
478     }
