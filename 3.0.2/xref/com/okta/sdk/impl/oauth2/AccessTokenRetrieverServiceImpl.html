125     /**
126      * Create signed JWT string with the supplied token client configuration details.
127      *
128      * @return signed JWT string
129      * @throws InvalidKeyException
130      * @throws IOException
131      */
132     String createSignedJWT() throws InvalidKeyException, IOException {
133         String clientId = tokenClientConfiguration.getClientId();
134         PrivateKey privateKey = parsePrivateKey(getPemReader());
135         Instant now = Instant.now();
136 
137         String jwt = Jwts.builder()
138             .setAudience(tokenClientConfiguration.getBaseUrl() + TOKEN_URI)
139             .setIssuedAt(Date.from(now))
140             .setExpiration(Date.from(now.plus(1L, ChronoUnit.HOURS)))
141             .setIssuer(clientId)
142             .setSubject(clientId)
143             .claim("jti", UUID.randomUUID().toString())
144             .signWith(privateKey)
145             .compact();
146 
147         return jwt;
148     }

150     /**
151      * Parse private key from the supplied configuration.
152      *
153      * @param pemReader a {@link Reader} that has access to a full PEM resource
154      * @return {@link PrivateKey}
155      * @throws IOException
156      * @throws InvalidKeyException
157      */
158     PrivateKey parsePrivateKey(Reader pemReader) throws IOException, InvalidKeyException {
159 
160         PrivateKey privateKey = getPrivateKeyFromPEM(pemReader);
161         String algorithm = privateKey.getAlgorithm();
162 
163         if (!algorithm.equals("RSA") &&
164             !algorithm.equals("EC")) {
165             throw new InvalidKeyException("Supplied privateKey is not an RSA or EC key - " + algorithm);
166         }
167 
168         return privateKey;
169     }

171     private Reader getPemReader() throws IOException {
172         String privateKey = tokenClientConfiguration.getPrivateKey();
173         if (ConfigUtil.hasPrivateKeyContentWrapper(privateKey)) {
174             return new StringReader(privateKey);
175         } else {
176             return Files.newBufferedReader(Paths.get(privateKey), Charset.defaultCharset());
177         }
178     }

180     /**
181      * Get Private key from input PEM file.
182      *
183      * @param reader
184      * @return {@link PrivateKey}
185      * @throws IOException
186      */
187     PrivateKey getPrivateKeyFromPEM(Reader reader) throws IOException {
188         PrivateKey privateKey;
189 
190         try (PEMParser pemParser = new PEMParser(reader)) {
191             JcaPEMKeyConverter jcaPEMKeyConverter = new JcaPEMKeyConverter();
192             Object pemContent = pemParser.readObject();
193 
194             if (pemContent == null) {
195                 throw new IllegalArgumentException("Invalid Private Key PEM file");
196             }
197 
198             if (pemContent instanceof PEMKeyPair) {
199                 PEMKeyPair pemKeyPair = (PEMKeyPair) pemContent;
200                 KeyPair keyPair = jcaPEMKeyConverter.getKeyPair(pemKeyPair);
201                 privateKey = keyPair.getPrivate();
202             } else if (pemContent instanceof PrivateKeyInfo) {
203                 PrivateKeyInfo privateKeyInfo = (PrivateKeyInfo) pemContent;
204                 privateKey = jcaPEMKeyConverter.getPrivateKey(privateKeyInfo);
205             } else {
206                 throw new IllegalArgumentException("Unsupported Private Key format '" +
207                     pemContent.getClass().getSimpleName() + '"');
208             }
209         }
210 
211         return privateKey;
212     }

214     /**
215      * Create token client config from the supplied API client config.
216      *
217      * Token client needs to retry http 401 errors only once which is not the case with the API client.
218      * We therefore effect this token client specific config by setting 'retryMaxElapsed'
219      * & 'retryMaxAttempts' fields.
220      *
221      * @param apiClientConfiguration supplied in the client configuration.
222      * @return ClientConfiguration to be used by token retrieval client.
223      */
224     ClientConfigurationpl/config/ClientConfiguration.html#ClientConfiguration">ClientConfiguration constructTokenClientConfig(ClientConfiguration apiClientConfiguration) {
225         ClientConfigurationConfiguration">ClientConfiguration tokenClientConfiguration = new ClientConfiguration();
226 
227         tokenClientConfiguration.setClientCredentialsResolver(
228             new DefaultClientCredentialsResolver(() -> Optional.empty()));
229 
230         tokenClientConfiguration.setRequestAuthenticator(new DisabledAuthenticator());
231 
232         tokenClientConfiguration.setCacheManagerEnabled(false);
233 
234         if (apiClientConfiguration.getBaseUrlResolver() != null)
235             tokenClientConfiguration.setBaseUrlResolver(apiClientConfiguration.getBaseUrlResolver());
236 
237         if (apiClientConfiguration.getProxy() != null)
238             tokenClientConfiguration.setProxy(apiClientConfiguration.getProxy());
239 
240         tokenClientConfiguration.setAuthenticationScheme(AuthenticationScheme.NONE);
241         tokenClientConfiguration.setAuthorizationMode(AuthorizationMode.get(tokenClientConfiguration.getAuthenticationScheme()));
242         tokenClientConfiguration.setClientId(apiClientConfiguration.getClientId());
243         tokenClientConfiguration.setScopes(apiClientConfiguration.getScopes());
244         tokenClientConfiguration.setPrivateKey(apiClientConfiguration.getPrivateKey());
245 
246         // setting this to '0' will disable this check and only 'retryMaxAttempts' will be effective
247         tokenClientConfiguration.setRetryMaxElapsed(0);
248         // retry only once for token requests (http 401 errors)
249         tokenClientConfiguration.setRetryMaxAttempts(1);
250 
251         return tokenClientConfiguration;
252     }
