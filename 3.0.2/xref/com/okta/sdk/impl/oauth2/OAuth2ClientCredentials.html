16  package com.okta.sdk.impl.oauth2;
17  
18  import com.okta.commons.lang.Assert;
19  import com.okta.sdk.authc.credentials.ClientCredentials;
20  
21  import java.io.IOException;
22  import java.security.InvalidKeyException;
23  
24  /**
25   * This implementation represents client credentials specific to OAuth2 Authentication scheme.
26   *
27   * @since 1.6.0
28   */
29  public class OAuth2ClientCredentials implements ClientCredentials<OAuth2AccessToken> {
30  
31      private OAuth2AccessToken oAuth2AccessToken;
32      private AccessTokenRetrieverService accessTokenRetrieverService;
33  
34      public OAuth2ClientCredentials(AccessTokenRetrieverService accessTokenRetrieverService) {
35          Assert.notNull(accessTokenRetrieverService, "accessTokenRetrieverService must not be null.");
36          this.accessTokenRetrieverService = accessTokenRetrieverService;
37          this.oAuth2AccessToken = eagerFetchOAuth2AccessToken();
38      }
39  
40      private OAuth2AccessToken eagerFetchOAuth2AccessToken() {
41          OAuth2AccessToken accessToken;
42  
43          try {
44              accessToken = accessTokenRetrieverService.getOAuth2AccessToken();
45          } catch (IOException | InvalidKeyException e) {
46              throw new OAuth2TokenRetrieverException("Failed to fetch OAuth2 access token eagerly", e);
47          }
48  
49          if (accessToken == null) {
50              throw new OAuth2TokenRetrieverException("Failed to fetch OAuth2 access token eagerly");
51          }
52  
53          return accessToken;
54      }
55  
56      public OAuth2AccessToken getCredentials() {
57          return oAuth2AccessToken;
58      }
59  
60      public void setCredentials(OAuth2AccessToken oAuth2AccessToken) {
61          this.oAuth2AccessToken = oAuth2AccessToken;
62      }
63  
64      public AccessTokenRetrieverService getAccessTokenRetrieverService() {
65          return accessTokenRetrieverService;
66      }
67  
68      @Override
69      public String toString() {
70          // never ever print the secret
71          return "<OAuth2ClientCredentials>";
72      }
73  }
