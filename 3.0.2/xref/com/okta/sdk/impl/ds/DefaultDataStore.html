184     @SuppressWarnings("unchecked")
185     private ResourceDataResult getResourceData(String href, Class<? extends Resource> clazz, Map<String,?> queryParameters, HttpHeaders httpHeaders) {
186 
187         Assert.hasText(href, "href argument cannot be null or empty.");
188         Assert.notNull(clazz, "Resource class argument cannot be null.");
189 
190         FilterChain chain = new DefaultFilterChain(this.filters, req -> {
191 
192             CanonicalUri uri = req.getUri();
193 
194             Request getRequest = new DefaultRequest(HttpMethod.GET, uri.getAbsolutePath(), uri.getQuery(), req.getHttpHeaders());
195             Response getResponse = execute(getRequest);
196             Map<String,?> body = getBody(getResponse);
197 
198             if (Collections.isEmpty(body)) {
199                 throw new IllegalStateException("Unable to obtain resource data from the API server or from cache.");
200             }
201 
202             return new DefaultResourceDataResult(req.getAction(), uri, req.getResourceClass(), (Map<String,Object>)body);
203         });
204 
205         CanonicalUri uri = canonicalize(href, queryParameters);
206         ResourceDataRequest req = new DefaultResourceDataRequest(ResourceAction.READ, uri, clazz, new HashMap<>(), httpHeaders);
207         return chain.filter(req);
208     }

210     private ResourceAction getPostAction(ResourceDataRequest request, Response response) {
211         int httpStatus = response.getHttpStatus();
212         if (httpStatus == 201) {
213             return ResourceAction.CREATE;
214         }
215         // TODO: verify this is an issue with Okta
216         //Fix for https://github.com/stormpath/stormpath-sdk-java/issues/403
217         if (httpStatus == 200) {
218             return ResourceAction.READ;
219         }
220 
221         return request.getAction();
222     }

270     @SuppressWarnings("unchecked")
271     private <T extends Resource, R extends Resource> R save(String href,
272                                                             final T resource,
273                                                             final T parentResource,
274                                                             HttpHeaders requestHeaders,
275                                                             final Class<? extends R> returnType,
276                                                             final QueryString qs,
277                                                             final boolean create) {
278         Assert.hasText(href, "href argument cannot be null or empty.");
279         Assert.notNull(resource, "resource argument cannot be null.");
280         Assert.notNull(returnType, "returnType class cannot be null.");
281         Assert.isInstanceOf(AbstractResource.class, resource);
282         Assert.isTrue(!CollectionResource.class.isAssignableFrom(resource.getClass()), "Collections cannot be persisted.");
283 
284         final CanonicalUri uri = canonicalize(href, qs);
285         final AbstractResourcekta/sdk/impl/resource/AbstractResource.html#AbstractResource">AbstractResource abstractResource = (AbstractResource) resource;
286         // Most Okta endpoints do not support partial update, we can revisit in the future.
287         final Map<String, Object> props = resourceConverter.convert(abstractResource, false);
288 
289         FilterChain chain = new DefaultFilterChain(this.filters, req -> {
290 
291             CanonicalUri uri1 = req.getUri();
292             String href1 = uri1.getAbsolutePath();
293             QueryString qs1 = uri1.getQuery();
294 
295             HttpHeaders httpHeaders = req.getHttpHeaders();
296 
297             // create == POST
298             HttpMethod method = HttpMethod.POST;
299             if (!create) {
300                 method = HttpMethod.PUT;
301             }
302 
303             InputStream body;
304             long length = 0;
305             if (resource instanceof VoidResource) {
306                 body = new ByteArrayInputStream(new byte[0]);
307             } else {
308                 ByteArrayOutputStream bodyOut = new ByteArrayOutputStream();
309                 mapMarshaller.marshal(bodyOut, req.getData());
310                 body = new ByteArrayInputStream(bodyOut.toByteArray());
311                 length = bodyOut.size();
312             }
313 
314             Request request = new DefaultRequest(method, href1, qs1, httpHeaders, body, length);
315 
316             Response response = execute(request);
317             Map<String, Object> responseBody = getBody(response);
318 
319             if (Collections.isEmpty(responseBody)) {
320                 // Fix for https://github.com/stormpath/stormpath-sdk-java/issues/218
321                 // Okta response with 200 for deactivate requests (i.e. /api/v1/apps/<id>/lifecycle/deactivate)
322                 if (response.getHttpStatus() == 202
323                         || response.getHttpStatus() == 200
324                         || response.getHttpStatus() == 204) {
325                     //202 means that the request has been accepted for processing, but the processing has not been completed. Therefore we do not have a response setBody.
326                     responseBody = java.util.Collections.emptyMap();
327                 } else {
328                     throw new IllegalStateException("Unable to obtain resource data from the API server.");
329                 }
330             }
331 
332             ResourceAction responseAction = getPostAction(req, response);
333 
334             return new DefaultResourceDataResult(responseAction, uri1, returnType, responseBody);
335         });
336 
337         ResourceAction action = create ? ResourceAction.CREATE : ResourceAction.UPDATE;
338 
339         ResourceDataRequest request = new DefaultResourceDataRequest(
340                 action,
341                 uri,
342                 canonicalizeParent(parentResource),
343                 returnType,
344                 getResourceClass(parentResource),
345                 props,
346                 requestHeaders);
347 
348         ResourceDataResult result = chain.filter(request);
349 
350         Map<String,Object> data = result.getData();
351 
352         //ensure the caller's argument is updated with what is returned from the server if the types are the same:
353         if (returnType.isAssignableFrom(abstractResource.getClass())) {
354             abstractResource.setInternalProperties(data);
355         }
356 
357         return resourceFactory.instantiate(returnType, data);
358     }

394     private void doDelete(String resourceHref, Class resourceClass, QueryString qs, HttpHeaders httpHeaders) {
395 
396         Assert.hasText(resourceHref, "This resource does not have an href value, therefore it cannot be deleted.");
397 
398         // if this URL is a partial, then we MUST add the baseUrl
399         final CanonicalUri uri = canonicalize(resourceHref, qs);
400 
401         FilterChain chain = new DefaultFilterChain(this.filters, request -> {
402             Request deleteRequest = new DefaultRequest(HttpMethod.DELETE, uri.getAbsolutePath(), uri.getQuery(), httpHeaders);
403             execute(deleteRequest);
404             //delete requests have HTTP 204 (no content), so just create an empty setBody for the result:
405             return new DefaultResourceDataResult(request.getAction(), request.getUri(), request.getResourceClass(), new HashMap<>());
406         });
407 
408         final CanonicalUri resourceUri = canonicalize(resourceHref, null);
409         ResourceDataRequest request = new DefaultResourceDataRequest(ResourceAction.DELETE, resourceUri, resourceClass, new HashMap<>());
410         chain.filter(request);
411     }

438     private Response execute(Request request) throws ResourceException {
439 
440         applyDefaultRequestHeaders(request);
441 
442         Response response = this.requestExecutor.executeRequest(request);
443         log.trace("Executed HTTP request.");
444 
445         if (requestLog.isTraceEnabled()) {
446             requestLog.trace("Executing request: method: '{}', url: {}", request.getMethod(), request.getResourceUrl());
447         }
448 
449         if (response.isError()) {
450             Map<String, Object> body = getBody(response);
451 
452             String requestId = response.getHeaders().getOktaRequestId();
453 
454             if (Strings.hasText(requestId)) {
455                 body.put(DefaultError.ERROR_ID.getName(), requestId);
456             }
457 
458             throw new ResourceException(new DefaultError(body)
459                                             .setHeaders(response.getHeaders().getXHeaders())
460                                             .setStatus(response.getHttpStatus()));
461         }
462 
463         return response;
464     }

479     protected void applyDefaultRequestHeaders(Request request) {
480         request.getHeaders().setAccept(java.util.Collections.singletonList(MediaType.APPLICATION_JSON));
481 
482         // Get runtime headers from http client
483         Map<String, List<String>> headerMap = HttpHeadersHolder.get();
484         String oktaAgentHeaderName = OKTA_AGENT.toLowerCase(Locale.ENGLISH);
485         if (headerMap != null && headerMap.get(oktaAgentHeaderName) != null) {
486             List<String> oktaAgents = headerMap.get(oktaAgentHeaderName);
487             if (oktaAgents != null && !oktaAgents.isEmpty()) {
488                 String oktaAgent = Strings.arrayToDelimitedString(oktaAgents.toArray(), " ");
489                 applyUserAgent(oktaAgent + " " + USER_AGENT_STRING, request);
490             }
491         } else {
492             applyUserAgent(USER_AGENT_STRING, request);
493         }
494 
495         if (request.getHeaders().getContentType() == null && request.getBody() != null) {
496             // We only add the default content type (application/json) if a content type is not already in the request
497             request.getHeaders().setContentType(MediaType.APPLICATION_JSON);
498         }
499 
500         // all other headers
501         if (headerMap != null) {
502             headerMap.entrySet().stream()
503                 // filter use agent, there is logic to add that above
504                 .filter(entry -> !OKTA_AGENT.equals(entry.getKey()))
505                 .filter(entry -> !USER_AGENT.equals(entry.getKey()))
506                 .forEach(entry -> request.getHeaders().put(entry.getKey(), entry.getValue()));
507         }
508     }

510     private void applyUserAgent(String userAgentString, Request request) {
511 
512         if (request.getHeaders().containsKey(USER_AGENT)) {
513             request.getHeaders().set(OKTA_USER_AGENT, userAgentString);
514         } else {
515             request.getHeaders().set(USER_AGENT, userAgentString);
516         }
517     }

519     private CanonicalUri canonicalize(String href, Map<String,?> queryParams) {
520         href = ensureFullyQualified(href);
521         return DefaultCanonicalUri.create(href, queryParams);
522     }

524     private CanonicalUri canonicalizeParent(Resource parentResource) {
525         if (parentResource != null && parentResource.getResourceHref() != null) {
526             String href = ensureFullyQualified(parentResource.getResourceHref());
527             return DefaultCanonicalUri.create(href, null);
528         }
529         return null;
530     }

532     private String ensureFullyQualified(String href) {
533         String value = href;
534         if (!isFullyQualified(href)) {
535             value = qualify(href);
536         }
537         return value;
538     }
539 
540     private boolean isFullyQualified(String href) {
541 
542         if (href == null || href.length() < 5) {
543             return false;
544         }
545         return href.regionMatches(true, 0, "http", 0, 4);
546     }

548     private String qualify(String href) {
549         StringBuilder sb = new StringBuilder(this.baseUrlResolver.getBaseUrl());
550         if (!href.startsWith("/")) {
551             sb.append("/");
552         }
553         sb.append(href);
554         return sb.toString();
555     }
