16  package com.okta.sdk.impl.retry;
17  
18  import com.okta.commons.lang.Assert;
19  import org.apache.hc.client5.http.impl.DefaultHttpRequestRetryStrategy;
20  import org.apache.hc.core5.concurrent.CancellableDependency;
21  import org.apache.hc.core5.http.*;
22  import org.apache.hc.core5.http.protocol.HttpContext;
23  import org.apache.hc.core5.util.TimeValue;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import javax.net.ssl.SSLException;
28  import java.io.IOException;
29  import java.io.InterruptedIOException;
30  import java.net.ConnectException;
31  import java.net.NoRouteToHostException;
32  import java.net.UnknownHostException;
33  import java.util.*;
34  import java.util.concurrent.TimeUnit;
35  
36  public final class OktaHttpRequestRetryStrategy extends DefaultHttpRequestRetryStrategy {
37  
38      private static final Logger logger = LoggerFactory.getLogger(OktaHttpRequestRetryStrategy.class);
39  
40      private final int maxRetries;
41      private final Set<Class<? extends IOException>> nonRetriableIOExceptionClasses;
42      private final Set<Integer> retriableCodes;
43  
44      public OktaHttpRequestRetryStrategy(int maxRetries, Collection<Class<? extends IOException>> clazzes, Collection<Integer> codes) {
45          Assert.isTrue(maxRetries >= 0, "maxRetries should be >= 0");
46          this.maxRetries = maxRetries;
47          this.nonRetriableIOExceptionClasses = new HashSet<>(clazzes);
48          this.retriableCodes = new HashSet<>(codes);
49      }
50  
51      public OktaHttpRequestRetryStrategy(int maxRetries) {
52          this(maxRetries, Arrays.asList(InterruptedIOException.class,
53                  UnknownHostException.class, ConnectException.class, ConnectionClosedException.class,
54                  NoRouteToHostException.class, SSLException.class),
55              Arrays.asList(429, 503, 504));
56      }
57  
58      @Override
59      public boolean retryRequest(HttpRequest request, IOException exception, int execCount, HttpContext context) {
60          Assert.notNull(request, "request");
61          Assert.notNull(exception, "exception");
62          if (execCount > this.maxRetries) {
63              return false;
64          } else if (this.nonRetriableIOExceptionClasses.contains(exception.getClass())) {
65              return false;
66          } else {
67              Iterator var5 = this.nonRetriableIOExceptionClasses.iterator();
68  
69              Class rejectException;
70              do {
71                  if (!var5.hasNext()) {
72                      if (request instanceof CancellableDependency && ((CancellableDependency) request).isCancelled()) {
73                          return false;
74                      }
75  
76                      return this.handleAsIdempotent(request);
77                  }
78  
79                  rejectException = (Class) var5.next();
80              } while (!rejectException.isInstance(exception));
81  
82              return false;
83          }
84      }
85  
86      @Override
87      public boolean retryRequest(HttpResponse response, int execCount, HttpContext context) {
88          Assert.notNull(response, "response");
89          if (this.maxRetries == 0) {
90              return false;
91          }
92          return execCount <= this.maxRetries && this.retriableCodes.contains(response.getCode());
93      }
94  
95      @Override
96      public TimeValue getRetryInterval(HttpResponse response, int execCount, HttpContext context) {
97          Assert.notNull(response, "response");
98  
99          long delay;
100         if (response.getCode() == 429) {
101             delay = RetryUtil.get429DelayMillis(response);
102         } else {
103             delay = RetryUtil.getDefaultDelayMillis(execCount);
104         }
105 
106         logger.info("Retry # {} after {} ms delay", execCount, delay);
107 
108         return TimeValue.of(delay, TimeUnit.MILLISECONDS);
109     }
110 }
