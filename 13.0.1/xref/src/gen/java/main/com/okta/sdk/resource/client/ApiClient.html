731                                                     return new StringEntity(objectMapper.writeValueAsString(obj), contentType);
732                                                     } catch (JsonProcessingException e) {
733                                                     throw new ApiException(e);
734                                                     }
735                                                     } else if (mimeType.equals(ContentType.MULTIPART_FORM_DATA.getMimeType())) {
736                                                     MultipartEntityBuilder multiPartBuilder = MultipartEntityBuilder.create();
737                                                     for (Entry<String, Object> paramEntry : formParams.entrySet()) {
738                                                     Object value = paramEntry.getValue();
739                                                     if (value instanceof File) {
740                                                     multiPartBuilder.addBinaryBody(paramEntry.getKey(), (File) value);
741                                                     } else if (value instanceof byte[]) {
742                                                     multiPartBuilder.addBinaryBody(paramEntry.getKey(), (byte[]) value);
743                                                     } else {
744                                                     Charset charset = contentType.getCharset();
745                                                     if (charset != null) {
746                                                     ContentType customContentType = ContentType.create(ContentType.TEXT_PLAIN.getMimeType(), charset);
747                                                     multiPartBuilder.addTextBody(paramEntry.getKey(), parameterToString(paramEntry.getValue()),
748                                                     customContentType);
749                                                     } else {
750                                                     multiPartBuilder.addTextBody(paramEntry.getKey(), parameterToString(paramEntry.getValue()));
751                                                     }
752                                                     }
753                                                     }
754                                                     return multiPartBuilder.build();
755                                                     } else if (mimeType.equals(ContentType.APPLICATION_FORM_URLENCODED.getMimeType())) {
756                                                     List<NameValuePair> formValues = new ArrayList<>();
757                                                         for (Entry<String, Object> paramEntry : formParams.entrySet()) {
758                                                         formValues.add(new BasicNameValuePair(paramEntry.getKey(), parameterToString(paramEntry.getValue())));
759                                                         }
760                                                         return new UrlEncodedFormEntity(formValues, contentType.getCharset());
761                                                         } else {
762                                                         // Handle files with unknown content type
763                                                         if (obj instanceof File) {
764                                                         return new FileEntity((File) obj, contentType);
765                                                         } else if (obj instanceof byte[]) {
766                                                         return new ByteArrayEntity((byte[]) obj, contentType);
767                                                         }
768                                                         throw new ApiException("Serialization for content type '" + contentType + "' not supported");
769                                                         }
770                                                         }

772                                                         /**
773                                                         * Deserialize response body to Java object according to the Content-Type.
774                                                         *
775                                                         * @param <T> Type
776                                                             * @param response Response
777                                                             * @param valueType Return type
778                                                             * @return Deserialized object
779                                                             * @throws ApiException API exception
780                                                             * @throws IOException IO exception
781                                                             */
782                                                             @SuppressWarnings("unchecked")
783                                                             public <T> T deserialize(CloseableHttpResponse response, TypeReference<T> valueType) throws ApiException, IOException, ParseException {
784                                                                 if (valueType == null) {
785                                                                 return null;
786                                                                 }
787                                                                 HttpEntity entity = response.getEntity();
788                                                                 Type valueRawType = valueType.getType();
789                                                                 if (valueRawType.equals(byte[].class)) {
790                                                                 return (T) EntityUtils.toByteArray(entity);
791                                                                 } else if (valueRawType.equals(File.class)) {
792                                                                 return (T) downloadFileFromResponse(response);
793                                                                 }
794                                                                 String mimeType = getResponseMimeType(response);
795                                                                 if (mimeType == null || isJsonMime(mimeType)) {
796                                                                 // Assume json if no mime type
797                                                                 // convert input stream to string
798                                                                 java.util.Scanner s = new java.util.Scanner(entity.getContent()).useDelimiter("\\A");
799                                                                 String content = (String) (s.hasNext() ? s.next() : "");
800 
801                                                                 if ("".equals(content)) { // returns null for empty body
802                                                                 return null;
803                                                                 }
804 
805                                                                 T value = objectMapper.readValue(content, valueType);
806                                                                 return value instanceof List ? PagedList.constructPagedList(response, value) : value;
807                                                                 } else if ("text/plain".equalsIgnoreCase(mimeType)) {
808                                                                 // convert input stream to string
809                                                                 java.util.Scanner s = new java.util.Scanner(entity.getContent()).useDelimiter("\\A");
810                                                                 return (T) (s.hasNext() ? s.next() : "");
811                                                                 } else {
812                                                                 throw new ApiException(
813                                                                 "Deserialization for content type '" + mimeType + "' not supported for type '" + valueType + "'",
814                                                                 response.getCode(),
815                                                                 responseHeaders,
816                                                                 EntityUtils.toString(entity)
817                                                                 );
818                                                                 }
819                                                                 }

821                                                                 private File downloadFileFromResponse(CloseableHttpResponse response) throws IOException {
822                                                                 Header contentDispositionHeader = response.getFirstHeader("Content-Disposition");
823                                                                 String contentDisposition = contentDispositionHeader == null ? null : contentDispositionHeader.getValue();
824                                                                 File file = prepareDownloadFile(contentDisposition);

826                                                                 return file;

829                                                                 protected File prepareDownloadFile(String contentDisposition) throws IOException {
830                                                                 String filename = null;
831                                                                 if (contentDisposition != null && !"".equals(contentDisposition)) {
832                                                                 // Get filename from the Content-Disposition header.
833                                                                 Pattern pattern = Pattern.compile("filename=['\"]?([^'\"\\s]+)['\"]?");
834                                                                 Matcher matcher = pattern.matcher(contentDisposition);
835                                                                 if (matcher.find())
836                                                                 filename = matcher.group(1);
837                                                                 }
838 
839                                                                 String prefix;
840                                                                 String suffix = null;
841                                                                 if (filename == null) {
842                                                                 prefix = "download-";
843                                                                 suffix = "";
844                                                                 } else {
845                                                                 int pos = filename.lastIndexOf('.');
846                                                                 if (pos == -1) {
847                                                                 prefix = filename + "-";
848                                                                 } else {
849                                                                 prefix = filename.substring(0, pos) + "-";
850                                                                 suffix = filename.substring(pos);
851                                                                 }
852                                                                 // Files.createTempFile requires the prefix to be at least three characters long
853                                                                 if (prefix.length() < 3)
854                                                                 prefix = "download-";
855                                                                 }
856 
857                                                                 if (tempFolderPath == null)
858                                                                 return Files.createTempFile(prefix, suffix).toFile();
859                                                                 else
860                                                                 return Files.createTempFile(Paths.get(tempFolderPath), prefix, suffix).toFile();
861                                                                 }

863                                                                 /**
864                                                                 * Build full URL by concatenating base path, the given sub path and query parameters.
865                                                                 *
866                                                                 * @param path The sub path
867                                                                 * @param queryParams The query parameters
868                                                                 * @param collectionQueryParams The collection query parameters
869                                                                 * @param urlQueryDeepObject URL query string of the deep object parameters
870                                                                 * @return The full URL
871                                                                 */
872                                                                 private String buildUrl(String path, List<Pair> queryParams, List<Pair> collectionQueryParams, String urlQueryDeepObject) {
873                                                                     String baseURL;
874                                                                     if (serverIndex != null) {
875                                                                     if (serverIndex < 0 || serverIndex >= servers.size()) {
876                                                                     throw new ArrayIndexOutOfBoundsException(String.format(
877                                                                     "Invalid index %d when selecting the host settings. Must be less than %d", serverIndex, servers.size()
878                                                                     ));
879                                                                     }
880                                                                     baseURL = servers.get(serverIndex).URL(serverVariables);
881                                                                     } else {
882                                                                     baseURL = basePath;
883                                                                     }
884 
885                                                                     final StringBuilder url = new StringBuilder();
886                                                                     url.append(baseURL).append(path);
887 
888                                                                     if (queryParams != null && !queryParams.isEmpty()) {
889                                                                     // support (constant) query string in `path`, e.g. "/posts?draft=1"
890                                                                     String prefix = path.contains("?") ? "&" : "?";
891                                                                     for (Pair param : queryParams) {
892                                                                     if (param.getValue() != null) {
893                                                                     if (prefix != null) {
894                                                                     url.append(prefix);
895                                                                     prefix = null;
896                                                                     } else {
897                                                                     url.append("&");
898                                                                     }
899                                                                     String value = parameterToString(param.getValue());
900                                                                     // query parameter value already escaped as part of parameterToPair
901                                                                     url.append(escapeString(param.getName())).append("=").append(value);
902                                                                     }
903                                                                     }
904                                                                     }
905 
906                                                                     if (collectionQueryParams != null && !collectionQueryParams.isEmpty()) {
907                                                                     String prefix = url.toString().contains("?") ? "&" : "?";
908                                                                     for (Pair param : collectionQueryParams) {
909                                                                     if (param.getValue() != null) {
910                                                                     if (prefix != null) {
911                                                                     url.append(prefix);
912                                                                     prefix = null;
913                                                                     } else {
914                                                                     url.append("&");
915                                                                     }
916                                                                     String value = parameterToString(param.getValue());
917                                                                     // collection query parameter value already escaped as part of parameterToPairs
918                                                                     url.append(escapeString(param.getName())).append("=").append(value);
919                                                                     }
920                                                                     }
921                                                                     }
922 
923                                                                     if (urlQueryDeepObject != null && urlQueryDeepObject.length() > 0) {
924                                                                     url.append(url.toString().contains("?") ? "&" : "?");
925                                                                     url.append(urlQueryDeepObject);
926                                                                     }
927 
928                                                                     return url.toString();
929                                                                     }

963                                                                         *
964                                                                         * @param <T> Type

966                                                                             * @param method The request method, one of "GET", "POST", "PUT", and "DELETE"
967                                                                             * @param queryParams The query parameters

969                                                                             * @param urlQueryDeepObject A URL query string for deep object parameters
970                                                                             * @param body The request body object - if it is not binary, otherwise null

981                                                                             public <T> T invokeAPI(

983                                                                                 String method,
984                                                                                 List<Pair> queryParams,
985                                                                                     List<Pair> collectionQueryParams,
986                                                                                         String urlQueryDeepObject,
987                                                                                         Object body,
988                                                                                         Map<String, String> headerParams,
989                                                                                         Map<String, String> cookieParams,
990                                                                                         Map<String, Object> formParams,
991                                                                                         String accept,
992                                                                                         String contentType,
993                                                                                         String[] authNames,
994                                                                                         TypeReference<T> returnType) throws ApiException {
995                                                                                             if (body != null && !formParams.isEmpty()) {
996                                                                                             throw new ApiException("Cannot have body and form params");
997                                                                                             }

1000                                                                                             final String url = buildUrl(path, queryParams, collectionQueryParams, urlQueryDeepObject);

1003                                                                                             builder.setUri(url);

1009                                                                                             builder.addHeader(keyValue.getKey(), keyValue.getValue());
1010                                                                                             }
1011                                                                                             for (Map.Entry<String,String> keyValue : defaultHeaderMap.entrySet()) {
1012                                                                                             if (!headerParams.containsKey(keyValue.getKey())) {
1013                                                                                             builder.addHeader(keyValue.getKey(), keyValue.getValue());
1014                                                                                             }
1015                                                                                             }

1018                                                                                             for (Entry<String, String> keyValue : cookieParams.entrySet()) {
1019                                                                                             store.addCookie(buildCookie(keyValue.getKey(), keyValue.getValue(), builder.getUri()));
1020                                                                                             }
1021                                                                                             for (Entry<String,String> keyValue : defaultCookieMap.entrySet()) {
1022                                                                                             if (!cookieParams.containsKey(keyValue.getKey())) {
1023                                                                                             store.addCookie(buildCookie(keyValue.getKey(), keyValue.getValue(), builder.getUri()));
1024                                                                                             }
1025                                                                                             }
1026 
1027                                                                                             HttpClientContext context = HttpClientContext.create();
1028                                                                                             context.setCookieStore(store);
1029 
1030                                                                                             ContentType contentTypeObj = getContentType(contentType);
1031                                                                                             if (body != null || !formParams.isEmpty()) {
1032                                                                                             if (isBodyAllowed(method)) {
1033                                                                                             // Add entity if we have content and a valid method
1034                                                                                             builder.setEntity(serialize(body, formParams, contentTypeObj));
1035                                                                                             } else {
