100     /**
101      * Private constructor that can create read-only {@code HttpHeader} instances.
102      */
103     private HttpHeaders(Map<String, List<String>> headers, boolean readOnly) {
104         Assert.notNull(headers, "'headers' must not be null");
105         if (readOnly) {
106             Map<String, List<String>> map =
107                     new LinkedCaseInsensitiveMap<>(headers.size(), Locale.ENGLISH);
108             for (Map.Entry<String, List<String>> entry : headers.entrySet()) {
109                 List<String> values = Collections.unmodifiableList(entry.getValue());
110                 map.put(entry.getKey(), values);
111             }
112             this.headers = Collections.unmodifiableMap(map);
113         } else {
114             this.headers = headers;
115         }
116     }

154     /**
155      * Set the list of acceptable {@linkplain java.nio.charset.Charset charsets}, as specified by the {@code Accept-Charset} header.
156      *
157      * @param acceptableCharsets the acceptable charsets
158      */
159     public void setAcceptCharset(List<Charset> acceptableCharsets) {
160         StringBuilder builder = new StringBuilder();
161         for (Iterator<Charset> iterator = acceptableCharsets.iterator(); iterator.hasNext(); ) {
162             Charset charset = iterator.next();
163             builder.append(charset.name().toLowerCase(Locale.ENGLISH));
164             if (iterator.hasNext()) {
165                 builder.append(", ");
166             }
167         }
168         set(ACCEPT_CHARSET, builder.toString());
169     }
170 
171     /**
172      * Return the list of acceptable {@linkplain Charset charsets}, as specified by the {@code Accept-Charset}
173      * header.
174      *
175      * @return the acceptable charsets
176      */
177     public List<Charset> getAcceptCharset() {
178         List<Charset> result = new ArrayList<Charset>();
179         String value = getFirst(ACCEPT_CHARSET);
180         if (value != null) {
181             String[] tokens = value.split(",\\s*");
182             for (String token : tokens) {
183                 int paramIdx = token.indexOf(';');
184                 String charsetName;
185                 if (paramIdx == -1) {
186                     charsetName = token;
187                 } else {
188                     charsetName = token.substring(0, paramIdx);
189                 }
190                 if (!charsetName.equals("*")) {
191                     result.add(Charset.forName(charsetName));
192                 }
193             }
194         }
195         return result;
196     }

207     /**
208      * Return the set of allowed {@link HttpMethod HTTP methods}, as specified by the {@code Allow} header.
209      * <p>Returns an empty set when the allowed methods are unspecified.
210      *
211      * @return the allowed methods
212      */
213     public Set<HttpMethod> getAllow() {
214         String value = getFirst(ALLOW);
215         if (value != null) {
216             List<HttpMethod> allowedMethod = new ArrayList<HttpMethod>(5);
217             String[] tokens = value.split(",\\s*");
218             for (String token : tokens) {
219                 allowedMethod.add(HttpMethod.valueOf(token));
220             }
221             return EnumSet.copyOf(allowedMethod);
222         } else {
223             return EnumSet.noneOf(HttpMethod.class);
224         }
225     }

510     public Map<String, String> getLinkMap() {
511 
512         List<String> links = getLinkHeaders();
513         if (!com.okta.commons.lang.Collections.isEmpty(links)) {
514             List<String> headerValues = headers.get(LINK);
515             return headerValues.stream()
516                     .map(HttpHeaders::parseLinkHeader)
517                     .collect(Collectors.toMap(Link::getRelationType, Link::getHref));
518         }
519         return Collections.emptyMap();
520     }
521 
522     private static Link parseLinkHeader(String rawHeader) {
523         Pattern pattern = Pattern.compile("\\<(.*)\\>;.*rel=\"(.*)\"");
524         Matcher matcher = pattern.matcher(rawHeader);
525         if (matcher.matches()) {
526             return new DefaultLink(matcher.group(2), matcher.group(1));
527         }
528         return null;
529     }

533     private long getFirstDate(String headerName) {
534         String headerValue = getFirst(headerName);
535         if (headerValue == null) {
536             return -1;
537         }
538         for (String dateFormat : DATE_FORMATS) {
539             SimpleDateFormat simpleDateFormat = new SimpleDateFormat(dateFormat, Locale.US);
540             simpleDateFormat.setTimeZone(GMT);
541             try {
542                 return simpleDateFormat.parse(headerValue).getTime();
543             } catch (ParseException e) { // NOPMD
544                 // ignore
545             }
546         }
547         throw new IllegalArgumentException("Cannot parse date value \"" + headerValue +
548                 "\" for \"" + headerName + "\" header");
549     }
550 
551     private void setDate(String headerName, long date) {
552         SimpleDateFormat dateFormat = new SimpleDateFormat(DATE_FORMATS[0], Locale.US);
553         dateFormat.setTimeZone(GMT);
554         set(headerName, dateFormat.format(new Date(date)));
555     }

570     /**
571      * Add the given, single header value under the given name.
572      *
573      * @param headerName  the header name
574      * @param headerValue the header value
575      * @throws UnsupportedOperationException if adding headers is not supported
576      * @see #put(String, List)
577      * @see #set(String, String)
578      */
579     public void add(String headerName, String headerValue) {
580         List<String> headerValues = headers.get(headerName);
581         if (headerValues == null) {
582             headerValues = new LinkedList<String>();
583             this.headers.put(headerName, headerValues);
584         }
585         headerValues.add(headerValue);
586     }

609     public Map<String, String> toSingleValueMap() {
610         LinkedHashMap<String, String> singleValueMap = new LinkedHashMap<String, String>(this.headers.size());
611         for (Map.Entry<String, List<String>> entry : headers.entrySet()) {
612             singleValueMap.put(entry.getKey(), entry.getValue().get(0));
