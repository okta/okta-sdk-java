17  package com.okta.sdk.impl.ds;
18  
19  import com.okta.sdk.lang.Classes;
20  import com.okta.sdk.resource.Resource;
21  
22  import java.lang.reflect.Constructor;
23  import java.util.ArrayList;
24  import java.util.Collections;
25  import java.util.List;
26  import java.util.Map;
27  
28  /**
29   * @since 0.5.0
30   */
31  public class DefaultResourceFactory implements ResourceFactory {
32  
33      private final InternalDataStore dataStore;
34  
35      private static final String BASE_PACKAGE = "com.okta.sdk.";
36      private static final String IMPL_PACKAGE_NAME_FRAGMENT = "impl";
37      private static final String IMPL_PACKAGE_NAME = IMPL_PACKAGE_NAME_FRAGMENT + ".";
38      private static final String IMPL_CLASS_PREFIX = "Default";
39  
40      private final DiscriminatorRegistry registry = new DefaultDiscriminatorRegistry();
41  
42      public DefaultResourceFactory(InternalDataStore dataStore) {
43          this.dataStore = dataStore;
44      }
45  
46      @Override
47      public <T extends Resource> T instantiate(Class<T> clazz, Object... constructorArgs) {
48  
49          if (clazz == null) {
50              throw new NullPointerException("Resource class cannot be null.");
51          }
52  
53          Class<T> classToResolve = clazz;
54  
55          Object[] ctorArgs = createConstructorArgs(constructorArgs);
56  
57          if (ctorArgs.length >= 2 && ctorArgs[1] instanceof Map) {
58  
59              Map data = (Map) ctorArgs[1];
60              if (registry.supportedClass(clazz)) {
61                  classToResolve = registry.resolve(clazz, data);
62              }
63          }
64  
65          Class<T> implClass = getImplementationClass(classToResolve);
66  
67          Constructor<T> ctor;
68  
69          if (ctorArgs.length == 1) {
70              ctor = Classes.getConstructor(implClass, InternalDataStore.class);
71          } else if (ctorArgs.length == 2) {
72              ctor = Classes.getConstructor(implClass, InternalDataStore.class, Map.class);
73          } else {
74              //collection resource - we want to retain the query parameters (3rd ctor argument):
75              ctor = Classes.getConstructor(implClass, InternalDataStore.class, Map.class, Map.class);
76          }
77  
78          return Classes.instantiate(ctor, ctorArgs);
79      }
80  
81      private static <T extends Resource> Class<T> getImplementationClass(Class<T> clazz) {
82          if (clazz.isInterface()) {
83              return convertToImplClass(clazz);
84          }
85          return clazz;
86      }
87  
88      public static <T extends Resource> Class<T> getInterfaceClass(Class<T> clazz) {
89          if (clazz.isInterface()) {
90              return clazz;
91          }
92          return convertToInterfaceClass(clazz);
93      }
94  
95      static <T extends Resource> Class<T> convertToInterfaceClass(Class<T> clazz) {
96          String fqcn = clazz.getName();
97          String afterBase = fqcn.substring(BASE_PACKAGE.length());
98  
99          //e.g. if impl is com.okta.sdk.impl.account.DefaultAccount, 'afterBase' is impl.account.DefaultAccount
100 
101         //split interface simple name and the remainder of the package structure:
102 
103         if (afterBase.startsWith(IMPL_PACKAGE_NAME)) {
104             afterBase = afterBase.substring(IMPL_PACKAGE_NAME.length());
105         }
106 
107         //now afterBase is account.DefaultAccount
108 
109         //now append the intermediate package name to the base package:
110         int index = afterBase.lastIndexOf('.');
111         String beforeSimpleName = afterBase.substring(0, index);
112 
113         String simpleName = clazz.getSimpleName();
114 
115         //strip off any 'Default' prefix:
116         index = simpleName.indexOf(IMPL_CLASS_PREFIX);
117         simpleName = simpleName.substring(index + IMPL_CLASS_PREFIX.length());
118 
119         String ifaceFqcn = BASE_PACKAGE + beforeSimpleName + "." + simpleName;
120 
121         return Classes.forName(ifaceFqcn);
122     }
123 
124     @SuppressWarnings("unchecked")
125     static <T extends Resource> Class<T> convertToImplClass(Class<T> clazz) {
126         String fqcn = clazz.getName();
127 
128         String afterBase = fqcn.substring(BASE_PACKAGE.length());
129         //e.g. if interface is com.okta.sdk.account.Account, 'afterBase' is account.Account
130 
131         //split interface simple name and the remainder of the package structure:
132 
133         //if this is an impl-module-specific class, strip off the existing 'impl.' package:
134         if (afterBase.startsWith(IMPL_PACKAGE_NAME)) {
135             afterBase = afterBase.substring(IMPL_PACKAGE_NAME.length());
136         }
137 
138         int index = afterBase.lastIndexOf('.');
139         String beforeConcreteClassName = afterBase.substring(0, index);
140 
141         String implFqcn = BASE_PACKAGE + IMPL_PACKAGE_NAME_FRAGMENT + "." +
142                 beforeConcreteClassName + "." + IMPL_CLASS_PREFIX + clazz.getSimpleName();
143 
144         return Classes.forName(implFqcn);
145     }
146 
147     private Object[] createConstructorArgs(Object[] existing) {
148         int argsLength = (existing != null ? existing.length : 0);
149         argsLength += 1; //account for the 'DataStore' instance that is required for every implementation.
150 
151         List<Object> args = new ArrayList<Object>(argsLength);
152         args.add(this.dataStore); //always first arg
153         if (existing != null) {
154             Collections.addAll(args, existing);
155         }
156 
157         return args.toArray();
158     }
159 }
