80      public void setInternalProperties(Map<String, Object> properties) {
81          writeLock.lock();
82          try {
83              this.dirtyProperties.clear();
84              this.dirty = false;
85              if (properties != null && !properties.isEmpty()) {
86                  if (this.properties != properties) {
87                      this.properties.clear();
88                      this.properties.putAll(properties);
89                  } else {
90                      this.properties = properties;
91                  }
92                  setResourceHref(hrefResolver.resolveHref(properties, getClass()));
93                  // Don't consider this resource materialized if it is only a reference.  A reference is any object that
94                  // has only one 'href' property.
95  
96                  // TODO: validate this flow
97  //                boolean hrefOnly = this.properties.size() == 1 && this.properties.containsKey(HREF_PROP_NAME);
98  //                this.materialized = !hrefOnly;
99                  this.materialized = this.properties.size() > 0;
100             } else {
101                 this.materialized = false;
102             }
103         } finally {
104             writeLock.unlock();
105         }
106     }

145     public void materialize() {
146         if (this.materialized) {
147             return;
148         }
149         AbstractResource resource = dataStore.getResource(getResourceHref(), getClass());
150         writeLock.lock();
151         try {
152             if (this.properties != resource.properties) {
153                 this.properties.clear();
154                 this.properties.putAll(resource.properties);
155             }
156 
157             //retain dirty properties:
158             this.properties.putAll(this.dirtyProperties);
159 
160             this.materialized = true;
161         } finally {
162             writeLock.unlock();
163         }
164     }

195     public Object getProperty(String name) {
196             //not the href/id, must be a property that requires materialization:
197             if (!isNew() && !isMaterialized()) {
198 
199                 //only materialize if the property hasn't been set previously (no need to execute a server
200                 // request since we have the most recent value already):
201                 boolean present;
202                 readLock.lock();
203                 try {
204                     present = this.dirtyProperties.containsKey(name);
205                 } finally {
206                     readLock.unlock();
207                 }
208 
209                 if (!present) {
210                     //exhausted present properties - we require a server call:
211                     materialize();
212                 }
213             }
214 
215         return readProperty(name);
216     }

218     /**
219      * Returns {@code true} if this resource has a property with the specified name, {@code false} otherwise.
220      *
221      * @param name the name of the property to check for existence
222      * @return {@code true} if this resource has a property with the specified name, {@code false} otherwise.
223      */
224     public boolean hasProperty(String name) {
225         readLock.lock();
226         try {
227             return !this.deletedPropertyNames.contains(name) &&
228                 (this.dirtyProperties.containsKey(name) || this.properties.containsKey(name));
229         } finally {
230             readLock.unlock();
231         }
232     }

234     private Object readProperty(String name) {
235         readLock.lock();
236         try {
237             if (this.deletedPropertyNames.contains(name)) {
238                 return null;
239             }
240             Object value = this.dirtyProperties.get(name);
241             if (value == null) {
242                 value = this.properties.get(name);
243             }
244             return value;
245         } finally {
246             readLock.unlock();
247         }
248     }

256     /**
257      * Use this method and the set the isNullable flag to true, to set the value to
258      * null for the Property. Certain properties can have a value=null in the REST API
259      * and therefore, this method will allow to explicitly do that.
260      * All other overloaded implementations of setProperty method will assume isNullable=false
261      * and therefore setting the value to null by calling those methods, will take no effect and
262      * retain the old/previous value for the property.
263      * @param property the key of the value to set
264      * @param value the actual value to be set
265      * @param dirty mark the object as dirty if true
266      * @param isNullable true if the value can be set to {@code null}
267      */
268     protected void setProperty(Property property, Object value, final boolean dirty, final boolean isNullable) {
269         setProperty(property.getName(), value, dirty, isNullable);
270     }

272     private Object setProperty(String name, Object value, final boolean dirty, final boolean isNullable) {
273         writeLock.lock();
274         Object previous;
275         try {
276             previous = this.dirtyProperties.put(name, value);
277             if (previous == null) {
278                 previous = this.properties.get(name);
279             }
280             this.dirty = dirty;
281 
282             /*
283              * The instance variable "deletedPropertyNames" is overloaded here.
284              * For "CustomData" value=null means that the property/field has been deleted from custom data,
285              * hence it is added to "deletedPropertyNames". See DefaultCustomData.java
286              * In this case, where value=null and the field is nullable, adding it to "deletedPropertyNames" forces
287              * and makes sure that the property is saved with value=null (but not deleted).
288              * e.g. matchingProperty in AccountLinkingPolicy
289              */
290             if (isNullable && value == null) { //fix for https://github.com/stormpath/stormpath-sdk-java/issues/966
291                 this.deletedPropertyNames.add(name);
292             } else {
293                 if (this.deletedPropertyNames.contains(name)) {
294                     this.deletedPropertyNames.remove(name);
295                 }
296             }
297         } finally {
298             writeLock.unlock();
299         }
300         return previous;
301     }

303     @SuppressWarnings("unchecked")
304     protected <T extends Resource> T getResourceProperty(ResourceReference<T> property) {
305         String key = property.getName();
306         Class<T> clazz = property.getType();
307 
308         Object value = getProperty(key);
309         if (value == null) {
310 
311             if (property.isCreateOnAccess()) {
312                 // create the new resource as an empty object so the developer does not need to deal with
313                 // calling .instantiate directly
314                 T resource = dataStore.instantiate(clazz);
315                 setProperty(key, resource, false);
316                 return resource;
317             }
318             return null;
319         }
320 
321         if (clazz.isInstance(value)) {
322             return (T) value;
323         }
324 
325         if (value instanceof Map) {
326             T resource = dataStore.instantiate(clazz, (Map<String, Object>) value);
327 
328             //replace the existing link object (map with an href) with the newly constructed Resource instance.
329             //Don't dirty the instance - we're just swapping out a property that already exists for the materialized version.
330             //let's not materialize internal collection resources, so they are always retrieved from the backend: https://github.com/stormpath/stormpath-sdk-java/issues/160
331             if (!CollectionResource.class.isAssignableFrom(clazz)) {
332                 setProperty(key, resource, false);
333             }
334             return resource;
335         }
336 
337         String msg = "'" + key + "' property value type does not match the specified type.  Specified type: " +
338             clazz.getName() + ".  Existing type: " + value.getClass().getName();
339         msg += (isPrintableProperty(key) ? ".  Value: " + value : ".");
340         throw new IllegalArgumentException(msg);
341     }

344      * Returns the {@link List} property identified by {@code key}
345      * @param property identifier
346      * @return property identified by {@code property}
347      */
348     protected List getResourceListProperty(ResourceListProperty property){
349         List rawList = (List) getProperty(property.getName());
350 
351         return (rawList == null) ? null : (List) rawList.stream()
352                 .map(item -> {
353                     if (property.getType().isInstance(item)) {
354                         return item;
355                     } else {
356                         return dataStore.instantiate(property.getType(), (Map<String, Object>) item);
357                     }
358                 })
359                 .collect(Collectors.toList());
360     }

362     public String toString() {
363         readLock.lock();
364         try {
365             StringBuilder sb = new StringBuilder();
366             for (Map.Entry<String, Object> entry : this.properties.entrySet()) {
367                 if (sb.length() > 0) {
368                     sb.append(", ");
369                 }
370                 String key = entry.getKey();
371                 //prevent printing of any sensitive values:
372                 if (isPrintableProperty(key)) {
373                     sb.append(key).append(": ").append(String.valueOf(entry.getValue()));
374                 }
375             }
376             return sb.toString();
377         } finally {
378             readLock.unlock();
379         }
380     }

402     @Override
403     public boolean equals(Object o) {
404         if (o == null) {
405             return false;
406         }
407         if (this == o) {
408             return true;
409         }
410         if (!o.getClass().equals(getClass())) {
411             return false;
412         }
413         AbstractResource other = (AbstractResource) o;
414         readLock.lock();
415         try {
416             other.readLock.lock();
417             try {
418                 return this.properties.equals(other.properties);
419             } finally {
420                 other.readLock.unlock();
421             }
422         } finally {
423             readLock.unlock();
424         }
425     }

478         Assert.isInstanceOf(String.class, key);
479         writeLock.lock();
480         try {
481             //noinspection SuspiciousMethodCalls
482             Object object = this.dirtyProperties.remove(key);

491     public void putAll(Map<? extends String, ?> m) {
492         if (Collections.isEmpty(m)) {
493             return;
494         }
495         Set<? extends Map.Entry<? extends String, ?>> entrySet = m.entrySet();
496         writeLock.lock();
497         try {
498             for (Map.Entry<? extends String, ?> entry : entrySet) {
499                 setProperty(entry.getKey(), entry.getValue());
500             }
501         } finally {
502             writeLock.unlock();
503         }

506     public void clear() {
507 
508         writeLock.lock();
509         try {
510             Set<String> propertiesToFilter = new HashSet<>(getPropertyDescriptors().keySet());
511             for (String propertyName : this.keySet()) {
512                 if (propertiesToFilter.contains(propertyName)) {
513                     continue;
514                 }
515                 this.dirtyProperties.remove(propertyName);
516                 this.deletedPropertyNames.add(propertyName);
517                 dirty = true;
518             }
519         } finally {
520             writeLock.unlock();
521         }
522     }

524     public Set<String> keySet() {
525         if(!isMaterialized()) {
526             writeLock.lock();
527             try {
528                 materialize();
529             } finally {
530                 writeLock.unlock();
531             }
532         }
533         readLock.lock();
534         try {
535             Set<String> keySet = new LinkedHashSet<String>();
536             keySet.addAll(this.properties.keySet());
537             keySet.addAll(this.dirtyProperties.keySet());
538             keySet.removeAll(this.deletedPropertyNames);
539             return java.util.Collections.unmodifiableSet(keySet);
540         } finally {
541             readLock.unlock();
542         }
543     }

545     public Collection<Object> values() {
546         Set<String> keySet = this.keySet();
547         Collection<Object> values = new ArrayList<>(keySet.size());
548         for (String key : keySet) {
549             values.add(this.get(key));
550         }
551         return java.util.Collections.unmodifiableCollection(values);
552     }

554     public Set<Map.Entry<String, Object>> entrySet() {
555         Set<String> keySet = this.keySet();
556         readLock.lock();
557         try {
558             Set<Map.Entry<String, Object>> entrySet = new LinkedHashSet<Map.Entry<String, Object>>(keySet.size());
559             for(String key : keySet) {
560                 entrySet.add(new AbstractMap.SimpleEntry<String, Object>(key, this.get(key)));
561             }
562             return java.util.Collections.unmodifiableSet(entrySet);
563         } finally {
564             readLock.unlock();
565         }

568     /**
569      * Parses path parameters from URL based on a template string.
570      * <p>
571      * Example:
572      * For the template string: {@code /users/{userId}/favorite/{iceCream}}, and the href of this resource:
573      * {@code /users/42/favorite/vanilla}}, the resulting map would be: {@code [userId: 42, iceCream: vanilla]}.
574      *
575      * @param templateUrl template of the URL to parse
576      * @return a map of path parameters
577      * @since 0.8.0
578      */
579     protected Map<String, String> getParamsFromHref(String templateUrl) {
580         Assert.hasText(href, "Resource 'href' must be set before attempting to call an operation.");
581         Assert.hasText(templateUrl, "Template URL must not be empty.");
582         Map<String, String> resultMap = new HashMap<>();
583 
584         try {
585             String hrefPath = new URI(href).getPath();
586 
587             Iterator<String> templatePartsIter = Arrays.asList(templateUrl.split("/")).iterator();
588             Iterator<String> hrefPartsIter = Arrays.asList(hrefPath.split("/")).iterator();
589 
590             while (templatePartsIter.hasNext() && hrefPartsIter.hasNext()) {
591                 String templatePart = templatePartsIter.next();
592                 String hrefPart = hrefPartsIter.next();
593 
594                 // if the template string starts with '{' we have named path param
595                 if (templatePart.startsWith("{") && templatePart.endsWith("}")) {
596                     String paramName = templatePart.substring(1,templatePart.length()-1);
597                     resultMap.put(paramName, hrefPart);
598                 } else if (!Objects.equals(templatePart, hrefPart)) {
599                     // otherwise break, the template and the href no longer match
600                     break;
601                 }
602                 // otherwise the parts match, just continue
603             }
604         } catch (URISyntaxException e) {
605             throw new URIParseException("Invalid URI for resource: " + href, e);
606         }
607         return resultMap;
608     }
