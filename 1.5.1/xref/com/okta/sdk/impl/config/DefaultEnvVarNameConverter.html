17  package com.okta.sdk.impl.config;
18  
19  import com.okta.commons.lang.Assert;
20  import com.okta.commons.lang.Strings;
21  import com.okta.sdk.client.ClientBuilder;
22  
23  import java.util.Arrays;
24  import java.util.Map;
25  import java.util.stream.Collectors;
26  
27  public class DefaultEnvVarNameConverter implements EnvVarNameConverter {
28  
29      private final Map<String, String> envToDotPropMap;
30  
31      public DefaultEnvVarNameConverter() {
32  
33          // this dependency on ClientBuilder isn't great, in the future we can change the API to ONLY support a one way conversion
34          this.envToDotPropMap = buildReverseLookupToMap(
35              ClientBuilder.DEFAULT_CLIENT_CACHE_TTL_PROPERTY_NAME,
36              ClientBuilder.DEFAULT_CLIENT_CACHE_TTI_PROPERTY_NAME,
37              ClientBuilder.DEFAULT_CLIENT_ORG_URL_PROPERTY_NAME,
38              ClientBuilder.DEFAULT_CLIENT_CONNECTION_TIMEOUT_PROPERTY_NAME,
39              ClientBuilder.DEFAULT_CLIENT_AUTHENTICATION_SCHEME_PROPERTY_NAME,
40              ClientBuilder.DEFAULT_CLIENT_REQUEST_TIMEOUT_PROPERTY_NAME,
41              ClientBuilder.DEFAULT_CLIENT_RETRY_MAX_ATTEMPTS_PROPERTY_NAME,
42              ClientBuilder.DEFAULT_CLIENT_TESTING_DISABLE_HTTPS_CHECK_PROPERTY_NAME);
43      }
44  
45      private Map<String, String> buildReverseLookupToMap(String... dottedPropertyNames) {
46          return Arrays.stream(dottedPropertyNames)
47              .collect(Collectors.toMap(this::toEnvVarName, dottedPropertyName -> dottedPropertyName));
48      }
49  
50      @Override
51      public String toEnvVarName(String dottedPropertyName) {
52          Assert.hasText(dottedPropertyName, "dottedPropertyName argument cannot be null or empty.");
53          dottedPropertyName = Strings.trimWhitespace(dottedPropertyName);
54  
55          StringBuilder sb = new StringBuilder();
56  
57          for(char c : dottedPropertyName.toCharArray()) {
58              if (c == '.') {
59                  sb.append('_');
60                  continue;
61              }
62              sb.append(Character.toUpperCase(c));
63          }
64  
65          return sb.toString();
66      }
67  
68      @Override
69      public String toDottedPropertyName(String envVarName) {
70          Assert.hasText(envVarName, "envVarName argument cannot be null or empty.");
71          envVarName = Strings.trimWhitespace(envVarName);
72  
73          //special cases (camel case):
74          if (envToDotPropMap.containsKey(envVarName)) {
75              return envToDotPropMap.get(envVarName);
76          }
77  
78          //default cases:
79          StringBuilder sb = new StringBuilder();
80  
81          for(char c : envVarName.toCharArray()) {
82              if (c == '_') {
83                  sb.append('.');
84                  continue;
85              }
86              sb.append(Character.toLowerCase(c));
87          }
88  
89          return sb.toString();
90      }
91  }
