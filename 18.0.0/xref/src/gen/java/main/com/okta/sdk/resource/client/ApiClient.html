783      *            Object
784      * @param contentType
785      *            Content type
786      * @param formParams
787      *            Form parameters
788      *
789      * @return Object
790      *
791      * @throws ApiException
792      *             API exception
793      */
794     public HttpEntity serialize(Object obj, Map<String, Object> formParams, ContentType contentType)
795             throws ApiException {
796         String mimeType = contentType.getMimeType();
797         if (isJsonMime(mimeType)) {
798             try {
799                 return new StringEntity(objectMapper.writeValueAsString(obj), ContentType.APPLICATION_JSON);
800             } catch (JsonProcessingException e) {
801                 throw new ApiException(e);
802             }
803         } else if (mimeType.equals(ContentType.MULTIPART_FORM_DATA.getMimeType())) {
804             MultipartEntityBuilder multiPartBuilder = MultipartEntityBuilder.create();
805             for (Entry<String, Object> paramEntry : formParams.entrySet()) {
806                 Object value = paramEntry.getValue();
807                 if (value instanceof File) {
808                     multiPartBuilder.addBinaryBody(paramEntry.getKey(), (File) value);
809                 } else if (value instanceof byte[]) {
810                     multiPartBuilder.addBinaryBody(paramEntry.getKey(), (byte[]) value);
811                 } else {
812                     Charset charset = contentType.getCharset();
813                     if (charset != null) {
814                         ContentType customContentType = ContentType.create(ContentType.TEXT_PLAIN.getMimeType(),
815                                 charset);
816                         multiPartBuilder.addTextBody(paramEntry.getKey(), parameterToString(paramEntry.getValue()),
817                                 customContentType);
818                     } else {
819                         multiPartBuilder.addTextBody(paramEntry.getKey(), parameterToString(paramEntry.getValue()));
820                     }
821                 }
822             }
823             return multiPartBuilder.build();
824         } else if (mimeType.equals(ContentType.APPLICATION_FORM_URLENCODED.getMimeType())) {
825             List<NameValuePair> formValues = new ArrayList<>();
826             for (Entry<String, Object> paramEntry : formParams.entrySet()) {
827                 formValues.add(new BasicNameValuePair(paramEntry.getKey(), parameterToString(paramEntry.getValue())));
828             }
829             return new UrlEncodedFormEntity(formValues, contentType.getCharset());
830         } else {
831             // Handle files with unknown content type
832             if (obj instanceof File) {
833                 return new FileEntity((File) obj, contentType);
834             } else if (obj instanceof byte[]) {
835                 return new ByteArrayEntity((byte[]) obj, contentType);
836             }
837             throw new ApiException("Serialization for content type '" + contentType + "' not supported");
838         }
839     }

841     /**
842      * Deserialize response body to Java object according to the Content-Type.
843      *
844      * @param <T>
845      *            Type
846      * @param response
847      *            Response
848      * @param valueType
849      *            Return type
850      *
851      * @return Deserialized object
852      *
853      * @throws ApiException
854      *             API exception
855      * @throws IOException
856      *             IO exception
857      */
858     @SuppressWarnings("unchecked")
859     public <T> T deserialize(CloseableHttpResponse response, TypeReference<T> valueType)
860             throws ApiException, IOException, ParseException {
861         if (valueType == null) {
862             return null;
863         }
864         HttpEntity entity = response.getEntity();
865         Type valueRawType = valueType.getType();
866         if (valueRawType.equals(byte[].class)) {
867             return (T) EntityUtils.toByteArray(entity);
868         } else if (valueRawType.equals(File.class)) {
869             return (T) downloadFileFromResponse(response);
870         }
871         String mimeType = getResponseMimeType(response);
872         if (mimeType == null || isJsonMime(mimeType)) {
873             // Assume json if no mime type
874             // convert input stream to string
875             java.util.Scanner s = new java.util.Scanner(entity.getContent()).useDelimiter("\\A");
876             String content = (String) (s.hasNext() ? s.next() : "");
877 
878             if ("".equals(content)) { // returns null for empty body
879                 return null;
880             }
881 
882             T value = objectMapper.readValue(content, valueType);
883             return value instanceof List ? PagedList.constructPagedList(response, value) : value;
884         } else if ("text/plain".equalsIgnoreCase(mimeType)) {
885             // convert input stream to string
886             java.util.Scanner s = new java.util.Scanner(entity.getContent()).useDelimiter("\\A");
887             return (T) (s.hasNext() ? s.next() : "");
888         } else {
889             throw new ApiException(
890                     "Deserialization for content type '" + mimeType + "' not supported for type '" + valueType + "'",
891                     response.getCode(), responseHeaders, EntityUtils.toString(entity));
892         }
893     }

895     private File downloadFileFromResponse(CloseableHttpResponse response) throws IOException {
896         Header contentDispositionHeader = response.getFirstHeader("Content-Disposition");
897         String contentDisposition = contentDispositionHeader == null ? null : contentDispositionHeader.getValue();
898         File file = prepareDownloadFile(contentDisposition);
899         Files.copy(response.getEntity().getContent(), file.toPath(), StandardCopyOption.REPLACE_EXISTING);
900         return file;
901     }

903     protected File prepareDownloadFile(String contentDisposition) throws IOException {
904         String filename = null;
905         if (contentDisposition != null && !"".equals(contentDisposition)) {
906             // Get filename from the Content-Disposition header.
907             Pattern pattern = Pattern.compile("filename=['\"]?([^'\"\\s]+)['\"]?");
908             Matcher matcher = pattern.matcher(contentDisposition);
909             if (matcher.find())
910                 filename = matcher.group(1);
911         }
912 
913         String prefix;
914         String suffix = null;
915         if (filename == null) {
916             prefix = "download-";
917             suffix = "";
918         } else {
919             int pos = filename.lastIndexOf('.');
920             if (pos == -1) {
921                 prefix = filename + "-";
922             } else {
923                 prefix = filename.substring(0, pos) + "-";
924                 suffix = filename.substring(pos);
925             }
926             // Files.createTempFile requires the prefix to be at least three characters long
927             if (prefix.length() < 3)
928                 prefix = "download-";
929         }
930 
931         if (tempFolderPath == null)
932             return Files.createTempFile(prefix, suffix).toFile();
933         else
934             return Files.createTempFile(Paths.get(tempFolderPath), prefix, suffix).toFile();
935     }

937     /**
938      * Build full URL by concatenating base path, the given sub path and query parameters.
939      *
940      * @param path
941      *            The sub path
942      * @param queryParams
943      *            The query parameters
944      * @param collectionQueryParams
945      *            The collection query parameters
946      * @param urlQueryDeepObject
947      *            URL query string of the deep object parameters
948      *
949      * @return The full URL
950      */
951     private String buildUrl(String path, List<Pair> queryParams, List<Pair> collectionQueryParams,
952             String urlQueryDeepObject) {
953         String baseURL;
954         if (serverIndex != null) {
955             if (serverIndex < 0 || serverIndex >= servers.size()) {
956                 throw new ArrayIndexOutOfBoundsException(
957                         String.format("Invalid index %d when selecting the host settings. Must be less than %d",
958                                 serverIndex, servers.size()));
959             }
960             baseURL = servers.get(serverIndex).URL(serverVariables);
961         } else {
962             baseURL = basePath;
963         }
964 
965         final StringBuilder url = new StringBuilder();
966         url.append(baseURL).append(path);
967 
968         if (queryParams != null && !queryParams.isEmpty()) {
969             // support (constant) query string in `path`, e.g. "/posts?draft=1"
970             String prefix = path.contains("?") ? "&" : "?";
971             for (Pair param : queryParams) {
972                 if (param.getValue() != null) {
973                     if (prefix != null) {
974                         url.append(prefix);
975                         prefix = null;
976                     } else {
977                         url.append("&");
978                     }
979                     String value = parameterToString(param.getValue());
980                     // query parameter value already escaped as part of parameterToPair
981                     url.append(escapeString(param.getName())).append("=").append(value);
982                 }
983             }
984         }
985 
986         if (collectionQueryParams != null && !collectionQueryParams.isEmpty()) {
987             String prefix = url.toString().contains("?") ? "&" : "?";
988             for (Pair param : collectionQueryParams) {
989                 if (param.getValue() != null) {
990                     if (prefix != null) {
991                         url.append(prefix);
992                         prefix = null;
993                     } else {
994                         url.append("&");
995                     }
996                     String value = parameterToString(param.getValue());
997                     // collection query parameter value already escaped as part of parameterToPairs
998                     url.append(escapeString(param.getName())).append("=").append(value);
999                 }
1000             }
1001         }
1002 
1003         if (urlQueryDeepObject != null && urlQueryDeepObject.length() > 0) {
1004             url.append(url.toString().contains("?") ? "&" : "?");
1005             url.append(urlQueryDeepObject);
1006         }
1007 
1008         return url.toString();
1009     }

1026     protected <T> T processResponse(CloseableHttpResponse response, TypeReference<T> returnType)
1027             throws ApiException, IOException, ParseException {
1028         statusCode = response.getCode();
1029         if (statusCode == HttpStatus.SC_NO_CONTENT) {
1030             return null;
1031         }
1032 
1033         responseHeaders = transformResponseHeaders(response.getHeaders());
1034         if (isSuccessfulStatus(statusCode)) {
1035             return this.deserialize(response, returnType);

1038             throw new ApiException(message, statusCode, responseHeaders, message);

1042     /**
1043      * Invoke API by sending HTTP request with the given options.
1044      *
1045      * @param <T>
1046      *            Type
1047      * @param path
1048      *            The sub-path of the HTTP URL
1049      * @param method
1050      *            The request method, one of "GET", "POST", "PUT", and "DELETE"
1051      * @param queryParams
1052      *            The query parameters
1053      * @param collectionQueryParams
1054      *            The collection query parameters
1055      * @param urlQueryDeepObject
1056      *            A URL query string for deep object parameters
1057      * @param body
1058      *            The request body object - if it is not binary, otherwise null
1059      * @param headerParams
1060      *            The header parameters
1061      * @param cookieParams
1062      *            The cookie parameters
1063      * @param formParams
1064      *            The form parameters
1065      * @param accept
1066      *            The request's Accept header
1067      * @param contentType
1068      *            The request's Content-Type header
1069      * @param authNames
1070      *            The authentications to apply
1071      * @param returnType
1072      *            Return type
1073      *
1074      * @return The response body in type of string
1075      *
1076      * @throws ApiException
1077      *             API exception
1078      */
1079     public <T> T invokeAPI(String path, String method, List<Pair> queryParams, List<Pair> collectionQueryParams,
1080             String urlQueryDeepObject, Object body, Map<String, String> headerParams, Map<String, String> cookieParams,
1081             Map<String, Object> formParams, String accept, String contentType, String[] authNames,
1082             TypeReference<T> returnType) throws ApiException {
1083         if (body != null && !formParams.isEmpty()) {
1084             throw new ApiException("Cannot have body and form params");
1085         }
1086 
1087         updateParamsForAuth(authNames, queryParams, headerParams, cookieParams);
1088         final String url = buildUrl(path, queryParams, collectionQueryParams, urlQueryDeepObject);
1089 
1090         ClassicRequestBuilder builder = ClassicRequestBuilder.create(method);
1091         builder.setUri(url);
1092 
1093         if (accept != null) {
1094             builder.addHeader("Accept", accept);
1095         }
1096         for (Entry<String, String> keyValue : headerParams.entrySet()) {
1097             builder.addHeader(keyValue.getKey(), keyValue.getValue());
1098         }
1099         for (Map.Entry<String, String> keyValue : defaultHeaderMap.entrySet()) {
1100             if (!headerParams.containsKey(keyValue.getKey())) {
1101                 builder.addHeader(keyValue.getKey(), keyValue.getValue());
1102             }
1103         }
1104 
1105         BasicCookieStore store = new BasicCookieStore();
1106         for (Entry<String, String> keyValue : cookieParams.entrySet()) {
1107             store.addCookie(buildCookie(keyValue.getKey(), keyValue.getValue(), builder.getUri()));
1108         }
1109         for (Entry<String, String> keyValue : defaultCookieMap.entrySet()) {
1110             if (!cookieParams.containsKey(keyValue.getKey())) {
1111                 store.addCookie(buildCookie(keyValue.getKey(), keyValue.getValue(), builder.getUri()));
1112             }
1113         }
1114 
1115         HttpClientContext context = HttpClientContext.create();
1116         context.setCookieStore(store);
1117 
1118         ContentType contentTypeObj = getContentType(contentType);
1119         if (body != null || !formParams.isEmpty()) {
1120             if (isBodyAllowed(method)) {
1121                 // Add entity if we have content and a valid method
1122                 builder.setEntity(serialize(body, formParams, contentTypeObj));
1123             } else {
1124                 throw new ApiException("method " + method + " does not support a request body");
1125             }
1126         } else {
1127             // for empty body
1128             builder.setEntity(new StringEntity("", contentTypeObj));
1129         }
1130 
1131         String cacheKey = url;
1132 
1133         if (Objects.nonNull(path)) {
1134             cacheKey = cacheKey.split("\\?")[0];
1135         }
1136 
1137         if (method == HttpMethod.DELETE.name()) {
1138             cache.remove(cacheKey);
1139         }
1140 
1141         String[] uriSegments = path.split("/");
1142 
1143         if (uriSegments != null && uriSegments.length >= 6) {
1144             String key = getBasePath() + "/" + uriSegments[1] + "/" + uriSegments[2] + "/" + uriSegments[3] + "/"
1145                     + uriSegments[4];
1146             if (cache.get(key) != null) {
1147                 cache.remove(key);
1148             }
1149         }
1150 
1151         if (method == HttpMethod.GET.name() && !returnType.getType().getTypeName().contains("List")
1152                 && !(url.contains("expand")) && !(url.contains(".well-known/okta-organization"))) {
1153 
1154             T cacheData = (T) cache.get(cacheKey);
1155 
1156             if (Objects.isNull(cacheData)) {
1157 
1158                 try (CloseableHttpResponse response = httpClient.execute(builder.build(), context)) {
1159                     T t = processResponse(response, returnType);
1160 
1161                     if (method.equals(HttpMethod.GET.name()) && Objects.nonNull(t)) {
1162 
1163                         Map<String, ?> map = getObjectMapper().convertValue(t, LinkedHashMap.class);
1164 
1165                         String href = null;
1166 
1167                         if (Objects.nonNull(map)) {
1168                             Map<String, LinkedHashMap> links = (Map<String, LinkedHashMap>) map.get("_links");
1169                             if (Objects.nonNull(links)) {
1170                                 LinkedHashMap<String, String> self = links.get("self");
1171                                 if (Objects.nonNull(self)) {
1172                                     href = (String) self.get("href");
1173                                 }
1174                             }
1175                         }
1176 
1177                         if (Objects.nonNull(href)) {
1178                             cache.put(href, t);
1179                         } else {
1180                             cache.put(cacheKey, t);
1181                         }
1182                     }
1183 
1184                     return t;
1185                 } catch (IOException | ParseException e) {
1186                     throw new ApiException(e);
1187                 }
1188             } else {
1189                 // return data from cache
1190                 return (T) cache.get(cacheKey);
1191             }
1192 
1193         } else {
1194 
1195             try (CloseableHttpResponse response = httpClient.execute(builder.build(), context)) {
1196                 T t = processResponse(response, returnType);
1197 
1198                 if (cache.get(cacheKey) != null && !method.equals(HttpMethod.DELETE.name())) {
1199                     cache.put(cacheKey, t);
1200                 }
1201                 return t;
1202             } catch (IOException | ParseException e) {
1203                 throw new ApiException(e);
1204             }
1205         }
1206     }
