93      /**
94       * a new List that contains the specified elements or an empty collection if the elements are null or empty.
95       *
96       * @param elements the elements to put in the list.
97       * @param <T>      the type of elements in the collection
98       * @return a new List that contains the specified elements or an empty collection if the elements are null or empty.
99       */
100     public static <T> List<T> toList(T... elements) {
101         if (elements == null || elements.length == 0) {
102             return java.util.Collections.emptyList();
103         }
104         // Avoid integer overflow when a large array is passed in
105         int capacity = computeListCapacity(elements.length);
106         ArrayList<T> list = new ArrayList<>(capacity);
107         java.util.Collections.addAll(list, elements);
108         return list;
109     }

111     /**
112      * a new List that contains the specified elements or an empty collection if the elements are null or empty.
113      *
114      * @param elements the elements to put in the list.
115      * @param <T>      the type of elements in the collection
116      * @return a new List that contains the specified elements or an empty collection if the elements are null or empty.
117      */
118     public static <T> List<T> toList(Collection<T> elements) {
119         if (elements instanceof List) {
120             return (List<T>) elements;
121         }
122         if (isEmpty(elements)) {
123             return java.util.Collections.emptyList();
124         }
125         // Avoid integer overflow when a large array is passed in
126         int capacity = computeListCapacity(elements.size());
127         ArrayList<T> list = new ArrayList<>(capacity);
128         list.addAll(elements);
129         return list;
130     }

132     /**
133      * Returns a new {@link Set} that contains the specified elements or an empty Set if the elements are null or empty.
134      *
135      * @param elements elements to add to the new set
136      * @param <E>      the type of elements in the set
137      * @return a new {@link Set} that contains the specified elements or an empty Set if the elements are null or empty.
138      */
139     public static <E> Set<E> toSet(E... elements) {
140         if (elements == null || elements.length == 0) {
141             return java.util.Collections.emptySet();
142         }
143         LinkedHashSet<E> set = new LinkedHashSet<>(elements.length * 4 / 3 + 1);
144         java.util.Collections.addAll(set, elements);
145         return set;
146     }

168     /**
169      * Merge the given Properties instance into the given Map,
170      * copying all properties (key-value pairs) over.
171      * <p>Uses {@code Properties.propertyNames()} to even catch
172      * default properties linked into the original Properties instance.
173      *
174      * @param props the Properties instance to merge (may be {@code null})
175      * @param map   the target Map to merge the properties into
176      */
177     @SuppressWarnings("unchecked")
178     public static void mergePropertiesIntoMap(Properties props, Map map) {
179         if (map == null) {
180             throw new IllegalArgumentException("Map must not be null");
181         }
182         if (props != null) {
183             for (Enumeration en = props.propertyNames(); en.hasMoreElements(); ) {
184                 String key = (String) en.nextElement();
185                 Object value = props.getProperty(key);
186                 if (value == null) {
187                     // Potentially a non-String value...
188                     value = props.get(key);
189                 }
190                 map.put(key, value);
191             }
192         }
193     }

197      * Check whether the given Iterator contains the given element.
198      *
199      * @param iterator the Iterator to check
200      * @param element  the element to look for
201      * @return {@code true} if found, {@code false} else
202      */
203     public static boolean contains(Iterator iterator, Object element) {
204         if (iterator != null) {
205             while (iterator.hasNext()) {
206                 Object candidate = iterator.next();
207                 if (Objects.nullSafeEquals(candidate, element)) {
208                     return true;
209                 }
210             }
211         }
212         return false;
213     }

274     /**
275      * Return the first element in '{@code candidates}' that is contained in
276      * '{@code source}'. If no element in '{@code candidates}' is present in
277      * '{@code source}' returns {@code null}. Iteration order is
278      * {@link Collection} implementation specific.
279      *
280      * @param source     the source Collection
281      * @param candidates the candidates to search for
282      * @return the first present object, or {@code null} if not found
283      */
284     public static Object findFirstMatch(Collection source, Collection candidates) {
285         if (isEmpty(source) || isEmpty(candidates)) {
286             return null;
287         }
288         for (Object candidate : candidates) {
289             if (source.contains(candidate)) {
290                 return candidate;
291             }
292         }
293         return null;
294     }

296     /**
297      * Find a single value of the given type in the given Collection.
298      *
299      * @param collection the Collection to search
300      * @param type       the type to look for
301      * @param <T> type type of object to find
302      * @return a value of the given type found if there is a clear match,
303      * or {@code null} if none or more than one such value found
304      */
305     @SuppressWarnings("unchecked")
306     private static <T> T findValueOfType(Collection<?> collection, Class<T> type) {
307         if (isEmpty(collection)) {
308             return null;
309         }
310         T value = null;
311         for (Object element : collection) {
312             if (type == null || type.isInstance(element)) {
313                 if (value != null) {
314                     // More than one value found... no clear single value.
315                     return null;
316                 }
317                 value = (T) element;
318             }
319         }
320         return value;
321     }

323     /**
324      * Find the first single value of the given type in the given Collection or {@code null} if none was found
325      *
326      * @param collection the Collection to search
327      * @param type       the type to look for
328      * @param <T> type type of object to find
329      * @return a value of the given type found or {@code null} if none found.
330      */
331     @SuppressWarnings("unchecked")
332     public static <T> T findFirstValueOfType(Collection<?> collection, Class<T> type) {
333 
334         Assert.notNull(type, "type argument cannot be null.");
335 
336         if (!isEmpty(collection)) {
337             for (Object element : collection) {
338                 if (element != null && type.isInstance(element)) {
339                     return (T) element;
340                 }
341             }
342         }
343 
344         return null;
345     }

347     /**
348      * Find a single value of one of the given types in the given Collection:
349      * searching the Collection for a value of the first type, then
350      * searching for a value of the second type, etc.
351      *
352      * @param collection the collection to search
353      * @param types      the types to look for, in prioritized order
354      * @return a value of one of the given types found if there is a clear match,
355      * or {@code null} if none or more than one such value found
356      */
357     public static Object findValueOfType(Collection<?> collection, Class<?>[] types) {
358         if (isEmpty(collection) || Objects.isEmpty(types)) {
359             return null;
360         }
361         for (Class<?> type : types) {
362             Object value = findValueOfType(collection, type);
363             if (value != null) {
364                 return value;
365             }
366         }
367         return null;
368     }

370     /**
371      * Determine whether the given Collection only contains a single unique object.
372      *
373      * @param collection the Collection to check
374      * @return {@code true} if the collection contains a single reference or
375      * multiple references to the same instance, {@code false} else
376      */
377     public static boolean hasUniqueObject(Collection collection) {
378         if (isEmpty(collection)) {
379             return false;
380         }
381         boolean hasCandidate = false;
382         Object candidate = null;
383         for (Object elem : collection) {
384             if (!hasCandidate) {
385                 hasCandidate = true;
386                 candidate = elem;
387             } else if (candidate != elem) {
388                 return false;
389             }
390         }
391         return true;
392     }

394     /**
395      * Find the common element type of the given Collection, if any.
396      *
397      * @param collection the Collection to check
398      * @return the common element type, or {@code null} if no clear
399      * common type has been found (or the collection was empty)
400      */
401     public static Class<?> findCommonElementType(Collection collection) {
402         if (isEmpty(collection)) {
403             return null;
404         }
405         Class<?> candidate = null;
406         for (Object val : collection) {
407             if (val != null) {
408                 if (candidate == null) {
409                     candidate = val.getClass();
410                 } else if (candidate != val.getClass()) {
411                     return null;
412                 }
413             }
414         }
415         return candidate;
416     }

418     /**
419      * Marshal the elements from the given enumeration into an array of the given type.
420      * Enumeration elements must be assignable to the type of the given array. The array
421      * returned will be a different instance than the array given.
422      * @param enumeration source enumeration
423      * @param array the array into which the elements of the list are to
424      *          be stored, if it is big enough; otherwise, a new array of the
425      *          same runtime type is allocated for this purpose.
426      * @param <A> type of array
427      * @param <E> type of enumeration
428      * @return an array with the contents of the enumeration
429      */
430     public static <A, E extends A> A[] toArray(Enumeration<E> enumeration, A[] array) {
431         ArrayList<A> elements = new ArrayList<>();
432         while (enumeration.hasMoreElements()) {
433             elements.add(enumeration.nextElement());
434         }
435         return elements.toArray(array);
436     }
