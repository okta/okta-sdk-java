417     @Override
418     protected void checkParameters(String attribute, String value) {
419         super.checkParameters(attribute, value);
420         if (PARAM_QUALITY_FACTOR.equals(attribute)) {
421             value = unquote(value);
422             double d = Double.parseDouble(value);
423             Assert.isTrue(d >= 0D && d <= 1D,
424                     "Invalid quality value \"" + value + "\": should be between 0.0 and 1.0");
425         }
426     }

428     /**
429      * Return the quality value, as indicated by a {@code q} parameter, if any.
430      * Defaults to {@code 1.0}.
431      * @return the quality factory
432      */
433     public double getQualityValue() {
434         String qualityFactory = getParameter(PARAM_QUALITY_FACTOR);
435         return (qualityFactory != null ? Double.parseDouble(unquote(qualityFactory)) : 1D);
436     }

460     /**
461      * Return a replica of this instance with the quality value of the given MediaType.
462      * @return the same instance if the given MediaType doesn't have a quality value, or a new one otherwise
463      */
464     public MediaType copyQualityValue(MediaType mediaType) {
465         if (!mediaType.getParameters().containsKey(PARAM_QUALITY_FACTOR)) {
466             return this;
467         }
468         Map<String, String> params = new LinkedHashMap<>(getParameters());
469         params.put(PARAM_QUALITY_FACTOR, mediaType.getParameters().get(PARAM_QUALITY_FACTOR));
470         return new MediaType(this, params);
471     }
472 
473     /**
474      * Return a replica of this instance with its quality value removed.
475      * @return the same instance if the media type doesn't contain a quality value, or a new one otherwise
476      */
477     public MediaType removeQualityValue() {
478         if (!getParameters().containsKey(PARAM_QUALITY_FACTOR)) {
479             return this;
480         }
481         Map<String, String> params = new LinkedHashMap<>(getParameters());
482         params.remove(PARAM_QUALITY_FACTOR);
483         return new MediaType(this, params);
484     }

498     /**
499      * Parse the given String into a single {@code MediaType}.
500      * @param mediaType the string to parse
501      * @return the media type
502      * @throws InvalidMediaTypeException if the media type value cannot be parsed
503      */
504     public static MediaType parseMediaType(String mediaType) {
505         MimeType type;
506         try {
507             type = MimeTypeUtils.parseMimeType(mediaType);
508         }
509         catch (InvalidMimeTypeException ex) {
510             throw new InvalidMediaTypeException(ex);
511         }
512         try {
513             return new MediaType(type.getType(), type.getSubtype(), type.getParameters());
514         }
515         catch (IllegalArgumentException ex) {
516             throw new InvalidMediaTypeException(mediaType, ex.getMessage());
517         }
518     }

520     /**
521      * Parse the given comma-separated string into a list of {@code MediaType} objects.
522      * <p>This method can be used to parse an Accept or Content-Type header.
523      * @param mediaTypes the string to parse
524      * @return the list of media types
525      * @throws InvalidMediaTypeException if the media type value cannot be parsed
526      */
527     public static List<MediaType> parseMediaTypes(String mediaTypes) {
528         if (!Strings.hasLength(mediaTypes)) {
529             return Collections.emptyList();
530         }
531         String[] tokens = Strings.tokenizeToStringArray(mediaTypes, ",");
532         List<MediaType> result = new ArrayList<>(tokens.length);
533         for (String token : tokens) {
534             result.add(parseMediaType(token));
535         }
536         return result;
537     }
538 
539     /**
540      * Parse the given list of (potentially) comma-separated strings into a
541      * list of {@code MediaType} objects.
542      * <p>This method can be used to parse an Accept or Content-Type header.
543      * @param mediaTypes the string to parse
544      * @return the list of media types
545      * @throws InvalidMediaTypeException if the media type value cannot be parsed
546      * @since 4.3.2
547      */
548     public static List<MediaType> parseMediaTypes(List<String> mediaTypes) {
549         if (com.okta.commons.lang.Collections.isEmpty(mediaTypes)) {
550             return Collections.emptyList();
551         }
552         else if (mediaTypes.size() == 1) {
553             return parseMediaTypes(mediaTypes.get(0));
554         }
555         else {
556             List<MediaType> result = new ArrayList<>(8);
557             for (String mediaType : mediaTypes) {
558                 result.addAll(parseMediaTypes(mediaType));
559             }
560             return result;
561         }
562     }

593     /**
594      * Sorts the given list of {@code MediaType} objects by specificity.
595      * <p>Given two media types:
596      * <ol>
597      * <li>if either media type has a {@linkplain #isWildcardType() wildcard type}, then the media type without the
598      * wildcard is ordered before the other.</li>
599      * <li>if the two media types have different {@linkplain #getType() types}, then they are considered equal and
600      * remain their current order.</li>
601      * <li>if either media type has a {@linkplain #isWildcardSubtype() wildcard subtype}, then the media type without
602      * the wildcard is sorted before the other.</li>
603      * <li>if the two media types have different {@linkplain #getSubtype() subtypes}, then they are considered equal
604      * and remain their current order.</li>
605      * <li>if the two media types have different {@linkplain #getQualityValue() quality value}, then the media type
606      * with the highest quality value is ordered before the other.</li>
607      * <li>if the two media types have a different amount of {@linkplain #getParameter(String) parameters}, then the
608      * media type with the most parameters is ordered before the other.</li>
609      * </ol>
610      * <p>For example:
611      * <blockquote>audio/basic &lt; audio/* &lt; *&#047;*</blockquote>
612      * <blockquote>audio/* &lt; audio/*;q=0.7; audio/*;q=0.3</blockquote>
613      * <blockquote>audio/basic;level=1 &lt; audio/basic</blockquote>
614      * <blockquote>audio/basic == text/html</blockquote>
615      * <blockquote>audio/basic == audio/wave</blockquote>
616      * @param mediaTypes the list of media types to be sorted
617      * @see <a href="http://tools.ietf.org/html/rfc7231#section-5.3.2">HTTP 1.1: Semantics
618      * and Content, section 5.3.2</a>
619      */
620     public static void sortBySpecificity(List<MediaType> mediaTypes) {
621         Assert.notNull(mediaTypes, "'mediaTypes' must not be null");
622         if (mediaTypes.size() > 1) {
623             Collections.sort(mediaTypes, SPECIFICITY_COMPARATOR);
624         }
625     }

627     /**
628      * Sorts the given list of {@code MediaType} objects by quality value.
629      * <p>Given two media types:
630      * <ol>
631      * <li>if the two media types have different {@linkplain #getQualityValue() quality value}, then the media type
632      * with the highest quality value is ordered before the other.</li>
633      * <li>if either media type has a {@linkplain #isWildcardType() wildcard type}, then the media type without the
634      * wildcard is ordered before the other.</li>
635      * <li>if the two media types have different {@linkplain #getType() types}, then they are considered equal and
636      * remain their current order.</li>
637      * <li>if either media type has a {@linkplain #isWildcardSubtype() wildcard subtype}, then the media type without
638      * the wildcard is sorted before the other.</li>
639      * <li>if the two media types have different {@linkplain #getSubtype() subtypes}, then they are considered equal
640      * and remain their current order.</li>
641      * <li>if the two media types have a different amount of {@linkplain #getParameter(String) parameters}, then the
642      * media type with the most parameters is ordered before the other.</li>
643      * </ol>
644      * @param mediaTypes the list of media types to be sorted
645      * @see #getQualityValue()
646      */
647     public static void sortByQualityValue(List<MediaType> mediaTypes) {
648         Assert.notNull(mediaTypes, "'mediaTypes' must not be null");
649         if (mediaTypes.size() > 1) {
650             Collections.sort(mediaTypes, QUALITY_VALUE_COMPARATOR);
651         }
652     }

654     /**
655      * Sorts the given list of {@code MediaType} objects by specificity as the
656      * primary criteria and quality value the secondary.
657      * @see MediaType#sortBySpecificity(List)
658      * @see MediaType#sortByQualityValue(List)
659      */
660     public static void sortBySpecificityAndQuality(List<MediaType> mediaTypes) {
661         Assert.notNull(mediaTypes, "'mediaTypes' must not be null");
662         if (mediaTypes.size() > 1) {
663             Collections.sort(mediaTypes,
664                     MediaType.SPECIFICITY_COMPARATOR.thenComparing(MediaType.QUALITY_VALUE_COMPARATOR));
665         }
666     }

669     /**
670      * Comparator used by {@link #sortByQualityValue(List)}.
671      */
672     public static final Comparator<MediaType> QUALITY_VALUE_COMPARATOR = (mediaType1, mediaType2) -> {
673         double quality1 = mediaType1.getQualityValue();
674         double quality2 = mediaType2.getQualityValue();
675         int qualityComparison = Double.compare(quality2, quality1);
676         if (qualityComparison != 0) {
677             return qualityComparison;  // audio/*;q=0.7 < audio/*;q=0.3
678         }
679         else if (mediaType1.isWildcardType() && !mediaType2.isWildcardType()) { // */* < audio/*
680             return 1;
681         }
682         else if (mediaType2.isWildcardType() && !mediaType1.isWildcardType()) { // audio/* > */*
683             return -1;
684         }
685         else if (!mediaType1.getType().equals(mediaType2.getType())) { // audio/basic == text/html
686             return 0;
687         }
688         else { // mediaType1.getType().equals(mediaType2.getType())
689             if (mediaType1.isWildcardSubtype() && !mediaType2.isWildcardSubtype()) { // audio/* < audio/basic
690                 return 1;
691             }
692             else if (mediaType2.isWildcardSubtype() && !mediaType1.isWildcardSubtype()) { // audio/basic > audio/*
693                 return -1;
694             }
695             else if (!mediaType1.getSubtype().equals(mediaType2.getSubtype())) { // audio/basic == audio/wave
696                 return 0;
697             }
698             else {
699                 int paramsSize1 = mediaType1.getParameters().size();
700                 int paramsSize2 = mediaType2.getParameters().size();
701                 // audio/basic;level=1 < audio/basic
702                 return (paramsSize2 < paramsSize1 ? -1 : (paramsSize2 == paramsSize1 ? 0 : 1));
703             }
704         }
705     };

711     public static final Comparator<MediaType> SPECIFICITY_COMPARATOR = new SpecificityComparator<MediaType>() {
