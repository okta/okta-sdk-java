130      * Check whether the given {@code CharSequence} contains actual <em>text</em>.
131      * <p>More specifically, this method returns {@code true} if the
132      * {@code CharSequence} is not {@code null}, its length is greater than
133      * 0, and it contains at least one non-whitespace character.
134      * <pre class="code">
135      * StringUtils.hasText(null) = false
136      * StringUtils.hasText("") = false
137      * StringUtils.hasText(" ") = false
138      * StringUtils.hasText("12345") = true
139      * StringUtils.hasText(" 12345 ") = true
140      * </pre>
141      * @param str the {@code CharSequence} to check (may be {@code null})
142      * @return {@code true} if the {@code CharSequence} is not {@code null},
143      * its length is greater than 0, and it does not contain whitespace only
144      * @see Character#isWhitespace
145      */
146     public static boolean hasText(CharSequence str) {
147         if (!hasLength(str)) {
148             return false;
149         }
150 
151         int strLen = str.length();
152         for (int i = 0; i < strLen; i++) {
153             if (!Character.isWhitespace(str.charAt(i))) {
154                 return true;
155             }
156         }
157         return false;
158     }

208      * @param str the {@code String} to check
209      * @return the trimmed {@code String}
210      * @see java.lang.Character#isWhitespace
211      */
212     public static String trimWhitespace(String str) {
213         if (!hasLength(str)) {
214             return str;
215         }
216 
217         StringBuilder sb = new StringBuilder(str);
218         while (sb.length() > 0 && Character.isWhitespace(sb.charAt(0))) {
219             sb.deleteCharAt(0);
220         }
221         while (sb.length() > 0 && Character.isWhitespace(sb.charAt(sb.length() - 1))) {

223         }
224         return sb.toString();

390      * Count the occurrences of the substring {@code sub} in string {@code str}.
391      * @param str string to search in. Return 0 if this is {@code null}.
392      * @param sub string to search for. Return 0 if this is {@code null}.
393      * @return count of {@code sub}
394      */
395     public static int countOccurrencesOf(String str, String sub) {
396         if (!hasLength(str) || !hasLength(sub)) {
397             return 0;
398         }
399 
400         int count = 0;
401         int pos = 0;
402         int idx;
403         while ((idx = str.indexOf(sub, pos)) != -1) {
404             ++count;
405             pos = idx + sub.length();
406         }
407         return count;

410     /**
411      * Replace all occurrences of a substring within a string with
412      * another string.
413      * @param inString {@code String} to examine
414      * @param oldPattern {@code String} to replace
415      * @param newPattern {@code String} to insert
416      * @return a {@code String} with the replacements
417      */
418     public static String replace(String inString, String oldPattern, String newPattern) {
419         if (!hasLength(inString) || !hasLength(oldPattern) || newPattern == null) {
420             return inString;
421         }
422         int index = inString.indexOf(oldPattern);
423         if (index == -1) {
424             // no occurrence -> can return input as-is
425             return inString;
426         }
427 
428         int capacity = inString.length();
429         if (newPattern.length() > oldPattern.length()) {
430             capacity += 16;
431         }
432         StringBuilder sb = new StringBuilder(capacity);
433 
434         int pos = 0;  // our position in the old string
435         int patLen = oldPattern.length();
436         while (index >= 0) {
437             sb.append(inString.substring(pos, index));
438             sb.append(newPattern);
439             pos = index + patLen;
440             index = inString.indexOf(oldPattern, pos);
441         }
442 
443         // append any characters to the right of a match
444         sb.append(inString.substring(pos));
445         return sb.toString();
446     }

551     private static String changeFirstCharacterCase(String str, boolean capitalize) {
552         if (!hasLength(str)) {
553             return str;
554         }
555 
556         char baseChar = str.charAt(0);
557         char updatedChar;
558         if (capitalize) {
559             updatedChar = Character.toUpperCase(baseChar);
560         }
561         else {
562             updatedChar = Character.toLowerCase(baseChar);
563         }
564         if (baseChar == updatedChar) {
565             return str;
566         }
567 
568         char[] chars = str.toCharArray();
569         chars[0] = updatedChar;
570         return new String(chars, 0, chars.length);
571     }

659     /**
660      * Normalize the path by suppressing sequences like "path/.." and
661      * inner simple dots.
662      * <p>The result is convenient for path comparison. For other uses,
663      * notice that Windows separators ("\") are replaced by simple slashes.
664      * @param path the original path
665      * @return the normalized path
666      */
667     public static String cleanPath(String path) {
668         if (path == null) {
669             return null;
670         }
671         String pathToUse = replace(path, WINDOWS_FOLDER_SEPARATOR, FOLDER_SEPARATOR);
672 
673         // Strip prefix from path to analyze, to not treat it as part of the
674         // first path element. This is necessary to correctly parse paths like
675         // "file:core/../core/io/Resource.class", where the ".." should just
676         // strip the first "core" directory while keeping the "file:" prefix.
677         int prefixIndex = pathToUse.indexOf(":");
678         String prefix = "";
679         if (prefixIndex != -1) {
680             prefix = pathToUse.substring(0, prefixIndex + 1);
681             if (prefix.contains("/")) {
682                 prefix = "";
683             }
684             else {
685                 pathToUse = pathToUse.substring(prefixIndex + 1);
686             }
687         }
688         if (pathToUse.startsWith(FOLDER_SEPARATOR)) {
689             prefix = prefix + FOLDER_SEPARATOR;
690             pathToUse = pathToUse.substring(1);
691         }
692 
693         String[] pathArray = delimitedListToStringArray(pathToUse, FOLDER_SEPARATOR);
694         List<String> pathElements = new LinkedList<>();
695         int tops = 0;
696 
697         for (int i = pathArray.length - 1; i >= 0; i--) {
698             String element = pathArray[i];
699             if (CURRENT_PATH.equals(element)) {
700                 // Points to current directory - drop it.
701             }
702             else if (TOP_PATH.equals(element)) {
703                 // Registering top path found.
704                 tops++;
705             }
706             else {
707                 if (tops > 0) {
708                     // Merging path element with element corresponding to top path.
709                     tops--;
710                 }
711                 else {
712                     // Normal path element found.
713                     pathElements.add(0, element);
714                 }
715             }
716         }
717 
718         // Remaining top paths need to be retained.
719         for (int i = 0; i < tops; i++) {
720             pathElements.add(0, TOP_PATH);
721         }
722 
723         return prefix + collectionToDelimitedString(pathElements, FOLDER_SEPARATOR);
724     }

736     /**
737      * Decode the given encoded URI component value. Based on the following rules:
738      * <ul>
739      * <li>Alphanumeric characters {@code "a"} through {@code "z"}, {@code "A"} through {@code "Z"},
740      * and {@code "0"} through {@code "9"} stay the same.</li>
741      * <li>Special characters {@code "-"}, {@code "_"}, {@code "."}, and {@code "*"} stay the same.</li>
742      * <li>A sequence "{@code %<i>xy</i>}" is interpreted as a hexadecimal representation of the character.</li>
743      * </ul>
744      * @param source the encoded String (may be {@code null})
745      * @param charset the character set
746      * @return the decoded value
747      * @throws IllegalArgumentException when the given source contains invalid encoded sequences
748      * @since 5.0
749      * @see java.net.URLDecoder#decode(String, String)
750      */
751     public static String uriDecode(String source, Charset charset) {
752         if (source == null) {
753             return null;
754         }
755         int length = source.length();
756         if (length == 0) {
757             return source;
758         }
759         Assert.notNull(charset, "Charset must not be null");
760 
761         ByteArrayOutputStream bos = new ByteArrayOutputStream(length);
762         boolean changed = false;
763         for (int i = 0; i < length; i++) {
764             int ch = source.charAt(i);
765             if (ch == '%') {
766                 if (i + 2 < length) {
767                     char hex1 = source.charAt(i + 1);
768                     char hex2 = source.charAt(i + 2);
769                     int u = Character.digit(hex1, 16);
770                     int l = Character.digit(hex2, 16);
771                     if (u == -1 || l == -1) {
772                         throw new IllegalArgumentException("Invalid encoded sequence \"" + source.substring(i) + "\"");
773                     }
774                     bos.write((char) ((u << 4) + l));
775                     i += 2;
776                     changed = true;
777                 }
778                 else {
779                     throw new IllegalArgumentException("Invalid encoded sequence \"" + source.substring(i) + "\"");
780                 }
781             }
782             else {
783                 bos.write(ch);
784             }
785         }
786         return (changed ? new String(bos.toByteArray(), charset) : source);
787     }

790      * Parse the given {@code localeString} value into a {@link Locale}.
791      * <p>This is the inverse operation of {@link Locale#toString Locale's toString}.
792      * @param localeString the locale {@code String}, following {@code Locale's}
793      * {@code toString()} format ("en", "en_UK", etc);
794      * also accepts spaces as separators, as an alternative to underscores
795      * @return a corresponding {@code Locale} instance
796      * @throws IllegalArgumentException in case of an invalid locale specification
797      */
798     public static Locale parseLocaleString(String localeString) {
799         String[] parts = tokenizeToStringArray(localeString, "_ ", false, false);
800         String language = (parts.length > 0 ? parts[0] : "");
801         String country = (parts.length > 1 ? parts[1] : "");
802 
803         validateLocalePart(language);
804         validateLocalePart(country);
805 
806         String variant = "";
807         if (parts.length > 2) {
808             // There is definitely a variant, and it is everything after the country
809             // code sans the separator between the country code and the variant.
810             int endIndexOfCountryCode = localeString.indexOf(country, language.length()) + country.length();
811             // Strip off any leading '_' and whitespace, what's left is the variant.
812             variant = trimLeadingWhitespace(localeString.substring(endIndexOfCountryCode));
813             if (variant.startsWith("_")) {
814                 variant = trimLeadingCharacter(variant, '_');
815             }
816         }
817         return (language.length() > 0 ? new Locale(language, country, variant) : null);
818     }

824                 throw new IllegalArgumentException(
825                         "Locale part \"" + localePart + "\" contains invalid characters");

827         }

1313      * Returns a 'cleaned' representation of the specified argument.  'Cleaned' is defined as the following:
1314      * <ol>
1315      * <li>If the specified {@code String} is <code>null</code>, return <code>null</code></li>
1316      * <li>If not {@code null}, {@link String#trim() trim()} it.</li>
1317      * <li>If the trimmed string is equal to the empty String (i.e. &quot;&quot;), return {@code null}</li>
1318      * <li>If the trimmed string is not the empty string, return the trimmed version</li>
1319      * </ol>
1320      * <p>
1321      * Therefore this method always ensures that any given string has trimmed text, and if it doesn't, {@code null}
1322      * is returned.
1323      *
1324      * @param s the input String to clean.
1325      * @return a populated-but-trimmed String or {@code null} otherwise
1326      */
1327     public static String clean(String s) {
1328         if (s == null) {
1329             return null;
1330         }
1331         String value = trimWhitespace(s);
1332         if (value == null || "".equals(value)) {
1333             return null;
1334         }
1335         return value;
1336     }
