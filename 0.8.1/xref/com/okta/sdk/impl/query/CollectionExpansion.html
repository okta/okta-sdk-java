16  package com.okta.sdk.impl.query;
17  
18  import com.okta.sdk.lang.Assert;
19  
20  /**
21   * A collection-specific {@link Expansion} that adds pagination capabilities.
22   *
23   * @since 0.5.0
24   */
25  public class CollectionExpansion extends Expansion {
26  
27      private static final long serialVersionUID = 42L;
28  
29      private final int limit;
30      private final int offset;
31  
32      public CollectionExpansion(String name, int limit, int offset) {
33          super(name);
34          Assert.isTrue(limit > 0 || offset > 0, "Either a limit or offset (or both) must be specified.");
35          this.limit = Pagination.sanitizeLimit(limit);
36          this.offset = Pagination.sanitizeOffset(offset);
37      }
38  
39      public int getLimit() {
40          return limit;
41      }
42  
43      public int getOffset() {
44          return offset;
45      }
46  
47      @Override
48      public String toString() {
49  
50          StringBuilder sb = new StringBuilder(getName());
51          if (limit > 0 || offset > 0) {
52              sb.append("(");
53              if (offset > 0) {
54                  sb.append("offset:").append(offset);
55              }
56              if (limit > 0) {
57                  if (offset > 0) {
58                      sb.append(",");
59                  }
60                  sb.append("limit:").append(limit);
61              }
62              sb.append(")");
63          }
64          return sb.toString();
65      }
66  }
