16  package com.okta.sdk.impl.query;
17  
18  import com.okta.sdk.impl.resource.CollectionReference;
19  import com.okta.sdk.impl.resource.ResourceReference;
20  import com.okta.sdk.lang.Assert;
21  import com.okta.sdk.query.Options;
22  
23  import java.util.ArrayList;
24  import java.util.Collections;
25  import java.util.List;
26  
27  /**
28   * @since 0.5.0
29   */
30  public class DefaultOptions<T extends Options> implements Options, Expandable {
31  
32      protected final List<Expansion> expansions;
33  
34      protected DefaultOptions() {
35          this.expansions = new ArrayList<Expansion>();
36      }
37  
38      protected T expand(ResourceReference property) {
39          Assert.notNull(property, "property argument cannot be null.");
40          this.expansions.add(new Expansion(property.getName()));
41          return (T) this;
42      }
43  
44      protected T expand(CollectionReference property, int limit) {
45          return expand(property, limit, Pagination.DEFAULT_OFFSET.getValue());
46      }
47  
48      protected T expand(CollectionReference property, int limit, int offset) {
49          Assert.notNull(property, "property argument cannot be null.");
50          int sLimit = Pagination.sanitizeLimit(limit);
51          int sOffset = Pagination.sanitizeOffset(offset);
52          this.expansions.add(new CollectionExpansion(property.getName(), sLimit, sOffset));
53          return (T) this;
54      }
55  
56      public List<Expansion> getExpansions() {
57          return Collections.unmodifiableList(this.expansions);
58      }
59  
60      public boolean isEmpty() {
61          return this.expansions.isEmpty();
62      }
63  }
