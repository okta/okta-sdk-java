16  package quickstart;
17  
18  import com.okta.sdk.client.Client;
19  import com.okta.sdk.client.ClientBuilder;
20  import com.okta.sdk.client.Clients;
21  import com.okta.sdk.resource.group.GroupBuilder;
22  import com.okta.sdk.resource.ResourceException;
23  import com.okta.sdk.resource.user.UserBuilder;
24  
25  import com.okta.sdk.resource.group.Group;
26  import com.okta.sdk.resource.user.User;
27  import com.okta.sdk.resource.user.UserList;
28  import com.okta.sdk.resource.user.UserStatus;
29  
30  import java.util.UUID;
31  
32  /**
33   * This class demonstrates the code found in the Okta Java SDK QuickStart Guide
34   *
35   * @since 0.5.0
36   */
37  @SuppressWarnings("PMD.UnusedLocalVariable")
38  public class Quickstart {
39  
40      public static void main(String[] args) {
41  
42          final String email = "joe.coder+" + UUID.randomUUID().toString() + "@example.com";
43          final String groupName = "java-sdk-quickstart-" + UUID.randomUUID().toString();
44          final char[] password = {'P','a','s','s','w','o','r','d','1'};
45  
46          ClientBuilder builder;
47          Client client;
48          Group group = null;
49          User user = null;
50  
51          try {
52              // Instantiate a builder for your Client. If needed, settings like Proxy and Caching can be defined here.
53              builder = Clients.builder();
54  
55              // No need to define anything else; build the Client instance. The ClientCredential information will be automatically found
56              // in pre-defined locations: i.e. ~/.okta/okta.yaml
57              client = builder.build();
58  
59              // Create a group
60              group = GroupBuilder.instance()
61                      .setName(groupName)
62                      .setDescription("Quickstart created Group")
63                      .buildAndCreate(client);
64  
65              println("Group: '" + group.getId() + "' was last updated on: " + group.getLastUpdated());
66  
67              // Create a User Account
68              user = UserBuilder.instance()
69                  .setEmail(email)
70                  .setFirstName("Joe")
71                  .setLastName("Coder")
72                  .setPassword(password)
73                  .setSecurityQuestion("Favorite security question?")
74                  .setSecurityQuestionAnswer("None of them!")
75                  .putProfileProperty("division", "Seven") // key/value pairs predefined in the user profile schema
76                  .setActive(true)
77                  .buildAndCreate(client);
78  
79              // add user to the newly created group
80              user.addToGroup(group.getId());
81  
82              String userId = user.getId();
83              println("User created with ID: " + userId);
84  
85              // You can look up user by ID
86              println("User lookup by ID: "+ client.getUser(userId).getProfile().getLogin());
87  
88              // or by Email
89              println("User lookup by Email: "+ client.getUser(email).getProfile().getLogin());
90  
91  
92              // get the list of users
93              UserList users = client.listUsers();
94  
95              // get the first user in the collection
96              println("First user in collection: " + users.iterator().next().getProfile().getEmail());
97  
98              // or loop through all of them (paging is automatic)
99  //            int ii = 0;
100 //            for (User tmpUser : users) {
101 //                println("["+ ii++ +"] User: " + tmpUser.getProfile().getEmail());
102 //            }
103 
104         }
105         catch (ResourceException e) {
106 
107             // we can get the user friendly message from the Exception
108             println(e.getMessage());
109 
110             // and you can get the details too
111             e.getCauses().forEach( cause -> println("\t" + cause.getSummary()));
112             throw e;
113         }
114         finally {
115             // cleanup
116 
117             // deactivate (if de-provisioned) and delete user
118             if (user != null) {
119                 if (user.getStatus() != UserStatus.DEPROVISIONED) {
120                     user.deactivate();
121                 }
122                 user.delete();
123             }
124 
125             // delete group
126             if (group != null) {
127                 group.delete();
128             }
129         }
130     }
131 
132     private static void println(String message) {
133         System.out.println(message);
134         System.out.flush();
135     }
136 }
