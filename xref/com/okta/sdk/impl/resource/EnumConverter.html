16  package com.okta.sdk.impl.resource;
17  
18  import com.okta.commons.lang.Assert;
19  
20  import java.util.Arrays;
21  import java.util.Map;
22  import java.util.concurrent.ConcurrentHashMap;
23  import java.util.stream.Collectors;
24  
25  /**
26   * Converts Strings to enum's based on the {code}.toString(){code} or {code}name{code} of the enum.  Reverse lookup maps
27   * from value-to-enum will be cached forever.
28   *
29   * @since 0.8.1
30   */
31  final class EnumConverter {
32  
33      private final Map<Class<Enum>, Map<String, Enum>> cachedEnums = new ConcurrentHashMap<>();
34  
35      /**
36       * Lookup enum by {code}toString(){code} or {code}name{code}. If the enum cannot be found based on the
37       * {code}toString(){code} value then {code}Enum.valueOf{code} is used.
38       *
39       * @param <E> The enum type whose constant is to be returned
40       * @param enumType the {@code Class} object of the enum type from which to return a constant
41       * @param value the toString() or name value of the enum to return
42       * @return the enum constant of the specified enum type with the specified value
43       */
44      <E extends Enum<E>> E fromValue(Class<E> enumType, String value) {
45  
46          Assert.notNull(value, "[Assertion failed] - 'value' is required; it must not be null");
47  
48          // Get the reverse enum map
49          Map<String, E> reverseMap = getReverseLookupMap(enumType);
50  
51          // lookup the value
52          E result = reverseMap.get(value);
53  
54          // if null, then revert to Enum.valueOf
55          if (result == null) {
56              result = Enum.valueOf(enumType, value);
57          }
58          return result;
59      }
60  
61      private <E extends Enum> Map<String, E> getReverseLookupMap(Class<E> type) {
62  
63          Map<String, E> result = (Map<String, E>) cachedEnums.get(type);
64          if (result == null) {
65              result = Arrays.stream(type.getEnumConstants())
66                  .collect(Collectors.toMap(Enum::toString, e -> e));
67              cachedEnums.put((Class<Enum>) type, (Map<String, Enum>) result);
68          }
69          return result;
70      }
71  }
