16  package com.okta.sdk.helper;
17  
18  import com.okta.commons.lang.Assert;
19  import org.openapitools.client.model.*;
20  
21  public class HelperUtil {
22  
23      static Class<? extends Application> getApplicationType(Application application) {
24  
25          Assert.notNull(application);
26          Assert.notNull(application.getSignOnMode());
27  
28          switch (application.getSignOnMode()) {
29              case AUTO_LOGIN:
30                  return AutoLoginApplication.class;
31              case BASIC_AUTH:
32                  return BasicAuthApplication.class;
33              case BOOKMARK:
34                  return BookmarkApplication.class;
35              case BROWSER_PLUGIN:
36                  return BrowserPluginApplication.class;
37              case OPENID_CONNECT:
38                  return OpenIdConnectApplication.class;
39              case SAML_1_1:
40              case SAML_2_0:
41                  return SamlApplication.class;
42              case SECURE_PASSWORD_STORE:
43                  return SecurePasswordStoreApplication.class;
44              case WS_FEDERATION:
45                  return WsFederationApplication.class;
46              default:
47                  return Application.class;
48          }
49      }
50  
51      static Class<? extends Policy> getPolicyType(Policy policy) {
52  
53          Assert.notNull(policy);
54          Assert.notNull(policy.getType());
55  
56          switch (policy.getType()) {
57              case ACCESS_POLICY:
58                  return AccessPolicy.class;
59              case IDP_DISCOVERY:
60                  return IdentityProviderPolicy.class;
61              case MFA_ENROLL:
62                  return MultifactorEnrollmentPolicy.class;
63              case OAUTH_AUTHORIZATION_POLICY:
64                  return AuthorizationServerPolicy.class;
65              case OKTA_SIGN_ON:
66                  return OktaSignOnPolicy.class;
67              case PASSWORD:
68                  return PasswordPolicy.class;
69              case PROFILE_ENROLLMENT:
70                  return ProfileEnrollmentPolicy.class;
71          }
72  
73          return Policy.class;
74      }
75  
76      static Class<? extends UserFactor> getUserFactorType(UserFactor userFactor) {
77  
78          Assert.notNull(userFactor);
79          Assert.notNull(userFactor.getFactorType());
80  
81          switch (userFactor.getFactorType()) {
82              case CALL:
83                  return CallUserFactor.class;
84  
85              case EMAIL:
86                  return EmailUserFactor.class;
87  
88              case HOTP:
89                  return CustomHotpUserFactor.class;
90  
91              case PUSH:
92                  return PushUserFactor.class;
93  
94              case SMS:
95                  return SmsUserFactor.class;
96  
97              case QUESTION:
98                  return SecurityQuestionUserFactor.class;
99  
100             case TOKEN:
101             case TOKEN_HARDWARE:
102             case TOKEN_HOTP:
103             case TOKEN_SOFTWARE_TOTP:
104                 return TokenUserFactor.class;
105 
106             case U2F:
107                 return U2fUserFactor.class;
108 
109             case WEB:
110                 return WebUserFactor.class;
111 
112             case WEBAUTHN:
113                 return WebAuthnUserFactor.class;
114 
115             default:
116                 return UserFactor.class;
117         }
118     }
119 }
