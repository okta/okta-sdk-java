17  /**
18   * Helper class for Pagination related functions.
19   *
20   * @since 16.0.0
21   */
22  public class PaginationUtil {
23  
24      private final static Logger log = LoggerFactory.getLogger(PaginationUtil.class);
25  
26      /**
27       * Gets the 'after' resource id from ApiClient instance.
28       * e.g. a next page URL such as  <a href="https://example.okta.com/api/v1/users?after=100uekbsy586JToqdQ1d7&limit=100">...</a> will fetch
29       * an output 'after' string value of 100uekbsy586JToqdQ1d7
30       *
31       * @param apiClient {@link ApiClient} instance
32       * @return the 'after' resource id
33       */
34      public static String getAfter(ApiClient apiClient) {
35          return getAfter(getNextPage(apiClient));
36      }
37  
38      /**
39       * Gets the 'after' resource id from the next page URL string.
40       * e.g. a next page URL such as  <a href="https://example.okta.com/api/v1/users?after=100uekbsy586JToqdQ1d7&limit=100">...</a> will fetch
41       * an output 'after' string value of 100uekbsy586JToqdQ1d7
42       *
43       * @param nextPage the next page URL string
44       * @return the 'after' resource id
45       */
46      private static String getAfter(String nextPage) {
47  
48          String after;
49          URL url;
50  
51          try {
52              url = new URL(nextPage);
53              after = splitQuery(url).get("after");
54              log.debug("after: {}", after);
55              return after;
56          } catch (MalformedURLException | UnsupportedEncodingException e) {
57              return null;
58          }
59      }
60  
61      /**
62       * Gets the Next Page URL (to paginate) from ApiClient response header.
63       *
64       * @param apiClient the {@link ApiClient} instance
65       * @return the next page URL string
66       */
67      private static String getNextPage(ApiClient apiClient) {
68  
69          Assert.notNull(apiClient, "apiClient cannot be null");
70          Assert.notNull(apiClient.getResponseHeaders(), "apiClient is missing response headers");
71          Assert.notNull(apiClient.getResponseHeaders().get("link"), "apiClient is missing 'link' response headers");
72  
73          List<String> linkHeaders = apiClient.getResponseHeaders().get("link");
74  
75          String nextPage = null;
76  
77          for (String linkHeader : linkHeaders) {
78              String[] parts = linkHeader.split("; *");
79              String url = parts[0]
80                  .replaceAll("<", "")
81                  .replaceAll(">", "");
82              String rel = parts[1];
83              if (rel.equals("rel=\"next\"")) {
84                  nextPage = url;
85              }
86          }
87  
88          log.debug("Next Page: {}", nextPage);
89          return nextPage;
90      }
91  
92      /**
93       * Split a URL with query strings into name value pairs.
94       *
95       * @param url the url to split
96       * @return map of query string name value pairs
97       * @throws UnsupportedEncodingException If character encoding needs to be consulted
98       */
99      private static Map<String, String> splitQuery(URL url) throws UnsupportedEncodingException {
100 
101         Assert.notNull(url, "url cannot be null");
102 
103         Map<String, String> query_pairs = new LinkedHashMap<>();
104         String query = url.getQuery();
105         String[] pairs = query.split("&");
106         for (String pair : pairs) {
107             int index = pair.indexOf("=");
108             query_pairs.put(URLDecoder.decode(pair.substring(0, index), StandardCharsets.UTF_8.name()),
109                 URLDecoder.decode(pair.substring(index + 1), StandardCharsets.UTF_8.name()));
110         }
111         return query_pairs;
112     }
