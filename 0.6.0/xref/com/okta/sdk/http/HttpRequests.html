16  package com.okta.sdk.http;
17  
18  import com.okta.sdk.lang.Assert;
19  import com.okta.sdk.lang.Classes;
20  
21  import java.lang.reflect.Constructor;
22  
23  /**
24   * Utility factory class for creating {@link HttpRequest} instances for SDK users that do not already depend on the
25   * {@code HttpServletRequest} API.
26   *
27   * <p>Once you obtain either a {@code HttpServletRequest} or construct a {@link HttpRequest}, either may be
28   * authenticated (presumably to assert an identity for calls to your REST API)
29   * </p>
30   *
31   * @since 0.5.0
32   */
33  public final class HttpRequests {
34  
35      private static final Class<HttpRequestBuilder> HTTP_REQUEST_BUILDER =
36          Classes.forName("com.okta.sdk.impl.http.DefaultHttpRequestBuilder");
37  
38      /**
39       * Creates and returns a new {@link HttpRequestBuilder} that builds request instances that will be used for request
40       * authentication.
41       *
42       * <p>This method is only useful for SDK users that do not depend on the {@code HttpServletRequest} API, as
43       * the {@code application.authenticate*} methods accept that type natively.</p>
44       *
45       * @param method the source request's http method.
46       * @return a new HttpRequestBuilder that can be used to construct a new {@link HttpRequest} instance.
47       * @throws IllegalArgumentException if the method argument is {@code null}.
48       */
49      public static HttpRequestBuilder method(HttpMethod method) throws IllegalArgumentException {
50          Assert.notNull(method, "method argument is required.");
51  
52          Constructor<HttpRequestBuilder> ctor = Classes.getConstructor(HTTP_REQUEST_BUILDER, HttpMethod.class);
53          return Classes.instantiate(ctor, method);
54      }
55  }
