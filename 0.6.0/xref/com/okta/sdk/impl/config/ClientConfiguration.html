16  package com.okta.sdk.impl.config;
17  
18  import com.okta.sdk.cache.CacheConfigurationBuilder;
19  import com.okta.sdk.client.AuthenticationScheme;
20  import com.okta.sdk.impl.api.ClientCredentialsResolver;
21  import com.okta.sdk.impl.http.authc.RequestAuthenticatorFactory;
22  import com.okta.sdk.impl.util.BaseUrlResolver;
23  
24  import java.util.LinkedHashMap;
25  import java.util.Map;
26  
27  /**
28   * This class holds the default configuration properties.
29   *
30   * During application initialization all the properties found in the pre-defined locations that are
31   * defined by the user will be added here in the order defined in {@link com.okta.sdk.impl.client.DefaultClientBuilder}.
32   * Unset values will use default values from {@code com/okta/sdk/config/okta.yaml}.
33   *
34   * @since 0.5.0
35   */
36  public class ClientConfiguration {
37  
38      private String apiToken;
39      private ClientCredentialsResolver clientCredentialsResolver;
40      private boolean cacheManagerEnabled;
41      private long cacheManagerTtl;
42      private long cacheManagerTti;
43      private Map<String, CacheConfigurationBuilder> cacheManagerCaches = new LinkedHashMap<>();
44      private String baseUrl;
45      private int connectionTimeout;
46      private AuthenticationScheme authenticationScheme;
47      private RequestAuthenticatorFactory requestAuthenticatorFactory;
48      private int proxyPort;
49      private String proxyHost;
50      private String proxyUsername;
51      private String proxyPassword;
52      private BaseUrlResolver baseUrlResolver;
53  
54      public String getApiToken() {
55          return apiToken;
56      }
57  
58      public void setApiToken(String apiToken) {
59          this.apiToken = apiToken;
60      }
61  
62      public ClientCredentialsResolver getClientCredentialsResolver() {
63          return clientCredentialsResolver;
64      }
65  
66      public void setClientCredentialsResolver(ClientCredentialsResolver clientCredentialsResolver) {
67          this.clientCredentialsResolver = clientCredentialsResolver;
68      }
69  
70      public AuthenticationScheme getAuthenticationScheme() {
71          return authenticationScheme;
72      }
73  
74      public void setAuthenticationScheme(AuthenticationScheme authenticationScheme) {
75          this.authenticationScheme = authenticationScheme;
76      }
77  
78      public RequestAuthenticatorFactory getRequestAuthenticatorFactory() {
79          return requestAuthenticatorFactory;
80      }
81  
82      public void setRequestAuthenticatorFactory(RequestAuthenticatorFactory requestAuthenticatorFactory) {
83          this.requestAuthenticatorFactory = requestAuthenticatorFactory;
84      }
85  
86      public String getBaseUrl() {
87          return baseUrl;
88      }
89  
90      public void setBaseUrl(String baseUrl) {
91          this.baseUrl = baseUrl;
92      }
93  
94      public boolean isCacheManagerEnabled() {
95          return cacheManagerEnabled;
96      }
97  
98      public void setCacheManagerEnabled(boolean cacheManagerEnabled) {
99          this.cacheManagerEnabled = cacheManagerEnabled;
100     }
101 
102     public Map<String, CacheConfigurationBuilder> getCacheManagerCaches() {
103         return cacheManagerCaches;
104     }
105 
106     public void setCacheManagerCaches(Map<String, CacheConfigurationBuilder> cacheManagerCaches) {
107         this.cacheManagerCaches = cacheManagerCaches;
108     }
109 
110     /**
111      * Time to idle for cache manager in seconds
112      * @return seconds until time to idle expires
113      */
114     public long getCacheManagerTti() {
115         return cacheManagerTti;
116     }
117 
118     /**
119      * The cache manager's time to idle in seconds
120      * @param cacheManagerTti the time to idle in seconds
121      */
122     public void setCacheManagerTti(long cacheManagerTti) {
123         this.cacheManagerTti = cacheManagerTti;
124     }
125 
126     /**
127      * Time to live for cache manager in seconds
128      * @return seconds until time to live expires
129      */
130     public long getCacheManagerTtl() {
131         return cacheManagerTtl;
132     }
133 
134     /**
135      * The cache manager's time to live in seconds
136      * @param cacheManagerTtl the time to live in seconds
137      */
138     public void setCacheManagerTtl(long cacheManagerTtl) {
139         this.cacheManagerTtl = cacheManagerTtl;
140     }
141 
142     /**
143      * Connection timeout in seconds
144      * @return seconds until connection timeout
145      */
146     public int getConnectionTimeout() {
147         return connectionTimeout;
148     }
149 
150     /**
151      * Connection timeout in seconds.
152      *
153      * @param connectionTimeout the timeout value in seconds
154      */
155     public void setConnectionTimeout(int connectionTimeout) {
156         this.connectionTimeout = connectionTimeout;
157     }
158 
159     public String getProxyHost() {
160         return proxyHost;
161     }
162 
163     public void setProxyHost(String proxyHost) {
164         this.proxyHost = proxyHost;
165     }
166 
167     public String getProxyPassword() {
168         return proxyPassword;
169     }
170 
171     public void setProxyPassword(String proxyPassword) {
172         this.proxyPassword = proxyPassword;
173     }
174 
175     public int getProxyPort() {
176         return proxyPort;
177     }
178 
179     public void setProxyPort(int proxyPort) {
180         this.proxyPort = proxyPort;
181     }
182 
183     public String getProxyUsername() {
184         return proxyUsername;
185     }
186 
187     public void setProxyUsername(String proxyUsername) {
188         this.proxyUsername = proxyUsername;
189     }
190 
191     public BaseUrlResolver getBaseUrlResolver() {
192         return baseUrlResolver;
193     }
194 
195     public void setBaseUrlResolver(BaseUrlResolver baseUrlResolver) {
196         this.baseUrlResolver = baseUrlResolver;
197     }
198 
199     @Override
200     public String toString() {
201         return "ClientConfiguration{" +
202                 ", cacheManagerTtl=" + cacheManagerTtl +
203                 ", cacheManagerTti=" + cacheManagerTti +
204                 ", cacheManagerCaches=" + cacheManagerCaches +
205                 ", baseUrl='" + baseUrl + '\'' +
206                 ", connectionTimeout=" + connectionTimeout +
207                 ", authenticationScheme=" + authenticationScheme +
208                 ", proxyPort=" + proxyPort +
209                 ", proxyHost='" + proxyHost + '\'' +
210                 ", proxyUsername='" + proxyUsername + '\'' +
211                 ", proxyPassword='" + proxyPassword + '\'' +
212                 '}';
213     }
214 }
