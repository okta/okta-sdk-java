49  /**
50   * Interceptor that handle DPoP handshake during auth and adds DPoP header to regular requests.
51   * It is always enabled, but is only active when a DPoP error is received during auth.
52   *
53   * @see <a href="https://developer.okta.com/docs/guides/dpop/oktaresourceserver/main/">documentation</a>
54   */
55  public class DPoPInterceptor implements ExecChainHandler {
56  
57      private static final Logger log = LoggerFactory.getLogger(DPoPInterceptor.class);
58  
59      private static final ObjectMapper OBJECT_MAPPER = new ObjectMapper();
60      private static final String DPOP_HEADER = "DPoP";
61      //nonce is valid for 24 hours, but can only refresh it when doing a token request => start refreshing after 22 hours
62      private static final int NONCE_VALID_SECONDS = 60 * 60 * 22;
63      //MessageDigest is not thread-safe, need one per thread
64      private static final ThreadLocal<MessageDigest> SHA256 = ThreadLocal.withInitial(() -> {
65          try {
66              return MessageDigest.getInstance("SHA-256");
67          } catch (NoSuchAlgorithmException e) {
68              throw new RuntimeException(e);
69          }
70      });
71  
72      //if null, means dpop is not enabled yet
73      private PrivateJwk<PrivateKey, PublicKey, ?> jwk;
74      private String nonce;
75      private Instant nonceValidUntil;
76  
77      @Override
78      public ClassicHttpResponse execute(ClassicHttpRequest request, ExecChain.Scope scope, ExecChain execChain)
79          throws IOException, HttpException {
80          boolean tokenRequest = request.getRequestUri().equals(TOKEN_URI);
81          if (tokenRequest && nonce != null && nonceValidUntil.isBefore(Instant.now())) {
82              log.debug("DPoP nonce expired, will refresh it");
83              nonce = null;
84              nonceValidUntil = null;
85          }
86          if (jwk != null) {
87              processRequest(request, tokenRequest);
88          }
89          ClassicHttpResponse response = execChain.proceed(request, scope);
90          if (tokenRequest) {
91              if (response.getCode() == 200 && nonce != null) {
92                  log.info("DPoP handshake successful");
93              }
94              if (response.getCode() == 400) {
95                  JsonNode errorBody = OBJECT_MAPPER.readTree(response.getEntity().getContent());
96                  Header nonceHeader = response.getFirstHeader("dpop-nonce");
97                  DPopHandshakeState handshakeState = handleHandshakeResponse(errorBody.get("error"), nonceHeader);
98                  throw new DPoPHandshakeException(handshakeState, OBJECT_MAPPER.writeValueAsString(errorBody));
99              }
100         }
101         return response;
102     }
103 
104     private void processRequest(HttpRequest request, boolean tokenRequest) {
105         JwtBuilder builder = Jwts.builder()
106             .header()
107             .type("dpop+jwt")
108             .jwk(jwk.toPublicJwk())
109             .and()
110             .claim("htm", request.getMethod())
111             .claim("htu", getUriWithoutQueryString(request))
112             .claim("jti", UUID.randomUUID().toString())
113             .issuedAt(new Date());
114         Header authorization = request.getFirstHeader("Authorization");
115         if (authorization != null) {
116             //already authenticated, need to replace Authorization header prefix and set ath claim
117             //the DPoP prefix might already be set if the request is retried
118             String token = StringUtils.substringAfter(authorization.getValue(), " ");
119             request.setHeader("Authorization", DPOP_HEADER + " " + token);
120             byte[] ath = SHA256.get().digest(token.getBytes(StandardCharsets.US_ASCII));
121             builder.claim("ath", Encoders.BASE64URL.encode(ath));
122         } else if (tokenRequest && nonce != null) {
123             //still in handshake, need to set nonce
124             builder.claim("nonce", nonce);
125         }
126         request.addHeader(DPOP_HEADER, builder.signWith(jwk.toKeyPair().getPrivate()).compact());
127     }
128 
129     private String getUriWithoutQueryString(HttpRequest request) {
130         try {
131             String urlWithoutQueryString = StringUtils.substringBefore(request.getUri().toString(), "?");
132             return URLDecoder.decode(urlWithoutQueryString, StandardCharsets.UTF_8.name())
133                 .replace("%", "%25") //must be replaced first
134                 .replace(" ", "%20")
135                 .replace("\"", "%22")
136                 .replace("#", "%23");
137         } catch (Exception e) {
138             throw new RuntimeException(e);
139         }
140     }
141 
142     private DPopHandshakeState handleHandshakeResponse(JsonNode errorField, Header nonceHeader) {
143         if (errorField != null && errorField.isTextual()) {
144             switch (errorField.textValue()) {
145                 case "invalid_dpop_proof": {
146                     if (jwk != null) {
147                         return DPopHandshakeState.REPEATED_INVALID_DPOP_PROOF;
148                     }
149                     log.info("DPoP detected, beginning handshake");
150                     this.jwk = Jwks.builder().keyPair(Jwts.SIG.ES256.keyPair().build()).build();
151                     return DPopHandshakeState.FIRST_INVALID_DPOP_PROOF;
152                 }
153                 case "use_dpop_nonce": {
154                     if (nonce != null) {
155                         return DPopHandshakeState.REPEATED_USE_DPOP_NONCE;
156                     }
157                     if (nonceHeader == null) {
158                         return DPopHandshakeState.MISSING_DPOP_NONCE_HEADER;
159                     }
160                     log.info("DPoP nonce obtained, finalizing handshake");
161                     this.nonce = nonceHeader.getValue();
162                     this.nonceValidUntil = Instant.now().plusSeconds(NONCE_VALID_SECONDS);
163                     return DPopHandshakeState.FIRST_USE_DPOP_NONCE;
164                 }
165                 default:
166                     return DPopHandshakeState.UNEXPECTED_STATE;
167             }
168         }
169         return DPopHandshakeState.UNEXPECTED_STATE;
170     }
