16  package com.okta.sdk.http;
17  
18  import java.util.Map;
19  
20  /**
21   * A <a href="http://en.wikipedia.org/wiki/Builder_pattern">Builder design pattern</a> used to
22   * construct {@link HttpRequest} instances.
23   *
24   * <p>The {@link HttpRequestBuilder} is useful to build {@link HttpRequest} instances for developers that don't
25   * have access to the Servlet container, and therefore, cannot execute operations using implementations
26   * of the {@code javax.servlet.HttpServlet} api.</p>
27   *
28   * @see HttpRequest
29   * @see HttpRequests
30   * @since 0.5.0
31   */
32  public interface HttpRequestBuilder {
33  
34      /**
35       * Sets the HTTP Headers that will be present in the resulting request instance.
36       *
37       * @param headers the HTTP Headers that will be present in the resulting built request instance.
38       * @return the builder instance for method chaining.
39       * @throws IllegalArgumentException if the method argument is {@code null}.
40       */
41      HttpRequestBuilder headers(Map<String, String[]> headers) throws IllegalArgumentException;
42  
43      /**
44       * Sets the request parameters that will be present in the resulting request instance.
45       *
46       * @param parameters the request parameters that will be present in the resulting request instance.
47       * @return the builder instance for method chaining.
48       */
49      HttpRequestBuilder parameters(Map<String, String[]> parameters);
50  
51      /**
52       * Adds a request header that will be part of the headers map in the resulting request instance.
53       *
54       * @param name the name of the HTTP header that will be present in the resulting HTTP request instance.
55       * @param value the value of the HTTP header that will be present in the resulting HTTP request instance.
56       * @return the builder instance for method chaining.
57       * @throws IllegalArgumentException if the name or value arguments are {@code null}.
58       *
59       */
60      HttpRequestBuilder addHeader(String name, String[] value) throws IllegalArgumentException;
61  
62      /**
63       * Adds a request parameter that will be part of the parameters map in the resulting request instance.
64       *
65       * @param name the name of the HTTP parameter that will be present in the resulting HTTP request instance.
66       * @param value the value of the HTTP parameter that will be present in the resulting HTTP request instance.
67       * @return the builder instance for method chaining.
68       * @throws IllegalArgumentException if the name or value arguments are {@code null}.
69       *
70       */
71      HttpRequestBuilder addParameter(String name, String[] value) throws IllegalArgumentException;
72  
73      /**
74       * Sets the query parameters that will be present in the resulting request instance.
75       *
76       * @param queryParameters the query parameters that will be present in the resulting request instance.
77       * @return the builder instance for method chaining.
78       */
79      HttpRequestBuilder queryParameters(String queryParameters);
80  
81      /**
82       * Returns a new {@code HttpRequest} instance reflecting the current builder state.
83       *
84       * @return a new {@code HttpRequest} instance reflecting the current builder state.
85       */
86      HttpRequest build();
87  }
