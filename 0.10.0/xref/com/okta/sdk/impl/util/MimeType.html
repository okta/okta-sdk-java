34  /**
35   * Represents a MIME Type, as originally defined in RFC 2046 and subsequently used in
36   * other Internet protocols including HTTP.
37   *
38   * <p>This class, however, does not contain support for the q-parameters used
39   * in HTTP content negotiation. Those can be found in the sub-class
40   * {@code org.springframework.http.MediaType} in the {@code spring-web} module.
41   *
42   * <p>Consists of a {@linkplain #getType() type} and a {@linkplain #getSubtype() subtype}.
43   * Also has functionality to parse media types from a string using
44   * {@link #valueOf(String)}. For more parsing options see {@link MimeTypeUtils}.
45   *
46   * @author Arjen Poutsma
47   * @author Juergen Hoeller
48   * @author Rossen Stoyanchev
49   * @author Sam Brannen
50   * @since 4.0
51   * @see MimeTypeUtils
52   */
53  public class MimeType implements Comparable<MimeType>, Serializable {

58      protected static final String WILDCARD_TYPE = "*";

60      private static final String PARAM_CHARSET = "charset";

62      private static final BitSet TOKEN;
63  
64      static {
65          // variable names refer to RFC 2616, section 2.2
66          BitSet ctl = new BitSet(128);
67          for (int i = 0; i <= 31; i++) {
68              ctl.set(i);
69          }
70          ctl.set(127);
71  
72          BitSet separators = new BitSet(128);
73          separators.set('(');
74          separators.set(')');
75          separators.set('<');
76          separators.set('>');
77          separators.set('@');
78          separators.set(',');
79          separators.set(';');
80          separators.set(':');
81          separators.set('\\');
82          separators.set('\"');
83          separators.set('/');
84          separators.set('[');
85          separators.set(']');
86          separators.set('?');
87          separators.set('=');
88          separators.set('{');
89          separators.set('}');
90          separators.set(' ');
91          separators.set('\t');
92  
93          TOKEN = new BitSet(128);
94          TOKEN.set(0, 128);
95          TOKEN.andNot(ctl);
96          TOKEN.andNot(separators);
97      }

163     /**
164      * Create a new {@code MimeType} for the given type, subtype, and parameters.
165      * @param type the primary type
166      * @param subtype the subtype
167      * @param parameters the parameters, may be {@code null}
168      * @throws IllegalArgumentException if any of the parameters contains illegal characters
169      */
170     public MimeType(String type, String subtype, Map<String, String> parameters) {
171         Assert.hasLength(type, "'type' must not be empty");
172         Assert.hasLength(subtype, "'subtype' must not be empty");
173         checkToken(type);
174         checkToken(subtype);
175         this.type = type.toLowerCase(Locale.ENGLISH);
176         this.subtype = subtype.toLowerCase(Locale.ENGLISH);
177         if (!com.okta.sdk.lang.Collections.isEmpty(parameters)) {
178             Map<String, String> map = new LinkedCaseInsensitiveMap<>(parameters.size(), Locale.ENGLISH);
179             for (Map.Entry<String, String> entry : parameters.entrySet()) {
180                 String attribute = entry.getKey();
181                 String value = entry.getValue();
182                 checkParameters(attribute, value);
183                 map.put(attribute, value);
184             }
185             this.parameters = Collections.unmodifiableMap(map);
186         }
187         else {
188             this.parameters = Collections.emptyMap();
189         }
190     }

192     /**
193      * Checks the given token string for illegal characters, as defined in RFC 2616,
194      * section 2.2.
195      * @throws IllegalArgumentException in case of illegal characters
196      * @see <a href="http://tools.ietf.org/html/rfc2616#section-2.2">HTTP 1.1, section 2.2</a>
197      */
198     private void checkToken(String token) {
199         for (int i = 0; i < token.length(); i++ ) {
200             char ch = token.charAt(i);
201             if (!TOKEN.get(ch)) {
202                 throw new IllegalArgumentException("Invalid token character '" + ch + "' in token \"" + token + "\"");
203             }
204         }
205     }
206 
207     protected void checkParameters(String attribute, String value) {
208         Assert.hasLength(attribute, "'attribute' must not be empty");
209         Assert.hasLength(value, "'value' must not be empty");
210         checkToken(attribute);
211         if (PARAM_CHARSET.equals(attribute)) {
212             value = unquote(value);
213             Charset.forName(value);
214         }
215         else if (!isQuotedString(value)) {
216             checkToken(value);
217         }
218     }
219 
220     private boolean isQuotedString(String s) {
221         if (s.length() < 2) {
222             return false;
223         }
224         else {
225             return ((s.startsWith("\"") && s.endsWith("\"")) || (s.startsWith("'") && s.endsWith("'")));
226         }
227     }
228 
229     protected String unquote(String s) {
230         if (s == null) {
231             return null;
232         }
233         return isQuotedString(s) ? s.substring(1, s.length() - 1) : s;
234     }

245     /**
246      * Indicates whether the {@linkplain #getSubtype() subtype} is the wildcard
247      * character <code>&#42;</code> or the wildcard character followed by a suffix
248      * (e.g. <code>&#42;+xml</code>).
249      * @return whether the subtype is a wildcard
250      */
251     public boolean isWildcardSubtype() {
252         return WILDCARD_TYPE.equals(getSubtype()) || getSubtype().startsWith("*+");
253     }

255     /**
256      * Indicates whether this media type is concrete, i.e. whether neither the type
257      * nor the subtype is a wildcard character <code>&#42;</code>.
258      * @return whether this media type is concrete

260     public boolean isConcrete() {
261         return !isWildcardType() && !isWildcardSubtype();

307     /**
308      * Indicate whether this {@code MediaType} includes the given media type.
309      * <p>For instance, {@code text/*} includes {@code text/plain} and {@code text/html},
310      * and {@code application/*+xml} includes {@code application/soap+xml}, etc. This
311      * method is <b>not</b> symmetric.
312      * @param other the reference media type with which to compare
313      * @return {@code true} if this media type includes the given media type;
314      * {@code false} otherwise
315      */
316     public boolean includes(MimeType other) {
317         if (other == null) {
318             return false;
319         }
320         if (this.isWildcardType()) {
321             // */* includes anything
322             return true;
323         }
324         else if (getType().equals(other.getType())) {
325             if (getSubtype().equals(other.getSubtype())) {
326                 return true;
327             }
328             if (this.isWildcardSubtype()) {
329                 // wildcard with suffix, e.g. application/*+xml
330                 int thisPlusIdx = getSubtype().indexOf('+');
331                 if (thisPlusIdx == -1) {
332                     return true;
333                 }
334                 else {
335                     // application/*+xml includes application/soap+xml
336                     int otherPlusIdx = other.getSubtype().indexOf('+');
337                     if (otherPlusIdx != -1) {
338                         String thisSubtypeNoSuffix = getSubtype().substring(0, thisPlusIdx);
339                         String thisSubtypeSuffix = getSubtype().substring(thisPlusIdx + 1);
340                         String otherSubtypeSuffix = other.getSubtype().substring(otherPlusIdx + 1);
341                         if (thisSubtypeSuffix.equals(otherSubtypeSuffix) && WILDCARD_TYPE.equals(thisSubtypeNoSuffix)) {
342                             return true;
343                         }
344                     }
345                 }
346             }
347         }
348         return false;
349     }

351     /**
352      * Indicate whether this {@code MediaType} is compatible with the given media type.
353      * <p>For instance, {@code text/*} is compatible with {@code text/plain},
354      * {@code text/html}, and vice versa. In effect, this method is similar to
355      * {@link #includes}, except that it <b>is</b> symmetric.
356      * @param other the reference media type with which to compare
357      * @return {@code true} if this media type is compatible with the given media type;
358      * {@code false} otherwise
359      */
360     public boolean isCompatibleWith(MimeType other) {
361         if (other == null) {
362             return false;
363         }
364         if (isWildcardType() || other.isWildcardType()) {
365             return true;
366         }
367         else if (getType().equals(other.getType())) {
368             if (getSubtype().equals(other.getSubtype())) {
369                 return true;
370             }
371             // wildcard with suffix? e.g. application/*+xml
372             if (this.isWildcardSubtype() || other.isWildcardSubtype()) {
373 
374                 int thisPlusIdx = getSubtype().indexOf('+');
375                 int otherPlusIdx = other.getSubtype().indexOf('+');
376 
377                 if (thisPlusIdx == -1 && otherPlusIdx == -1) {
378                     return true;
379                 }
380                 else if (thisPlusIdx != -1 && otherPlusIdx != -1) {
381                     String thisSubtypeNoSuffix = getSubtype().substring(0, thisPlusIdx);
382                     String otherSubtypeNoSuffix = other.getSubtype().substring(0, otherPlusIdx);
383 
384                     String thisSubtypeSuffix = getSubtype().substring(thisPlusIdx + 1);
385                     String otherSubtypeSuffix = other.getSubtype().substring(otherPlusIdx + 1);
386 
387                     if (thisSubtypeSuffix.equals(otherSubtypeSuffix) &&
388                             (WILDCARD_TYPE.equals(thisSubtypeNoSuffix) || WILDCARD_TYPE.equals(otherSubtypeNoSuffix))) {
389                         return true;
390                     }
391                 }
392             }
393         }
394         return false;
395     }

398     @Override
399     public boolean equals(Object other) {
400         if (this == other) {
401             return true;
402         }
403         if (!(other instanceof MimeType)) {
404             return false;
405         }
406         MimeType otherType = (MimeType) other;
407         return (this.type.equalsIgnoreCase(otherType.type) &&
408                 this.subtype.equalsIgnoreCase(otherType.subtype) &&
409                 parametersAreEqual(otherType));
410     }

412     /**
413      * Determine if the parameters in this {@code MimeType} and the supplied
414      * {@code MimeType} are equal, performing case-insensitive comparisons
415      * for {@link Charset}s.
416      * @since 4.2
417      */
418     private boolean parametersAreEqual(MimeType other) {
419         if (this.parameters.size() != other.parameters.size()) {
420             return false;
421         }
422 
423         for (String key : this.parameters.keySet()) {
424             if (!other.parameters.containsKey(key)) {
425                 return false;
426             }
427 
428             if (PARAM_CHARSET.equals(key)) {
429                 if (!Objects.nullSafeEquals(getCharset(), other.getCharset())) {
430                     return false;
431                 }
432             }
433             else if (!Objects.nullSafeEquals(this.parameters.get(key), other.parameters.get(key))) {
434                 return false;
435             }
436         }
437 
438         return true;
439     }

443         int result = this.type.hashCode();
444         result = 31 * result + this.subtype.hashCode();
445         result = 31 * result + this.parameters.hashCode();

456     protected void appendTo(StringBuilder builder) {

458         builder.append('/');
459         builder.append(this.subtype);
460         appendTo(this.parameters, builder);

463     private void appendTo(Map<String, String> map, StringBuilder builder) {
464         for (Map.Entry<String, String> entry : map.entrySet()) {
465             builder.append(';');
466             builder.append(entry.getKey());
467             builder.append('=');
468             builder.append(entry.getValue());
469         }
470     }

472     /**
473      * Compares this {@code MediaType} to another alphabetically.
474      * @param other media type to compare to
475      * @see MimeTypeUtils#sortBySpecificity(List)
476      */
477     @Override
478     public int compareTo(MimeType other) {
479         int comp = getType().compareToIgnoreCase(other.getType());
480         if (comp != 0) {
481             return comp;
482         }
483         comp = getSubtype().compareToIgnoreCase(other.getSubtype());
484         if (comp != 0) {
485             return comp;
486         }
487         comp = getParameters().size() - other.getParameters().size();
488         if (comp != 0) {
489             return comp;
490         }
491         TreeSet<String> thisAttributes = new TreeSet<>(String.CASE_INSENSITIVE_ORDER);
492         thisAttributes.addAll(getParameters().keySet());
493         TreeSet<String> otherAttributes = new TreeSet<>(String.CASE_INSENSITIVE_ORDER);
494         otherAttributes.addAll(other.getParameters().keySet());
495         Iterator<String> thisAttributesIterator = thisAttributes.iterator();
496         Iterator<String> otherAttributesIterator = otherAttributes.iterator();
497         while (thisAttributesIterator.hasNext()) {
498             String thisAttribute = thisAttributesIterator.next();
499             String otherAttribute = otherAttributesIterator.next();
500             comp = thisAttribute.compareToIgnoreCase(otherAttribute);
501             if (comp != 0) {
502                 return comp;
503             }
504             String thisValue = getParameters().get(thisAttribute);
505             String otherValue = other.getParameters().get(otherAttribute);
506             if (otherValue == null) {
507                 otherValue = "";
508             }
509             comp = thisValue.compareTo(otherValue);
510             if (comp != 0) {
511                 return comp;
512             }
513         }
514         return 0;
515     }

537     public static class SpecificityComparator<T extends MimeType> implements Comparator<T> {
538 
539         @Override
540         public int compare(T mimeType1, T mimeType2) {
541             if (mimeType1.isWildcardType() && !mimeType2.isWildcardType()) { // */* < audio/*
542                 return 1;
543             }
544             else if (mimeType2.isWildcardType() && !mimeType1.isWildcardType()) { // audio/* > */*
545                 return -1;
546             }
547             else if (!mimeType1.getType().equals(mimeType2.getType())) { // audio/basic == text/html
548                 return 0;
549             }
550             else { // mediaType1.getType().equals(mediaType2.getType())
551                 if (mimeType1.isWildcardSubtype() && !mimeType2.isWildcardSubtype()) { // audio/* < audio/basic
552                     return 1;
553                 }
554                 else if (mimeType2.isWildcardSubtype() && !mimeType1.isWildcardSubtype()) { // audio/basic > audio/*
555                     return -1;
556                 }
557                 else if (!mimeType1.getSubtype().equals(mimeType2.getSubtype())) { // audio/basic == audio/wave
558                     return 0;
559                 }
560                 else { // mediaType2.getSubtype().equals(mediaType2.getSubtype())
561                     return compareParameters(mimeType1, mimeType2);
562                 }
563             }
564         }
565 
566         protected int compareParameters(T mimeType1, T mimeType2) {
567             int paramsSize1 = mimeType1.getParameters().size();
568             int paramsSize2 = mimeType2.getParameters().size();
569             return (paramsSize2 < paramsSize1 ? -1 : (paramsSize2 == paramsSize1 ? 0 : 1)); // audio/basic;level=1 < audio/basic
570         }
571     }
