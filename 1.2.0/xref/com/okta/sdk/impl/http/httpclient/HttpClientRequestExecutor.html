159      * Creates a new {@code HttpClientRequestExecutor} using the specified {@code ClientCredentials} and optional {@code Proxy}
160      * configuration.
161      * @param clientCredentials the Okta account API Key that will be used to authenticate the client with Okta's API sever
162      * @param proxy the HTTP proxy to be used when communicating with the Okta API server (can be null)
163      * @param authenticationScheme the HTTP authentication scheme to be used when communicating with the Okta API server.
164      *                             If null, then SSWS will be used.

166     @Deprecated
167     public HttpClientRequestExecutor(ClientCredentials clientCredentials, Proxy proxy, AuthenticationScheme authenticationScheme, RequestAuthenticatorFactory requestAuthenticatorFactory, Integer connectionTimeout) {
168         Assert.notNull(clientCredentials, "clientCredentials argument is required.");
169         Assert.isTrue(connectionTimeout >= 0, "Timeout cannot be a negative number.");
170 
171         RequestAuthenticatorFactory factory = (requestAuthenticatorFactory != null)
172                 ? requestAuthenticatorFactory
173                 : new DefaultRequestAuthenticatorFactory();
174 
175         this.requestAuthenticator = factory.create(authenticationScheme, clientCredentials);
176 
177         PoolingHttpClientConnectionManager connMgr = new PoolingHttpClientConnectionManager();
178 
179         if (MAX_CONNECTIONS_TOTAL >= MAX_CONNECTIONS_PER_ROUTE) {
180             connMgr.setDefaultMaxPerRoute(MAX_CONNECTIONS_PER_ROUTE);
181             connMgr.setMaxTotal(MAX_CONNECTIONS_TOTAL);
182         } else {
183             connMgr.setDefaultMaxPerRoute(DEFAULT_MAX_CONNECTIONS_PER_ROUTE);
184             connMgr.setMaxTotal(DEFAULT_MAX_CONNECTIONS_TOTAL);
185 
186             log.warn(
187                 "{} ({}) is less than {} ({}). " +
188                 "Reverting to defaults: connectionMaxTotal ({}) and connectionMaxPerRoute ({}).",
189                 MAX_CONNECTIONS_TOTAL_PROPERTY_KEY, MAX_CONNECTIONS_TOTAL,
190                 MAX_CONNECTIONS_PER_ROUTE_PROPERTY_KEY, MAX_CONNECTIONS_PER_ROUTE,
191                 DEFAULT_MAX_CONNECTIONS_TOTAL, DEFAULT_MAX_CONNECTIONS_PER_ROUTE
192             );
193         }
194 
195         // The connectionTimeout value is specified in seconds in Okta configuration settings.
196         // Therefore, multiply it by 1000 to be milliseconds since RequestConfig expects milliseconds.
197         int connectionTimeoutAsMilliseconds = connectionTimeout * 1000;
198 
199         RequestConfig requestConfig = RequestConfig.custom().setConnectTimeout(connectionTimeoutAsMilliseconds)
200                 .setSocketTimeout(connectionTimeoutAsMilliseconds).setRedirectsEnabled(false).build();
201 
202         ConnectionConfig connectionConfig = ConnectionConfig.custom().setCharset(Consts.UTF_8).build();
203 
204         HttpClientBuilder httpClientBuilder = HttpClientBuilder.create().setDefaultRequestConfig(requestConfig)
205                 .disableCookieManagement().setDefaultConnectionConfig(connectionConfig).setConnectionManager(connMgr);
206 
207         this.httpClientRequestFactory = new HttpClientRequestFactory(requestConfig);
208 
209         if (proxy != null) {
210             //We have some proxy setting to use!
211             HttpHost httpProxyHost = new HttpHost(proxy.getHost(), proxy.getPort());
212             httpClientBuilder.setProxy(httpProxyHost);
213 
214             if (proxy.isAuthenticationRequired()) {
215                 AuthScope authScope = new AuthScope(proxy.getHost(), proxy.getPort());
216                 Credentials credentials = new UsernamePasswordCredentials(proxy.getUsername(), proxy.getPassword());
217                 CredentialsProvider credentialsProviderProvider = new BasicCredentialsProvider();
218                 credentialsProviderProvider.setCredentials(authScope, credentials);
219                 httpClientBuilder.setDefaultCredentialsProvider(credentialsProviderProvider);
220             }
221         }
222 
223         this.httpClient = httpClientBuilder.build();
224     }

248     @Override
249     public Response executeRequest(Request request) throws RestException {
250 
251         Assert.notNull(request, "Request argument cannot be null.");
252 
253         int retryCount = 0;
254         URI redirectUri = null;
255         HttpEntity entity = null;
256         HttpResponse httpResponse = null;
257         String requestId = null;
258         Timer timer = new Timer();
259 
260         // Make a copy of the original request params and headers so that we can
261         // permute them in the loop and start over with the original every time.
262         QueryString originalQuery = new QueryString();
263         originalQuery.putAll(request.getQueryString());
264 
265         HttpHeaders originalHeaders = new HttpHeaders();
266         originalHeaders.putAll(request.getHeaders());
267 
268         while (true) {
269 
270             if (redirectUri != null) {
271                 request = new DefaultRequest(
272                         request.getMethod(),
273                         redirectUri.toString(),
274                         null,
275                         null,
276                         request.getBody(),
277                         request.getHeaders().getContentLength()
278                 );
279             }
280 
281             if (retryCount > 0) {
282                 request.setQueryString(originalQuery);
283                 request.setHeaders(originalHeaders);
284 
285                 // remember the request-id header if we need to retry
286                 if (requestId == null) {
287                     requestId = getRequestId(httpResponse);
288                 }
289             }
290 
291             // Sign the request
292             this.requestAuthenticator.authenticate(request);
293 
294             HttpRequestBase httpRequest = this.httpClientRequestFactory.createHttpClientRequest(request, entity);
295 
296             if (httpRequest instanceof HttpEntityEnclosingRequest) {
297                 entity = ((HttpEntityEnclosingRequest) httpRequest).getEntity();
298             }
299 
300             try {
301                 // We don't want to treat a redirect like a retry,
302                 // so if redirectUri is not null, we won't pause
303                 // before executing the request below.
304                 if (retryCount > 0 && redirectUri == null) {
305                     try {
306                         // if we cannot pause, then return the original response
307                         pauseBeforeRetry(retryCount, httpResponse, timer.split());
308                     } catch (RestException e) {
309                         if (log.isDebugEnabled()) {
310                             log.warn("Unable to pause for retry: {}", e.getMessage(), e);
311                         } else {
312                             log.warn("Unable to pause for retry: {}", e.getMessage());
313                         }
314 
315                         return toSdkResponse(httpResponse);
316                     }
317 
318                     resetStream(entity);
319                 }
320 
321                 // reset redirectUri so that if there is an exception, we will pause on retry
322                 redirectUri = null;
323                 retryCount++;
324 
325                 // include X-Okta headers when retrying
326                 setOktaHeaders(request, requestId, retryCount);
327 
328                 httpResponse = httpClient.execute(httpRequest);
329 
330                 if (isRedirect(httpResponse)) {
331                     redirectUri = getRedirectUri(httpResponse);
332                     httpRequest.setURI(redirectUri);
333                 } else {
334 
335                     Response response = toSdkResponse(httpResponse);
336 
337                     //allow the loop to continue to execute a retry request
338                     if (!shouldRetry(response, retryCount, timer.split())) {
339                         return response;
340                     }
341                 }
342             } catch (Throwable t) {
343                 log.warn("Unable to execute HTTP request: ", t.getMessage(), t);
344 
345                 if (!shouldRetry(httpRequest, t, retryCount, timer.split())) {
346                     throw new RestException("Unable to execute HTTP request: " + t.getMessage(), t);
347                 }
348             } finally {
349                 try {
350                     httpResponse.getEntity().getContent().close();
351                 } catch (Throwable ignored) { // NOPMD
352                 }
353             }
354         }
355     }

358      * Resets entities InputStream if supported

360     private void resetStream(HttpEntity entity) throws IOException {
361         if (entity != null) {
362             InputStream content = entity.getContent();

364                 content.reset();
365             }

370      * Adds {@code X-Okta-Retry-For} and {@code X-Okta-Retry-Count} headers to request if not null/empty or zero.
371      *
372      * @param request the request to add headers too
373      * @param requestId request ID of the original request that failed
374      * @param retryCount the number of times the request has been retried
375      */
376     private void setOktaHeaders(Request request, String requestId, int retryCount) {
377         if (Strings.hasText(requestId)) {
378             request.getHeaders().add("X-Okta-Retry-For", requestId);
379         }
380         if (retryCount > 1) {
381             request.getHeaders().add("X-Okta-Retry-Count", Integer.toString(retryCount));
382         }
383     }

392     private String getOnlySingleHeaderValue(HttpResponse response, String name) {
393         Header[] headers = response.getHeaders(name);

407     private boolean isRedirect(HttpResponse response) {
408         int status = response.getStatusLine().getStatusCode();
409         return (status == HttpStatus.SC_MOVED_PERMANENTLY ||
410                 status == HttpStatus.SC_MOVED_TEMPORARILY ||
411                 status == HttpStatus.SC_TEMPORARY_REDIRECT) &&
412                 response.getHeaders("Location") != null &&
413                 response.getHeaders("Location").length > 0;
414     }

416     private URI getRedirectUri(HttpResponse httpResponse) {
417         Header[] locationHeaders = httpResponse.getHeaders("Location");
418         String location = locationHeaders[0].getValue();
419         log.debug("Redirecting to: {}", location);
420         return URI.create(location);
421     }

423     /**
424      * Exponential sleep on failed request to avoid flooding a service with
425      * retries.
426      *
427      * @param retries           Current retry count.
428      */
429     private void pauseBeforeRetry(int retries, HttpResponse httpResponse, long timeElapsed) throws RestException {
430         long delay = -1;
431         long timeElapsedLeft = maxElapsedMillis - timeElapsed;
432         if (backoffStrategy != null) {
433             delay = Math.min(this.backoffStrategy.getDelayMillis(retries), timeElapsedLeft);
434         } else if (httpResponse != null && httpResponse.getStatusLine().getStatusCode() == 429) {
435             delay = get429DelayMillis(httpResponse);
436             if (!shouldRetry(retries, timeElapsed + delay)) {
437                 throw new RestException("Cannot retry request, next request will exceed retry configuration.");
438             }
439             log.debug("429 detected, will retry in {}ms, attempt number: {}", delay, retries);
440         }
441 
442         // default / fallback strategy
443         if (delay < 0) {
444             delay = Math.min(getDefaultDelayMillis(retries), timeElapsedLeft);
445         }
446 
447         log.debug("Retryable condition detected, will retry in {}ms, attempt number: {}", delay, retries);
448 
449         try {
450             Thread.sleep(delay);
451         } catch (InterruptedException e) {
452             Thread.currentThread().interrupt();
453             throw new RestException(e.getMessage(), e);
454         }
455     }

457     private long get429DelayMillis(HttpResponse httpResponse) {
458 
459         // the time at which the rate limit will reset, specified in UTC epoch time.
460         String resetLimit = getOnlySingleHeaderValue(httpResponse,"X-Rate-Limit-Reset");
461         if (resetLimit == null || !resetLimit.chars().allMatch(Character::isDigit)) {
462             return -1;
463         }
464 
465         // If the Date header is not set, do not continue
466         Date requestDate = dateFromHeader(httpResponse);
467         if (requestDate == null) {
468             return -1;
469         }
470 
471         long waitUntil = Long.parseLong(resetLimit) * 1000L;
472         long requestTime = requestDate.getTime();
473         long delay = waitUntil - requestTime + 1000;
474         log.debug("429 wait: {} - {} + {} = {}", waitUntil, requestTime, 1000, delay);
475 
476         return delay;
477     }

479     private Date dateFromHeader(HttpResponse httpResponse) {
480         Date result = null;
481         Header dateHeader = httpResponse.getFirstHeader("Date");
482         if (dateHeader != null) {
483             result = DateUtils.parseDate(dateHeader.getValue());
484         }
485         return result;
486     }

488     private long getDefaultDelayMillis(int retries) {
489         long scaleFactor = 300;
490         long result = (long) (Math.pow(2, retries) * scaleFactor);
491         return Math.min(result, DEFAULT_MAX_BACKOFF_IN_MILLISECONDS);
492     }

494     /**
495      * Returns true if a failed request should be retried.
496      *
497      * @param method  The current HTTP method being executed.
498      * @param t       The throwable from the failed request.
499      * @param retryCount  The number of times the current request has been attempted.
500      * @param timeElapsed The time elapsed for this attempt.
501      * @return True if the failed request should be retried.
502      */
503     private boolean shouldRetry(HttpRequestBase method, Throwable t, int retryCount, long timeElapsed) {
504         if (!shouldRetry(retryCount, timeElapsed)) {
505             return false;
506         }
507 
508         if (method instanceof HttpEntityEnclosingRequest) {
509             HttpEntity entity = ((HttpEntityEnclosingRequest) method).getEntity();
510             if (entity != null && !entity.isRepeatable()) {
511                 return false;
512             }
513         }
514 
515         if (t instanceof NoHttpResponseException ||
516                 t instanceof SocketException ||
517                 t instanceof SocketTimeoutException ||
518                 t instanceof ConnectTimeoutException) {
519             log.debug("Retrying on {}: {}", t.getClass().getName(), t.getMessage());
520             return true;
521         }
522 
523         return false;
524     }
525 
526     private boolean shouldRetry(int retryCount, long timeElapsed) {
527                // either maxRetries or maxElapsedMillis is enabled
528         return (maxRetries > 0 || maxElapsedMillis > 0)
529 
530                // maxRetries count is disabled OR if set check it
531                && (maxRetries <= 0 || retryCount <= this.maxRetries)
532 
533                // maxElapsedMillis is disabled OR if set check it
534                && (maxElapsedMillis <= 0 || timeElapsed < maxElapsedMillis);
535     }
536 
537     private boolean shouldRetry(Response response, int retryCount, long timeElapsed) {
538         int httpStatus = response.getHttpStatus();
539 
540         // supported status codes
541         return shouldRetry(retryCount, timeElapsed)
542             && (httpStatus == 429
543              || httpStatus == 503
544              || httpStatus == 504);
545     }

551     protected Response toSdkResponse(HttpResponse httpResponse) throws IOException {
552 
553         int httpStatus = httpResponse.getStatusLine().getStatusCode();
554 
555         HttpHeaders headers = getHeaders(httpResponse);
556         MediaType mediaType = headers.getContentType();
557 
558         HttpEntity entity = getHttpEntity(httpResponse);
559 
560         InputStream body = entity != null ? entity.getContent() : null;
561         long contentLength = entity != null ? entity.getContentLength() : -1;
562 
563         //ensure that the content has been fully acquired before closing the http stream
564         if (body != null) {
565             byte[] bytes = toBytes(entity);
566 
567             if(bytes != null) {
568                 body = new ByteArrayInputStream(bytes);
569             }  else {
570                 body = null;
571             }
572         }
573 
574         Response response = new DefaultResponse(httpStatus, mediaType, body, contentLength);
575 
576         response.getHeaders().add(HttpHeaders.OKTA_REQUEST_ID, headers.getOktaRequestId());
577         response.getHeaders().put(HttpHeaders.LINK, headers.getLinkHeaders());
578 
579         return response;
580     }

582     private HttpEntity getHttpEntity(HttpResponse response) {
583 
584         HttpEntity entity = response.getEntity();
585         if (entity != null) {
586             Header contentEncodingHeader = entity.getContentEncoding();
587             if (contentEncodingHeader != null) {
588                 for (HeaderElement element : contentEncodingHeader.getElements()) {
589                     if (element.getName().equalsIgnoreCase("gzip")) {
590                         return new GzipDecompressingEntity(response.getEntity());
591                     }
592                 }
593             }
594         }
595         return entity;
596     }
597 
598     private HttpHeaders getHeaders(HttpResponse response) {
599 
600         HttpHeaders headers = new HttpHeaders();
601         Header[] httpHeaders = response.getAllHeaders();
602 
603         if (httpHeaders != null) {
604             for (Header httpHeader : httpHeaders) {
605                 headers.add(httpHeader.getName(), httpHeader.getValue());
606             }
607         }
608 
609         return headers;
610     }
