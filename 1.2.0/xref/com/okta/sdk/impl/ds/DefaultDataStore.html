98      public DefaultDataStore(RequestExecutor requestExecutor, BaseUrlResolver baseUrlResolver, ClientCredentialsResolver clientCredentialsResolver, CacheManager cacheManager) {
99          Assert.notNull(baseUrlResolver, "baseUrlResolver cannot be null");
100         Assert.notNull(requestExecutor, "RequestExecutor cannot be null.");
101         Assert.notNull(clientCredentialsResolver, "clientCredentialsResolver cannot be null.");
102         Assert.notNull(cacheManager, "CacheManager cannot be null.  Use the DisabledCacheManager if you wish to turn off caching.");
103         this.requestExecutor = requestExecutor;
104         this.baseUrlResolver = baseUrlResolver;
105         this.cacheManager = cacheManager;
106         this.resourceFactory = new DefaultResourceFactory(this);
107         this.mapMarshaller = new JacksonMapMarshaller();
108         CacheResolver cacheResolver = new DefaultCacheResolver(this.cacheManager, new DefaultCacheRegionNameResolver());
109         this.clientCredentialsResolver = clientCredentialsResolver;
110 
111         ReferenceFactory referenceFactory = new ReferenceFactory();
112         this.resourceConverter = new DefaultResourceConverter(referenceFactory);
113 
114         this.filters = new ArrayList<>();
115 
116         if (isCachingEnabled()) {
117             ResourceCacheStrategy cacheStrategy = new DefaultResourceCacheStrategy(new HalResourceHrefResolver(), cacheResolver);
118             this.filters.add(new ReadCacheFilter(cacheStrategy));
119             this.filters.add(new WriteCacheFilter(cacheStrategy));
120         }
121     }

147     private <T extends Resource> T instantiate(Class<T> clazz, Map<String, ?> properties, QueryString qs) {
148         if (CollectionResource.class.isAssignableFrom(clazz)) {
149             //only collections can support a query string constructor argument:
150             return this.resourceFactory.instantiate(clazz, properties, qs);
151         }
152         //otherwise it must be an instance resource, so use the two-arg constructor:
153         return this.resourceFactory.instantiate(clazz, properties);
154     }

166     public <T extends Resource> T getResource(String href, Class<T> clazz, Map<String, Object> queryParameters) {
167         ResourceDataResult result = getResourceData(href, clazz, queryParameters);
168         T resource = instantiate(clazz, result.getData(), result.getUri().getQuery());
169 
170         resource.setResourceHref(href);
171 
172         return resource;
173 
174     }

176     @SuppressWarnings("unchecked")
177     private ResourceDataResult getResourceData(String href, Class<? extends Resource> clazz, Map<String,?> queryParameters) {
178 
179         Assert.hasText(href, "href argument cannot be null or empty.");
180         Assert.notNull(clazz, "Resource class argument cannot be null.");
181 
182         FilterChain chain = new DefaultFilterChain(this.filters, req -> {
183 
184             CanonicalUri uri = req.getUri();
185 
186             Request getRequest = new DefaultRequest(HttpMethod.GET, uri.getAbsolutePath(), uri.getQuery());
187             Response getResponse = execute(getRequest);
188             Map<String,?> body = getBody(getResponse);
189 
190             if (Collections.isEmpty(body)) {
191                 throw new IllegalStateException("Unable to obtain resource data from the API server or from cache.");
192             }
193 
194             return new DefaultResourceDataResult(req.getAction(), uri, req.getResourceClass(), (Map<String,Object>)body);
195         });
196 
197         CanonicalUri uri = canonicalize(href, queryParameters);
198         ResourceDataRequest req = new DefaultResourceDataRequest(ResourceAction.READ, uri, clazz, new HashMap<>());
199         return chain.filter(req);
200     }

202     private ResourceAction getPostAction(ResourceDataRequest request, Response response) {
203         int httpStatus = response.getHttpStatus();
204         if (httpStatus == 201) {
205             return ResourceAction.CREATE;
206         }
207         // TODO: verify this is an issue with Okta
208         //Fix for https://github.com/stormpath/stormpath-sdk-java/issues/403
209         if (httpStatus == 200) {
210             return ResourceAction.READ;
211         }
212 
213         return request.getAction();
214     }

252     @SuppressWarnings("unchecked")
253     private <T extends Resource, R extends Resource> R save(String href,
254                                                             final T resource,
255                                                             final T parentResource,
256                                                             HttpHeaders requestHeaders,
257                                                             final Class<? extends R> returnType,
258                                                             final QueryString qs,
259                                                             final boolean create) {
260         Assert.hasText(href, "href argument cannot be null or empty.");
261         Assert.notNull(resource, "resource argument cannot be null.");
262         Assert.notNull(returnType, "returnType class cannot be null.");
263         Assert.isInstanceOf(AbstractResource.class, resource);
264         Assert.isTrue(!CollectionResource.class.isAssignableFrom(resource.getClass()), "Collections cannot be persisted.");
265 
266         final CanonicalUri uri = canonicalize(href, qs);
267         final AbstractResource abstractResource = (AbstractResource) resource;
268         // Most Okta endpoints do not support partial update, we can revisit in the future.
269         final Map<String, Object> props = resourceConverter.convert(abstractResource, false);
270 
271         FilterChain chain = new DefaultFilterChain(this.filters, req -> {
272 
273             CanonicalUri uri1 = req.getUri();
274             String href1 = uri1.getAbsolutePath();
275             QueryString qs1 = uri1.getQuery();
276 
277             HttpHeaders httpHeaders = req.getHttpHeaders();
278 
279             // create == POST
280             HttpMethod method = HttpMethod.POST;
281             if (!create) {
282                 method = HttpMethod.PUT;
283             }
284 
285             InputStream body;
286             long length = 0;
287             if (resource instanceof VoidResource) {
288                 body = new ByteArrayInputStream(new byte[0]);
289             } else {
290                 ByteArrayOutputStream bodyOut = new ByteArrayOutputStream();
291                 mapMarshaller.marshal(bodyOut, req.getData());
292                 body = new ByteArrayInputStream(bodyOut.toByteArray());
293                 length = bodyOut.size();
294             }
295 
296             Request request = new DefaultRequest(method, href1, qs1, httpHeaders, body, length);
297 
298             Response response = execute(request);
299             Map<String, Object> responseBody = getBody(response);
300 
301             if (Collections.isEmpty(responseBody)) {
302                 // Fix for https://github.com/stormpath/stormpath-sdk-java/issues/218
303                 // Okta response with 200 for deactivate requests (i.e. /api/v1/apps/<id>/lifecycle/deactivate)
304                 if (response.getHttpStatus() == 202
305                         || response.getHttpStatus() == 200
306                         || response.getHttpStatus() == 204) {
307                     //202 means that the request has been accepted for processing, but the processing has not been completed. Therefore we do not have a response setBody.
308                     responseBody = java.util.Collections.emptyMap();
309                 } else {
310                     throw new IllegalStateException("Unable to obtain resource data from the API server.");
311                 }
312             }
313 
314             ResourceAction responseAction = getPostAction(req, response);
315 
316             return new DefaultResourceDataResult(responseAction, uri1, returnType, responseBody);
317         });
318 
319         ResourceAction action = create ? ResourceAction.CREATE : ResourceAction.UPDATE;
320 
321         ResourceDataRequest request = new DefaultResourceDataRequest(
322                 action,
323                 uri,
324                 canonicalizeParent(parentResource),
325                 returnType,
326                 getResourceClass(parentResource),
327                 props,
328                 requestHeaders);
329 
330         ResourceDataResult result = chain.filter(request);
331 
332         Map<String,Object> data = result.getData();
333 
334         //ensure the caller's argument is updated with what is returned from the server if the types are the same:
335         if (returnType.isAssignableFrom(abstractResource.getClass())) {
336             abstractResource.setInternalProperties(data);
337         }
338 
339         return resourceFactory.instantiate(returnType, data);
340     }

366     private void doDelete(String resourceHref, Class resourceClass, QueryString qs) {
367 
368         Assert.hasText(resourceHref, "This resource does not have an href value, therefore it cannot be deleted.");
369 
370         // if this URL is a partial, then we MUST add the baseUrl
371         final CanonicalUri uri = canonicalize(resourceHref, qs);
372 
373         FilterChain chain = new DefaultFilterChain(this.filters, request -> {
374             Request deleteRequest = new DefaultRequest(HttpMethod.DELETE, uri.getAbsolutePath(), uri.getQuery());
375             execute(deleteRequest);
376             //delete requests have HTTP 204 (no content), so just create an empty setBody for the result:
377             return new DefaultResourceDataResult(request.getAction(), request.getUri(), request.getResourceClass(), new HashMap<>());
378         });
379 
380         final CanonicalUri resourceUri = canonicalize(resourceHref, null);
381         ResourceDataRequest request = new DefaultResourceDataRequest(ResourceAction.DELETE, resourceUri, resourceClass, new HashMap<>());
382         chain.filter(request);
383     }

385     private <T extends Resource> void doDelete(String href, T resource) {
386 
387         Assert.notNull(resource, "resource argument cannot be null.");
388         Assert.isInstanceOf(AbstractResource.class, resource, "Resource argument must be an AbstractResource.");
389 
390         doDelete(href != null ? href : resource.getResourceHref(), getResourceClass(resource), null);
391     }

410     private Response execute(Request request) throws ResourceException {
411 
412         applyDefaultRequestHeaders(request);
413 
414         Response response = this.requestExecutor.executeRequest(request);
415         log.trace("Executed HTTP request.");
416 
417         if (requestLog.isTraceEnabled()) {
418             requestLog.trace("Executing request: method: '{}', url: {}", request.getMethod(), request.getResourceUrl());
419         }
420 
421         if (response.isError()) {
422             Map<String, Object> body = getBody(response);
423 
424             String requestId = response.getHeaders().getOktaRequestId();
425 
426             if (Strings.hasText(requestId)) {
427                 body.put(DefaultError.ERROR_ID.getName(), requestId);
428             }
429 
430             throw new ResourceException(new DefaultError(body)
431                                             .setHeaders(response.getHeaders().getXHeaders())
432                                             .setStatus(response.getHttpStatus()));
433         }
434 
435         return response;
436     }

438     private Map<String, Object> getBody(Response response) {
439 
440         Assert.notNull(response, "response argument cannot be null.");
441 
442         Map<String, Object> out = null;
443 
444         if (response.hasBody()) {
445             out = mapMarshaller.unmarshal(response.getBody(), response.getHeaders().getLinkMap());
446         }
447 
448         return out;
449     }

451     protected void applyDefaultRequestHeaders(Request request) {
452         request.getHeaders().setAccept(java.util.Collections.singletonList(MediaType.APPLICATION_JSON));
453 
454         // Get runtime headers from http client
455         Map<String, List<String>> headerMap = HttpHeadersHolder.get();
456         String oktaAgentHeaderName = OKTA_AGENT.toLowerCase(Locale.ENGLISH);
457         if (headerMap != null && headerMap.get(oktaAgentHeaderName) != null) {
458             List<String> oktaAgents = headerMap.get(oktaAgentHeaderName);
459             if (oktaAgents != null && !oktaAgents.isEmpty()) {
460                 String oktaAgent = Strings.arrayToDelimitedString(oktaAgents.toArray(), " ");
461                 request.getHeaders().set("User-Agent", oktaAgent + " " + USER_AGENT_STRING);
462             }
463         } else {
464             request.getHeaders().set("User-Agent", USER_AGENT_STRING);
465         }
466         if (request.getHeaders().getContentType() == null && request.getBody() != null) {
467             // We only add the default content type (application/json) if a content type is not already in the request
468             request.getHeaders().setContentType(MediaType.APPLICATION_JSON);
469         }
470 
471         List<String> clientRequestId;
472         if (headerMap != null && (clientRequestId = headerMap.get(OKTA_CLIENT_REQUEST_ID)) != null) {
473             request.getHeaders().put(OKTA_CLIENT_REQUEST_ID, clientRequestId);
474         }
475     }

477     private CanonicalUri canonicalize(String href, Map<String,?> queryParams) {
478         href = ensureFullyQualified(href);
479         return DefaultCanonicalUri.create(href, queryParams);
480     }
481 
482     private CanonicalUri canonicalizeParent(Resource parentResource) {
483         if (parentResource != null && parentResource.getResourceHref() != null) {
484             String href = ensureFullyQualified(parentResource.getResourceHref());
485             return DefaultCanonicalUri.create(href, null);
486         }
487         return null;
488     }
489 
490     private String ensureFullyQualified(String href) {
491         String value = href;
492         if (!isFullyQualified(href)) {
493             value = qualify(href);
494         }
495         return value;
496     }
497 
498     private boolean isFullyQualified(String href) {
499 
500         if (href == null || href.length() < 5) {
501             return false;
502         }
503         return href.regionMatches(true, 0, "http", 0, 4);
504     }
505 
506     private String qualify(String href) {
507         StringBuilder sb = new StringBuilder(this.baseUrlResolver.getBaseUrl());
508         if (!href.startsWith("/")) {
509             sb.append("/");
510         }
511         sb.append(href);
512         return sb.toString();
513     }

515     private Class<? extends Resource> getResourceClass(Resource resource) {
516         if (AbstractInstanceResource.class.isInstance(resource)) {
517             return ((AbstractInstanceResource)resource).getResourceClass();
518         }
519         return resource != null ? resource.getClass() : null;
520     }
