17  package com.okta.sdk.impl.client;
18  
19  import com.okta.sdk.cache.CacheManager;
20  import com.okta.sdk.client.AuthenticationScheme;
21  import com.okta.sdk.client.Client;
22  import com.okta.sdk.client.Proxy;
23  import com.okta.sdk.impl.api.ClientCredentialsResolver;
24  import com.okta.sdk.impl.ds.InternalDataStore;
25  import com.okta.sdk.impl.http.authc.RequestAuthenticatorFactory;
26  import com.okta.sdk.impl.util.BaseUrlResolver;
27  
28  /**
29   * Abstract client implementation, use BaseClient instead. Kept only for backwards compatibility, this class will
30   * be removed in future versions.
31   *
32   * @deprecated see {@link BaseClient}
33   * @see <a href="http://www.okta.com/docs/quickstart/connect">Communicating with Okta: Get your API Key</a>
34   * @since 0.5.0
35   */
36  @Deprecated
37  public abstract class AbstractClient extends BaseClient implements Client {
38  
39      /**
40       * Instantiates a new Client instance that will communicate with the Okta REST API.  See the class-level
41       * JavaDoc for a usage example.
42       *
43       * @param clientCredentialsResolver       Okta API Key resolver
44       * @param baseUrlResolver      Okta base URL resolver
45       * @param proxy                the HTTP proxy to be used when communicating with the Okta API server (can be
46       *                             null)
47       * @param cacheManager         the {@link com.okta.sdk.cache.CacheManager} that should be used to cache
48       *                             Okta REST resources (can be null)
49       * @param authenticationScheme the HTTP authentication scheme to be used when communicating with the Okta API
50       *                             server (can be null)
51       * @param requestAuthenticatorFactory factory used to handle creating autentication requests
52       * @param connectionTimeout    connection timeout in seconds
53       */
54      public AbstractClient(ClientCredentialsResolver clientCredentialsResolver, BaseUrlResolver baseUrlResolver, Proxy proxy, CacheManager cacheManager, AuthenticationScheme authenticationScheme, RequestAuthenticatorFactory requestAuthenticatorFactory, int connectionTimeout) {
55          super(clientCredentialsResolver, baseUrlResolver, proxy, cacheManager, authenticationScheme, requestAuthenticatorFactory, connectionTimeout);
56      }
57  
58      @Override
59      @SuppressWarnings("PMD.UselessOverridingMethod")
60      public InternalDataStore getDataStore() {
61          return super.getDataStore();
62      }
63  }
