36  /**
37   * The base client class.
38   * @since 1.1.0
39   */
40  public abstract class BaseClient implements DataStore {
41  
42      private final InternalDataStore dataStore;
43  
44      /**
45       * Instantiates a new Client instance that will communicate with the Okta REST API.  See the class-level
46       * JavaDoc for a usage example.
47       *
48       * @param clientCredentialsResolver       Okta API Key resolver
49       * @param baseUrlResolver      Okta base URL resolver
50       * @param proxy                the HTTP proxy to be used when communicating with the Okta API server (can be
51       *                             null)
52       * @param cacheManager         the {@link CacheManager} that should be used to cache
53       *                             Okta REST resources (can be null)
54       * @param authenticationScheme the HTTP authentication scheme to be used when communicating with the Okta API
55       *                             server (can be null)
56       * @param requestAuthenticatorFactory factory used to handle creating autentication requests
57       * @param connectionTimeout    connection timeout in seconds
58       */
59      public BaseClient(ClientCredentialsResolver clientCredentialsResolver, BaseUrlResolver baseUrlResolver, Proxy proxy, CacheManager cacheManager, AuthenticationScheme authenticationScheme, RequestAuthenticatorFactory requestAuthenticatorFactory, int connectionTimeout) {
60          Assert.notNull(clientCredentialsResolver, "clientCredentialsResolver argument cannot be null.");
61          Assert.notNull(baseUrlResolver, "baseUrlResolver argument cannot be null.");
62          Assert.isTrue(connectionTimeout >= 0, "connectionTimeout cannot be a negative number.");
63          RequestExecutor requestExecutor = createRequestExecutor(clientCredentialsResolver.getClientCredentials(), proxy, authenticationScheme, requestAuthenticatorFactory, connectionTimeout);
64          this.dataStore = createDataStore(requestExecutor, baseUrlResolver, clientCredentialsResolver, cacheManager);
65      }
66  
67  
68      protected InternalDataStore createDataStore(RequestExecutor requestExecutor, BaseUrlResolver baseUrlResolver, ClientCredentialsResolver clientCredentialsResolver, CacheManager cacheManager) {
69          return new DefaultDataStore(requestExecutor, baseUrlResolver, clientCredentialsResolver, cacheManager);
70      }
71  
72      @Override
73      public ClientCredentials getClientCredentials() {
74          return this.dataStore.getClientCredentials();
75      }
76  
77      @Override
78      public CacheManager getCacheManager() {
79          return this.dataStore.getCacheManager();
80      }
81  
82      public InternalDataStore getDataStore() {
83          return this.dataStore;
84      }
85  
86      @SuppressWarnings({"unchecked", "rawtypes"})
87      protected RequestExecutor createRequestExecutor(ClientCredentials clientCredentials, Proxy proxy, AuthenticationScheme authenticationScheme, RequestAuthenticatorFactory requestAuthenticatorFactory, int connectionTimeout) {
88  
89          String className = "com.okta.sdk.impl.http.httpclient.HttpClientRequestExecutor";
90  
91          Class requestExecutorClass;
92  
93          if (Classes.isAvailable(className)) {
94              requestExecutorClass = Classes.forName(className);
95          } else {
96              //we might be able to check for other implementations in the future, but for now, we only support
97              //HTTP calls via the HttpClient.  Throw an exception:
98  
99              String msg = "Unable to find the '" + className + "' implementation on the classpath.  Please ensure you " +
100                     "have added the okta-sdk-httpclient .jar file to your runtime classpath.";
101             throw new RuntimeException(msg);
102         }
103 
104         Constructor<RequestExecutor> ctor = Classes.getConstructor(requestExecutorClass, ClientCredentials.class, Proxy.class, AuthenticationScheme.class, RequestAuthenticatorFactory.class, Integer.class);
105 
106         return Classes.instantiate(ctor, clientCredentials, proxy, authenticationScheme, requestAuthenticatorFactory, connectionTimeout);
107     }
108 
109     // ========================================================================
110     // DataStore methods (delegate to underlying DataStore instance)
111     // ========================================================================
112 
113     /**
114      * Delegates to the internal {@code dataStore} instance. This is a convenience mechanism to eliminate the constant
115      * need to call {@code client.getDataStore()} every time one needs to instantiate Resource.
116      *
117      * @param clazz the Resource class to instantiate.
118      * @param <T>   the Resource sub-type
119      * @return a new instance of the specified Resource.
120      */
121     @Override
122     public <T extends Resource> T instantiate(Class<T> clazz) {
123         return this.dataStore.instantiate(clazz);
124     }
125 
126     /**
127      * Delegates to the internal {@code dataStore} instance. This is a convenience mechanism to eliminate the constant
128      * need to call {@code client.getDataStore()} every time one needs to look up a Resource.
129      *
130      * @param href  the resource URL of the resource to retrieve
131      * @param clazz the {@link Resource} sub-interface to instantiate
132      * @param <T>   type parameter indicating the returned value is a {@link Resource} instance.
133      * @return an instance of the specified class based on the data returned from the specified {@code href} URL.
134      */
135     @Override
136     public <T extends Resource> T getResource(String href, Class<T> clazz) {
137         return this.dataStore.getResource(href, clazz);
138     }
139 
140     @Override
141     public <T extends Resource> T create(String parentHref, T resource) {
142         return this.dataStore.create(parentHref, resource);
143     }
144 
145     @Override
146     public <T extends Resource> void save(String href, T resource) {
147         this.dataStore.save(href, resource);
148     }
149 
150     @Override
151     public <T extends Resource> void delete(String href, T resource) {
152         this.dataStore.delete(href, resource);
153     }
154 }
