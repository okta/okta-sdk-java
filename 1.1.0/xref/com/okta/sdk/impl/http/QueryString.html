17  package com.okta.sdk.impl.http;
18  
19  import com.okta.sdk.impl.util.RequestUtils;
20  import com.okta.sdk.lang.Collections;
21  import com.okta.sdk.lang.Strings;
22  
23  import java.io.UnsupportedEncodingException;
24  import java.net.URLDecoder;
25  import java.util.Map;
26  import java.util.TreeMap;
27  
28  /**
29   * @since 0.5.0
30   */
31  public class QueryString extends TreeMap<String,String> {
32  
33      private static final long serialVersionUID = 42L;
34  
35      public QueryString(){}
36  
37      public QueryString(Map<String,?> source) {
38          super();
39          if (!Collections.isEmpty(source)) {
40              for(Map.Entry<String,?> entry : source.entrySet()) {
41                  String key = entry.getKey();
42                  Object value = entry.getValue();
43                  String sValue = value != null ? String.valueOf(value) : null;
44                  put(key, sValue);
45              }
46          }
47      }
48  
49      public String put(String key, Object value) {
50          if (value != null) {
51              return super.put(key, value.toString());
52          }
53          return null;
54      }
55  
56      public String toString() {
57          return toString(false);
58      }
59  
60      /**
61       * The canonicalized query string is formed by first sorting all the query
62       * string parameters, then URI encoding both the key and value and then
63       * joining them, in order, separating key value pairs with an '&amp;'.
64       *
65       * @param canonical whether or not the string should be canonicalized
66       * @return the canonical query string
67       */
68      public String toString(boolean canonical) {
69          if (isEmpty()) {
70              return "";
71          }
72          StringBuilder sb = new StringBuilder();
73  
74          for (Map.Entry<String,String> entry : entrySet()) {
75              String key = RequestUtils.encodeUrl(entry.getKey(), false, canonical);
76              String value = RequestUtils.encodeUrl(entry.getValue(), false, canonical);
77  
78              if (sb.length() > 0) {
79                  sb.append('&');
80              }
81  
82              sb.append(key).append("=").append(value);
83          }
84  
85          return sb.toString();
86      }
87  
88      public static QueryString create(String query) {
89          if (!Strings.hasLength(query)) {
90              return null;
91          }
92  
93          QueryString queryString = new QueryString();
94  
95          // only returns null if string is null
96          String[] tokens = Strings.tokenizeToStringArray(query, "&", false, false);
97          for( String token : tokens) {
98              applyKeyValuePair(queryString, token);
99          }
100 
101         return queryString;
102     }
103 
104     private static void applyKeyValuePair(QueryString qs, String kv) {
105 
106         String[] pair = Strings.split(kv, "=");
107 
108         if (pair != null) {
109             String key = pair[0];
110             String value = pair[1] != null ? pair[1] : "";
111             try {
112                 qs.put(URLDecoder.decode(key, "UTF-8"), URLDecoder.decode(value, "UTF-8"));
113             } catch (UnsupportedEncodingException e) {
114                 // should never happen
115                 qs.put(key, value);
116             }
117         } else {
118             //no equals sign, it's just a key:
119             qs.put(kv, null);
120         }
121     }
122 
123     /**
124      * Build an href with query string. Only appends it queryArgs is NOT empty.
125      * @param href URL path
126      * @param qs query string to append to href
127      * @return href + query string if query string is NOT empty, otherwise, just returns the href
128      */
129     public static String buildHref(String href, QueryString qs) {
130         StringBuilder sb = new StringBuilder(href);
131         String query = qs.toString();
132         if (!Strings.isEmpty(query)) {
133             sb.append('?').append(query);
134         }
135         return sb.toString();
136     }
