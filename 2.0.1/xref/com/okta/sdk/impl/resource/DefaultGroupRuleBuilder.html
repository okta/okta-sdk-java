16  package com.okta.sdk.impl.resource;
17  
18  import com.okta.commons.lang.Collections;
19  import com.okta.commons.lang.Strings;
20  import com.okta.sdk.client.Client;
21  import com.okta.sdk.resource.group.rule.*;
22  
23  import java.util.ArrayList;
24  import java.util.List;
25  
26  public class DefaultGroupRuleBuilder implements GroupRuleBuilder {
27  
28      private String name;
29      private String type;
30      private List<String> assignUserToGroups = new ArrayList<>();
31      private List<String> groupIds = new ArrayList<>();
32      private List<String> userIds = new ArrayList<>();
33      private String groupRuleExpressionType;
34      private String groupRuleExpressionValue;
35  
36      @Override
37      public GroupRuleBuilder setName(String name) {
38          this.name = name;
39          return this;
40      }
41  
42      @Override
43      public GroupRuleBuilder setType(String type) {
44          this.type = type;
45          return this;
46      }
47  
48      @Override
49      public GroupRuleBuilder setAssignUserToGroups(List<String> assignUserToGroups) {
50          this.assignUserToGroups = assignUserToGroups;
51          return this;
52      }
53  
54      @Override
55      public GroupRuleBuilder setGroups(List<String> groupIds) {
56          this.groupIds = groupIds;
57          return this;
58      }
59  
60      @Override
61      public GroupRuleBuilder addGroup(String groupId) {
62          this.groupIds.add(groupId);
63          return this;
64      }
65  
66      @Override
67      public GroupRuleBuilder setUsers(List<String> userIds) {
68          this.userIds = userIds;
69          return this;
70      }
71  
72      @Override
73      public GroupRuleBuilder addUser(String userId) {
74          this.userIds.add(userId);
75          return this;
76      }
77  
78      @Override
79      public GroupRuleBuilder setGroupRuleExpressionType(String groupRuleExpressionType) {
80          this.groupRuleExpressionType = groupRuleExpressionType;
81          return this;
82      }
83  
84      @Override
85      public GroupRuleBuilder setGroupRuleExpressionValue(String groupRuleExpressionValue) {
86          this.groupRuleExpressionValue = groupRuleExpressionValue;
87          return this;
88      }
89  
90      @Override
91      public GroupRule buildAndCreate(Client client) {
92          return client.createGroupRule(build(client));
93      }
94  
95      private GroupRule build(Client client){
96          GroupRule groupRule = client.instantiate(GroupRule.class);
97  
98          if (Strings.hasText(name)) groupRule.setName(name);
99  
100         if (Strings.hasText(type)) groupRule.setType(type);
101 
102         // Actions
103         groupRule.setActions(client.instantiate(GroupRuleAction.class));
104         GroupRuleAction groupRuleActions = groupRule.getActions();
105 
106         if (!Collections.isEmpty(assignUserToGroups))
107             groupRuleActions.setAssignUserToGroups(client.instantiate(GroupRuleGroupAssignment.class)
108                 .setGroupIds(assignUserToGroups));
109 
110         // Conditions
111         groupRule.setConditions(client.instantiate(GroupRuleConditions.class));
112         GroupRuleConditions groupRuleConditions = groupRule.getConditions();
113         GroupRulePeopleCondition groupRulePeopleCondition = client.instantiate(GroupRulePeopleCondition.class);
114 
115         if (!Collections.isEmpty(groupIds))
116             groupRuleConditions.setPeople(groupRulePeopleCondition
117                 .setGroups(client.instantiate(GroupRuleGroupCondition.class)
118                     .setInclude(groupIds)));
119 
120         if (!Collections.isEmpty(userIds))
121             groupRuleConditions.setPeople(groupRulePeopleCondition
122                 .setUsers(client.instantiate(GroupRuleUserCondition.class)
123                     .setInclude(userIds)));
124 
125         GroupRuleExpression groupRuleExpression = client.instantiate(GroupRuleExpression.class);
126 
127         if (Strings.hasText(groupRuleExpressionType))
128             groupRuleConditions.setExpression(groupRuleExpression.setType(groupRuleExpressionType));
129 
130         if (Strings.hasText(groupRuleExpressionValue))
131             groupRuleConditions.setExpression(groupRuleExpression.setValue(groupRuleExpressionValue));
132 
133         return groupRule;
134     }
135 }
