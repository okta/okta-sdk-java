17  package com.okta.sdk.impl.client;
18  
19  import com.okta.commons.http.RequestExecutor;
20  import com.okta.commons.http.RequestExecutorFactory;
21  import com.okta.commons.http.config.BaseUrlResolver;
22  import com.okta.commons.lang.Assert;
23  import com.okta.commons.lang.Classes;
24  import com.okta.sdk.authc.credentials.ClientCredentials;
25  import com.okta.sdk.cache.CacheManager;
26  import com.okta.sdk.ds.DataStore;
27  import com.okta.sdk.ds.RequestBuilder;
28  import com.okta.sdk.impl.api.ClientCredentialsResolver;
29  import com.okta.sdk.impl.config.ClientConfiguration;
30  import com.okta.sdk.impl.ds.DefaultDataStore;
31  import com.okta.sdk.impl.ds.InternalDataStore;
32  import com.okta.sdk.resource.Resource;
33  
34  /**
35   * The base client class.
36   * @since 1.1.0
37   */
38  public abstract class BaseClient implements DataStore {
39  
40      private final InternalDataStore dataStore;
41  
42      public BaseClient(ClientConfiguration clientConfiguration, CacheManager cacheManager) {
43          Assert.notNull(clientConfiguration, "clientConfiguration argument cannot be null.");
44          RequestExecutor requestExecutor = createRequestExecutor(clientConfiguration);
45          this.dataStore = createDataStore(requestExecutor,
46                                           clientConfiguration.getBaseUrlResolver(),
47                                           clientConfiguration.getClientCredentialsResolver(),
48                                           cacheManager);
49      }
50  
51      protected InternalDataStore createDataStore(RequestExecutor requestExecutor, BaseUrlResolver baseUrlResolver, ClientCredentialsResolver clientCredentialsResolver, CacheManager cacheManager) {
52          return new DefaultDataStore(requestExecutor, baseUrlResolver, clientCredentialsResolver, cacheManager);
53      }
54  
55      @Override
56      public ClientCredentials getClientCredentials() {
57          return this.dataStore.getClientCredentials();
58      }
59  
60      @Override
61      public CacheManager getCacheManager() {
62          return this.dataStore.getCacheManager();
63      }
64  
65      public InternalDataStore getDataStore() {
66          return this.dataStore;
67      }
68  
69      @SuppressWarnings({"unchecked", "rawtypes"})
70      protected RequestExecutor createRequestExecutor(ClientConfiguration clientConfiguration) {
71  
72          String msg = "Unable to find a '" + RequestExecutorFactory.class.getName() + "' " +
73                  "implementation on the classpath.  Please ensure you have added the " +
74                  "okta-sdk-httpclient.jar file to your runtime classpath."; // TODO fix jar name
75          return Classes.loadFromService(RequestExecutorFactory.class, msg).create(clientConfiguration);
76      }
77  
78      // ========================================================================
79      // DataStore methods (delegate to underlying DataStore instance)
80      // ========================================================================
81  
82      /**
83       * Delegates to the internal {@code dataStore} instance. This is a convenience mechanism to eliminate the constant
84       * need to call {@code client.getDataStore()} every time one needs to instantiate Resource.
85       *
86       * @param clazz the Resource class to instantiate.
87       * @param <T>   the Resource sub-type
88       * @return a new instance of the specified Resource.
89       */
90      @Override
91      public <T extends Resource> T instantiate(Class<T> clazz) {
92          return this.dataStore.instantiate(clazz);
93      }
94  
95      /**
96       * Delegates to the internal {@code dataStore} instance. This is a convenience mechanism to eliminate the constant
97       * need to call {@code client.getDataStore()} every time one needs to look up a Resource.
98       *
99       * @param href  the resource URL of the resource to retrieve
100      * @param clazz the {@link Resource} sub-interface to instantiate
101      * @param <T>   type parameter indicating the returned value is a {@link Resource} instance.
102      * @return an instance of the specified class based on the data returned from the specified {@code href} URL.
103      */
104     @Override
105     public <T extends Resource> T getResource(String href, Class<T> clazz) {
106         return this.dataStore.getResource(href, clazz);
107     }
108 
109     @Override
110     public <T extends Resource> T create(String parentHref, T resource) {
111         return this.dataStore.create(parentHref, resource);
112     }
113 
114     @Override
115     public <T extends Resource> void save(String href, T resource) {
116         this.dataStore.save(href, resource);
117     }
118 
119     @Override
120     public <T extends Resource> void delete(String href, T resource) {
121         this.dataStore.delete(href, resource);
122     }
123 
124     @Override
125     public RequestBuilder http() {
126         return this.dataStore.http();
127     }
128 }
