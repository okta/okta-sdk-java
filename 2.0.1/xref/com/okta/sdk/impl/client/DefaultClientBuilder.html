69  /**
70   * <p>The default {@link ClientBuilder} implementation. This looks for configuration files
71   * in the following locations and order of precedence (last one wins).</p>
72   * <ul>
73   * <li>classpath:com/okta/sdk/config/okta.properties</li>
74   * <li>classpath:com/okta/sdk/config/okta.yaml</li>
75   * <li>classpath:okta.properties</li>
76   * <li>classpath:okta.yaml</li>
77   * <li>~/.okta/okta.yaml</li>
78   * <li>Environment Variables (with dot notation converted to uppercase + underscores)</li>
79   * <li>System Properties</li>
80   * <li>Programmatically</li>
81   * </ul>
82   *
83   * @since 0.5.0
84   */

118     DefaultClientBuilder(ResourceFactory resourceFactory) {
119         Collection<PropertiesSource> sources = new ArrayList<>();
120 
121         for (String location : DEFAULT_OKTA_PROPERTIES_FILE_LOCATIONS) {
122 
123             if (ENVVARS_TOKEN.equalsIgnoreCase(location)) {
124                 sources.add(EnvironmentVariablesPropertiesSource.oktaFilteredPropertiesSource());
125             }
126             else if (SYSPROPS_TOKEN.equalsIgnoreCase(location)) {
127                 sources.add(SystemPropertiesSource.oktaFilteredPropertiesSource());
128             }
129             else {
130                 Resource resource = resourceFactory.createResource(location);
131 
132                 PropertiesSource wrappedSource;
133                 if (Strings.endsWithIgnoreCase(location, ".yaml")) {
134                     wrappedSource = new YAMLPropertiesSource(resource);
135                 } else {
136                     wrappedSource = new ResourcePropertiesSource(resource);
137                 }
138 
139                 PropertiesSource propertiesSource = new OptionalPropertiesSource(wrappedSource);
140                 sources.add(propertiesSource);
141             }
142         }
143 
144         Map<String, String> props = new LinkedHashMap<>();
145 
146         for (PropertiesSource source : sources) {
147             Map<String, String> srcProps = source.getProperties();
148             props.putAll(srcProps);
149         }
150 
151         // check to see if property value is null before setting value
152         // if != null, allow it to override previously set values
153         if (Strings.hasText(props.get(DEFAULT_CLIENT_API_TOKEN_PROPERTY_NAME))) {
154             clientConfig.setApiToken(props.get(DEFAULT_CLIENT_API_TOKEN_PROPERTY_NAME));
155         }
156 
157         if (Strings.hasText(props.get(DEFAULT_CLIENT_CACHE_ENABLED_PROPERTY_NAME))) {
158             clientConfig.setCacheManagerEnabled(Boolean.valueOf(props.get(DEFAULT_CLIENT_CACHE_ENABLED_PROPERTY_NAME)));
159         }
160 
161         if (Strings.hasText(props.get(DEFAULT_CLIENT_CACHE_TTL_PROPERTY_NAME))) {
162             clientConfig.setCacheManagerTtl(Long.parseLong(props.get(DEFAULT_CLIENT_CACHE_TTL_PROPERTY_NAME)));
163         }
164 
165         if (Strings.hasText(props.get(DEFAULT_CLIENT_CACHE_TTI_PROPERTY_NAME))) {
166             clientConfig.setCacheManagerTti(Long.parseLong(props.get(DEFAULT_CLIENT_CACHE_TTI_PROPERTY_NAME)));
167         }
168 
169         for (String prop : props.keySet()) {
170             boolean isPrefix = prop.length() == DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME.length();
171             if (!isPrefix && prop.startsWith(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME)) {
172                 // get class from prop name
173                 String cacheClass = prop.substring(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME.length() + 1, prop.length() - 4);
174                 String cacheTti = props.get(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME + "." + cacheClass + ".tti");
175                 String cacheTtl = props.get(DEFAULT_CLIENT_CACHE_CACHES_PROPERTY_NAME + "." + cacheClass + ".ttl");
176                 CacheConfigurationBuilder cacheBuilder = Caches.forResource(Classes.forName(cacheClass));
177                 if (Strings.hasText(cacheTti)) {
178                     cacheBuilder.withTimeToIdle(Long.parseLong(cacheTti), TimeUnit.SECONDS);
179                 }
180                 if (Strings.hasText(cacheTtl)) {
181                     cacheBuilder.withTimeToLive(Long.parseLong(cacheTtl), TimeUnit.SECONDS);
182                 }
183                 if (!clientConfig.getCacheManagerCaches().containsKey(cacheClass)) {
184                     clientConfig.getCacheManagerCaches().put(cacheClass, cacheBuilder);
185                 }
186             }
187         }
188 
189         if (Strings.hasText(props.get(DEFAULT_CLIENT_TESTING_DISABLE_HTTPS_CHECK_PROPERTY_NAME))) {
190             allowNonHttpsForTesting = Boolean.valueOf(props.get(DEFAULT_CLIENT_TESTING_DISABLE_HTTPS_CHECK_PROPERTY_NAME));
191         }
192 
193         if (Strings.hasText(props.get(DEFAULT_CLIENT_ORG_URL_PROPERTY_NAME))) {
194             String baseUrl = props.get(DEFAULT_CLIENT_ORG_URL_PROPERTY_NAME);
195             // remove backslashes that can end up in file when it's written programmatically, e.g. in a test
196             baseUrl = baseUrl.replace("\\:", ":");
197             ConfigurationValidator.assertOrgUrl(baseUrl, allowNonHttpsForTesting);
198             clientConfig.setBaseUrl(baseUrl);
199         }
200 
201         if (Strings.hasText(props.get(DEFAULT_CLIENT_CONNECTION_TIMEOUT_PROPERTY_NAME))) {
202             clientConfig.setConnectionTimeout(Integer.parseInt(props.get(DEFAULT_CLIENT_CONNECTION_TIMEOUT_PROPERTY_NAME)));
203         }
204 
205         if (Strings.hasText(props.get(DEFAULT_CLIENT_AUTHENTICATION_SCHEME_PROPERTY_NAME))) {
206             clientConfig.setAuthenticationScheme(Enum.valueOf(AuthenticationScheme.class, props.get(DEFAULT_CLIENT_AUTHENTICATION_SCHEME_PROPERTY_NAME)));
207         }
208 
209         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_PORT_PROPERTY_NAME))) {
210             clientConfig.setProxyPort(Integer.parseInt(props.get(DEFAULT_CLIENT_PROXY_PORT_PROPERTY_NAME)));
211         }
212 
213         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_HOST_PROPERTY_NAME))) {
214             clientConfig.setProxyHost(props.get(DEFAULT_CLIENT_PROXY_HOST_PROPERTY_NAME));
215         }
216 
217         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_USERNAME_PROPERTY_NAME))) {
218             clientConfig.setProxyUsername(props.get(DEFAULT_CLIENT_PROXY_USERNAME_PROPERTY_NAME));
219         }
220 
221         if (Strings.hasText(props.get(DEFAULT_CLIENT_PROXY_PASSWORD_PROPERTY_NAME))) {
222             clientConfig.setProxyPassword(props.get(DEFAULT_CLIENT_PROXY_PASSWORD_PROPERTY_NAME));
223         }
224 
225         if (Strings.hasText(props.get(DEFAULT_CLIENT_AUTHORIZATION_MODE_PROPERTY_NAME))) {
226             clientConfig.setAuthorizationMode(AuthorizationMode.getAuthorizationMode(props.get(DEFAULT_CLIENT_AUTHORIZATION_MODE_PROPERTY_NAME)));
227         }
228 
229         if (Strings.hasText(props.get(DEFAULT_CLIENT_ID_PROPERTY_NAME))) {
230             clientConfig.setClientId(props.get(DEFAULT_CLIENT_ID_PROPERTY_NAME));
231         }
232 
233         if (Strings.hasText(props.get(DEFAULT_CLIENT_SCOPES_PROPERTY_NAME))) {
234             Set<String> scopes = new HashSet<>(Arrays.asList(props.get(DEFAULT_CLIENT_SCOPES_PROPERTY_NAME).split("[\\s,]+")));
235             clientConfig.setScopes(scopes);
236         }
237 
238         if (Strings.hasText(props.get(DEFAULT_CLIENT_PRIVATE_KEY_PROPERTY_NAME))) {
239             clientConfig.setPrivateKey(props.get(DEFAULT_CLIENT_PRIVATE_KEY_PROPERTY_NAME));
240         }
241 
242         if (Strings.hasText(props.get(DEFAULT_CLIENT_REQUEST_TIMEOUT_PROPERTY_NAME))) {
243             clientConfig.setRetryMaxElapsed(Integer.parseInt(props.get(DEFAULT_CLIENT_REQUEST_TIMEOUT_PROPERTY_NAME)));
244         }
245 
246         if (Strings.hasText(props.get(DEFAULT_CLIENT_RETRY_MAX_ATTEMPTS_PROPERTY_NAME))) {
247             clientConfig.setRetryMaxAttempts(Integer.parseInt(props.get(DEFAULT_CLIENT_RETRY_MAX_ATTEMPTS_PROPERTY_NAME)));
248         }
249     }

313     @Override
314     public Client build() {
315         if (!this.clientConfig.isCacheManagerEnabled()) {
316             log.debug("CacheManager disabled. Defaulting to DisabledCacheManager");
317             this.cacheManager = Caches.newDisabledCacheManager();
318         } else if (this.cacheManager == null) {
319             log.debug("No CacheManager configured. Defaulting to in-memory CacheManager with default TTL and TTI of five minutes.");
320 
321             CacheManagerBuilder cacheManagerBuilder = Caches.newCacheManager()
322                     .withDefaultTimeToIdle(this.clientConfig.getCacheManagerTti(), TimeUnit.SECONDS)
323                     .withDefaultTimeToLive(this.clientConfig.getCacheManagerTtl(), TimeUnit.SECONDS);
324             if (this.clientConfig.getCacheManagerCaches().size() > 0) {
325                 for (CacheConfigurationBuilder builder : this.clientConfig.getCacheManagerCaches().values()) {
326                     cacheManagerBuilder.withCache(builder);
327                 }
328             }
329 
330             this.cacheManager = cacheManagerBuilder.build();
331         }
332 
333         if (this.clientConfig.getBaseUrlResolver() == null) {
334             ConfigurationValidator.assertOrgUrl(this.clientConfig.getBaseUrl(), allowNonHttpsForTesting);
335             this.clientConfig.setBaseUrlResolver(new DefaultBaseUrlResolver(this.clientConfig.getBaseUrl()));
336         }
337 
338         if (!isOAuth2Flow()) {
339             if (this.clientConfig.getClientCredentialsResolver() == null && this.clientCredentials != null) {
340                 this.clientConfig.setClientCredentialsResolver(new DefaultClientCredentialsResolver(this.clientCredentials));
341             } else if (this.clientConfig.getClientCredentialsResolver() == null) {
342                 this.clientConfig.setClientCredentialsResolver(new DefaultClientCredentialsResolver(this.clientConfig));
343             }
344         } else {
345             this.clientConfig.setAuthenticationScheme(AuthenticationScheme.OAUTH2_PRIVATE_KEY);
346 
347             validateOAuth2ClientConfig(this.clientConfig);
348 
349             accessTokenRetrieverService = new AccessTokenRetrieverServiceImpl(clientConfig);
350 
351             OAuth2ClientCredentials oAuth2ClientCredentials =
352                 new OAuth2ClientCredentials(accessTokenRetrieverService);
353 
354             this.clientConfig.setClientCredentialsResolver(new DefaultClientCredentialsResolver(oAuth2ClientCredentials));
355         }
356 
357         return new DefaultClient(clientConfig, cacheManager);
358     }

360     /**
361      * @since 1.6.0
362      */
363     private void validateOAuth2ClientConfig(ClientConfiguration clientConfiguration) {
364         Assert.notNull(clientConfiguration.getClientId(), "clientId cannot be null");
365         Assert.isTrue(clientConfiguration.getScopes() != null && !clientConfiguration.getScopes().isEmpty(),
366             "At least one scope is required");
367         Assert.notNull(clientConfiguration.getPrivateKey(), "privateKey cannot be null");
368         Path privateKeyPemFilePath = Paths.get(clientConfiguration.getPrivateKey());
369         boolean privateKeyPemFileExists = Files.exists(privateKeyPemFilePath, new LinkOption[]{ LinkOption.NOFOLLOW_LINKS });
370         Assert.isTrue(privateKeyPemFileExists, "privateKey file does not exist");
371     }
