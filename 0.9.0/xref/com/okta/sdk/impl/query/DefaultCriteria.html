16  package com.okta.sdk.impl.query;
17  
18  import com.okta.sdk.impl.resource.Property;
19  import com.okta.sdk.lang.Assert;
20  import com.okta.sdk.lang.Strings;
21  import com.okta.sdk.query.Criteria;
22  import com.okta.sdk.query.Criterion;
23  import com.okta.sdk.query.Options;
24  
25  import java.util.ArrayList;
26  import java.util.Collections;
27  import java.util.HashMap;
28  import java.util.List;
29  import java.util.Map;
30  
31  /**
32   * @since 0.5.0
33   */
34  public class DefaultCriteria<T extends Criteria<T>, O extends Options> implements Criteria<T>, Expandable {
35  
36      protected final List<Criterion> criterionEntries;
37      protected final List<Order> orderEntries;
38      protected final O options;
39      protected Integer limit;
40      protected Integer offset;
41      private Map<String, String> customAttributes;
42  
43      protected int currentOrderIndex = -1; //used for order clause building
44  
45      protected DefaultCriteria(O options) {
46          Assert.notNull(options, "options argument cannot be null.");
47          Assert.isInstanceOf(Expandable.class, options, "options argument is expected to implement the " + Expandable.class.getName() + " interface.");
48          this.options = options;
49          this.criterionEntries = new ArrayList<Criterion>();
50          this.orderEntries = new ArrayList<Order>();
51          this.customAttributes = new HashMap<>();
52      }
53  
54      public T add(Criterion criterion) {
55          Assert.notNull(criterion, "criterion cannot be null.");
56          this.criterionEntries.add(criterion);
57          return (T) this;
58      }
59  
60      public T and(Criterion criterion) {
61          return add(criterion);
62      }
63  
64      public T add(Order order) {
65          Assert.notNull(order, "order cannot be null.");
66          this.orderEntries.add(order);
67          this.currentOrderIndex = this.orderEntries.size() - 1;
68          return (T) this;
69      }
70  
71      public T orderBy(Property property) {
72          Order order = Order.asc(property.getName()); //ascending by default - they can call descending() to reverse
73          return add(order);
74      }
75  
76      private int ensureOrderIndex() {
77          int i = this.currentOrderIndex;
78          Assert.state(i >= 0, "There is no current orderBy clause to declare as ascending or descending!");
79          return i;
80      }
81  
82      public List<Expansion> getExpansions() {
83          //we can do this assertion because of the isInstance guarantee in the constructor:
84          assert this.options instanceof Expandable;
85          return ((Expandable)this.options).getExpansions();
86      }
87  
88      public T ascending() {
89          return orderDirection(true);
90      }
91  
92      public T descending() {
93          return orderDirection(false);
94      }
95  
96      private T orderDirection(boolean ascending) {
97          int i = ensureOrderIndex();
98          Order order = this.orderEntries.get(i);
99          if (order.isAscending() != ascending) { //only swap out the Order statement if it is different:
100             String name = order.getPropertyName();
101             Order newOrder = ascending ? Order.asc(name) : Order.desc(name);
102             this.orderEntries.set(i, newOrder);
103         }
104         return (T) this;
105     }
106 
107     public List<Criterion> getCriterionEntries() {
108         return Collections.unmodifiableList(this.criterionEntries);
109     }
110 
111     public List<Order> getOrderEntries() {
112         return Collections.unmodifiableList(this.orderEntries);
113     }
114 
115     protected O getOptions() {
116         return this.options;
117     }
118 
119     public T limitTo(int limit) {
120         this.limit = Pagination.sanitizeLimit(limit);
121         return (T) this;
122     }
123 
124     public Integer getLimit() {
125         return limit;
126     }
127 
128     public T offsetBy(int offset) {
129         this.offset = Pagination.sanitizeOffset(offset);
130         return (T) this;
131     }
132 
133     public Integer getOffset() {
134         return offset;
135     }
136 
137     public boolean isEmpty() {
138         return options.isEmpty() && criterionEntries.isEmpty() && orderEntries.isEmpty() && (offset == null || offset == 0) && (limit == null || limit == 0);
139     }
140 
141     public boolean hasCustomAttributes() {
142         return !this.customAttributes.isEmpty();
143     }
144 
145     public Map<String, String> getCustomAttributes() {
146         return this.customAttributes;
147     }
148 
149     @Override
150     public String toString() {
151         StringBuilder sb = new StringBuilder(128);
152         for (Criterion c : criterionEntries) {
153             if (sb.length() > 0) {
154                 sb.append(" and ");
155             }
156             sb.append(c);
157         }
158 
159         if (!orderEntries.isEmpty()) {
160             if (sb.length() > 0) {
161                 sb.append(" ");
162             }
163             sb.append("order by ").append(Strings.collectionToDelimitedString(orderEntries, ", "));
164         }
165 
166         if (offset != null) {
167             if (sb.length() > 0) {
168                 sb.append(" ");
169             }
170             sb.append("offset ").append(offset);
171         }
172 
173         if (limit != null) {
174             if (sb.length() > 0) {
175                 sb.append(" ");
176             }
177             sb.append("limit ").append(limit);
178         }
179 
180         if (!options.isEmpty() && options instanceof Expandable) {
181             Expandable expandable = (Expandable)options;
182             if (sb.length() > 0) {
183                 sb.append(" ");
184             }
185             sb.append("expand ").append(Strings.collectionToDelimitedString(expandable.getExpansions(), ", "));
186         }
187 
188         if(hasCustomAttributes()){
189             if (sb.length() > 0) {
190                 sb.append(" ");
191             }
192             sb.append("custom attributes: ").append(getCustomAttributes().toString());
193         }
194 
195         return sb.toString();
196     }
197 
198 }
