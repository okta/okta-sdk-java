16  package com.okta.sdk.resource;
17  
18  import com.okta.sdk.error.Error;
19  import com.okta.sdk.error.ErrorCause;
20  import com.okta.sdk.lang.Assert;
21  import com.okta.sdk.lang.Collections;
22  import com.okta.sdk.lang.Strings;
23  
24  import java.util.List;
25  import java.util.Map;
26  
27  /**
28   * A Runtime exception typically thrown when the remote server returns a non 20x response.
29   * @since 0.5.0
30   */
31  public class ResourceException extends RuntimeException implements Error {
32  
33      private final Error error;
34  
35      /**
36       * Ensures the message used for the exception (i.e. exception.getMessage()) reports the {@code developerMessage}
37       * returned by the Okta API Server.  The regular Okta response body {@code message} field is targeted
38       * at applicadtion end-users that could very likely be non-technical.  Since an exception should be helpful to
39       * developers, it is better to show a more technical message.
40       * <p/>
41       * Added as a fix for <a href="https://github.com/stormpath/stormpath-sdk-java/issues/28">Issue #28</a>.
42       *
43       * @param error the response Error. Cannot be null.
44       * @return {@code error.getDeveloperMessage()}
45       */
46      private static String buildExceptionMessage(Error error) {
47          Assert.notNull(error, "Error argument cannot be null.");
48          StringBuilder sb = new StringBuilder();
49          sb.append("HTTP ").append(error.getStatus())
50            .append(", Okta ").append(error.getCode())
51            .append(" (").append(error.getMessage());
52  
53          // if there is only one cause (most common) just include it, otherwise show the cause count
54          int causeCount = Collections.size(error.getCauses());
55          if (causeCount == 1) {
56              sb.append(" - ").append(error.getCauses().get(0).getSummary());
57          } else if (causeCount > 1) {
58              sb.append(" - '").append(causeCount).append(" causes'");
59          }
60          sb.append(")");
61  
62          String errorId = error.getId();
63          if (Strings.hasText(errorId)) {
64              sb.append(", ErrorId ").append(errorId);
65          }
66          return sb.toString();
67      }
68  
69      public ResourceException(Error error) {
70          super(buildExceptionMessage(error));
71          this.error = error;
72      }
73  
74      @Override
75      public int getStatus() {
76          return error.getStatus();
77      }
78  
79      /**
80       * Get the Okta Error Code
81       * Check http://docs.okta.com/errors/ for the list of Okta Error Codes
82       * @return the code of the error
83       */
84      @Override
85      public String getCode() {
86          return error.getCode();
87      }
88  
89  
90      @Override
91      public String getId() {
92          return error.getId();
93      }
94  
95      @Override
96      public List<ErrorCause> getCauses() {
97          return error.getCauses();
98      }
99  
100     @Override
101     public Map<String, List<String>> getHeaders() {
102         return error.getHeaders();
103     }
104 
105     public Error getOktaError() {
106         return this.error;
107     }
108 
109     /**
110      * Returns the underlying REST {@link Error} returned from the Okta API server.
111      * <p/>
112      * Because this class's {@link #getMessage() getMessage()} value returns a developer-friendly message to help you
113      * debug when you see stack traces, you might want to acquire the underlying {@code Error} to show an end-user
114      * the simpler end-user appropriate error message.  The end-user error message is non-technical in nature - as a
115      * convenience, you can show this message directly to your application end-users.
116      * <p/>
117      * For example:
118      * <pre>
119      * try {
120      *
121      *     //something that causes a ResourceException
122      *
123      * } catch (ResourceException re) {
124      *
125      *     String endUserMessage = re.getError().getMessage();
126      *
127      *     warningDialog.setText(endUserMessage);
128      * }
129      * </pre>
130      *
131      * @return the underlying REST {@link Error} resource representation returned from the Okta API server.
132 
133      *
134         public Error getError() {
135         return this.error;
136     }*/
137 }
